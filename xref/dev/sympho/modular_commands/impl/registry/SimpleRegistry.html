23  /**
24   * A simple, directly-mapped registry implementation that does not support overrides. The
25   * (partial) exception to this are aliases, where a command <i>may</i> have an alias that
26   * matches the regular invocation of another command (the regular invocation will always
27   * have precedence over the alias). It <i>may not</i>, however, have the same alias as
28   * another command.
29   *
30   * @version 1.0
31   * @since 1.0
32   */
33  public final class SimpleRegistry implements Registry {
34  
35      /** The logger. */
36      private static final Logger LOGGER = LoggerFactory.getLogger( SimpleRegistry.class );
37  
38      /** The commands registered to this registry. */
39      private final Map<String, Command<?>> commands = new ConcurrentHashMap<>();
40      /** The known invocations. */
41      private final Map<Invocation, Command<?>> invocations = new ConcurrentHashMap<>();
42      /** The known alias invocations. */
43      private final Map<Invocation, Command<?>> aliasInvocations = new ConcurrentHashMap<>();
44  
45      /**
46       * Creates an empty registry.
47       */
48      public SimpleRegistry() {}
49  
50      @Override
51      public <H extends Handlers> @Nullable Command<? extends H> findCommand( 
52              final Invocation invocation, final Class<H> type ) {
53  
54          Command<?> found = invocations.get( invocation );
55          if ( found == null ) {
56              found = aliasInvocations.get( invocation );
57          } 
58          return found == null ? null : InvocationUtils.checkType( found, type );
59  
60      }
61  
62      @Override
63      @SuppressWarnings( "return" ) // https://github.com/typetools/checker-framework/issues/5237
64      public <H extends Handlers> Collection<Command<? extends H>> getCommands( 
65              final Class<H> type ) {
66  
67          return commands.values().stream()
68                  .map( c -> ( Command<? extends H> ) InvocationUtils.checkType( c, type ) )
69                  .filter( Objects::nonNull )
70                  .collect( Collectors.toUnmodifiableList() );
71  
72      }
73  
74      @Override
75      public @Nullable Command<?> getCommand( final String id ) {
76  
77          return commands.get( id );
78  
79      }
80  
81      @Override
82      public synchronized boolean registerCommand( final Command<?> command )
83              throws IllegalArgumentException {
84  
85          CommandUtils.validateCommand( command );
86  
87          final String id = command.id();
88  
89          LOGGER.info( "Registering command {}", id );
90  
91          final Invocation invocation = command.invocation();
92          final Set<Invocation> aliases = command.handlers() instanceof MessageHandlers
93                  ? command.aliasInvocations() : Collections.emptySet();
94  
95          if ( invocations.containsKey( invocation ) 
96                  || !Collections.disjoint( aliases, aliasInvocations.keySet() ) ) {
97              return false;
98          }
99  
100         commands.put( id, command );
101         invocations.put( invocation, command );
102         for ( final Invocation alias : aliases ) {
103             aliasInvocations.put( alias, command );
104         }
105 
106         return true;
107 
108     }
109 
110     @Override
111     public synchronized @Nullable Command<?> removeCommand( final String id ) {
112 
113         final Command<?> command = commands.remove( id );
114         if ( command == null ) {
115             return null;
116         } else {
117             invocations.remove( command.invocation() );
118             if ( command.handlers() instanceof MessageHandlers ) {
119                 command.aliasInvocations().stream().forEach( aliasInvocations::remove );
120             }
121             return command;
122         }
123 
124     }
125     
126 }
