94      /**
95       * Assigns arguments to the corresponding parameters.
96       *
97       * @param <T>        The raw argument type.
98       * @param parameters The parameters.
99       * @param arguments  The arguments to assign.
100      * @param toString   The function to use to convert an argument into an
101      *                   identifier for error messages.
102      * @return The arguments, keyed by the corresponding parameter name.
103      * @throws ResultException if there is a mismatch.
104      */
105     @SideEffectFree
106     private static <T> Map<String, T> assign( final List<? extends Parameter<?>> parameters,
107             final List<T> arguments, final Function<T, String> toString ) throws ResultException {
108 
109         if ( arguments.size() > parameters.size() ) {
110 
111             final var extra = arguments.stream()
112                     .skip( parameters.size() )
113                     .map( toString )
114                     .toList();
115             throw new ResultException( new CommandFailureArgumentExtra( extra ) );
116 
117         }
118 
119         return Streams.zip( 
120                     parameters.stream().map( Parameter::name ),
121                     arguments.stream(), 
122                     Map::entry 
123                 )
124                 .collect( Collectors.toMap( 
125                     Entry::getKey, 
126                     Entry::getValue 
127                 ) );
128 
129     }
130 
131     /**
132      * Adjusts the args received so that trailing arguments are merged into the
133      * final argument, if the final parameter is a string and allows merging.
134      * 
135      * <p>This is to allow multi-word arguments without needing to use quotation marks.
136      *
137      * @param parameters The command parameters.
138      * @param args       The received raw arguments.
139      * @return The adjusted raw arguments.
140      */
141     @SideEffectFree
142     private static Flux<String> adjustArgs( final List<Parameter<?>> parameters,
143             final Iterator args ) {
144 
145         if ( parameters.isEmpty() ) {
146             return Flux.empty();
147         }
148 
149         final var lastIdx = parameters.size() - 1;
150         final var last = parameters.get( lastIdx ).parser();
151         final var merge = last instanceof StringParser<?> p && p.allowMerge();
152         final var index = new AtomicInteger( 0 );
153 
154         return Flux.generate( sink -> {
155 
156             if ( !args.hasNext() ) {
157                 sink.complete();
158             } else if ( merge && index.getAndIncrement() == lastIdx ) {
159                 sink.next( args.remainder() );
160                 sink.complete();
161             } else {
162                 sink.next( args.next() );
163             }
164 
165         } );
166 
167     }

169     @Override
170     protected Mono<Void> initArgs() {
171 
172         return Mono.defer( () -> {
173 
174             final var inputParams = command.parameters().stream()
175                     .filter( p -> p.parser() instanceof InputParser<?, ?> ).toList();
176             final var attachmentParams = command.parameters().stream()
177                     .filter( p -> p.parser() instanceof AttachmentParser<?> ).toList();
178 
179             final var attachments = event().getMessage().getAttachments();
180             this.attachmentArgs = assign( attachmentParams, attachments, Attachment::getFilename );
181 
182             return adjustArgs( inputParams, arguments )
183                     .collectList()
184                     .map( Collections::unmodifiableList )
185                     .doOnNext( args -> {
186                         this.inputArgsList = args;
187                     } )
188                     .map( args -> assign( inputParams, args, Function.identity() ) )
189                     .map( Collections::unmodifiableMap )
190                     .doOnNext( args -> { 
191                         this.inputArgs = args; 
192                     } )
193                     .then();
194 
195         } );
196 
197     }

219         return Objects.requireNonNullElse( attachmentArgs, Collections.emptyMap() );
220     }

222     /**
223      * Retrieves an input argument and performs initial parsing on it.
224      *
225      * @param <P> The raw value type.
226      * @param name The parameter name.
227      * @param parser The parser to use.
228      * @return The pre-parsed raw argument. Will be empty if the argument
229      *         is missing.
230      */
231     private <P extends @NonNull Object> Mono<P> parse( final String name,
232             final RawParser<P> parser ) {
233 
234         final var raw = getInputArgs().get( name );
235         return raw == null ? Mono.empty() : parser.parse( raw );
236 
237     }

273         return parse( name, switch ( type ) {
274             case ANY -> RawParser.SNOWFLAKE;
275             case CHANNEL -> RawParser.channelId( this );
276             case MESSAGE -> RawParser.messageId( this );
277             case ROLE -> RawParser.roleId( this );
278             case USER -> RawParser.userId( this );
279         } );
