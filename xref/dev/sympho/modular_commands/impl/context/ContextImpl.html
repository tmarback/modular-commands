288     /**
289      * Handles a missing argument according to the parameter specification.
290      *
291      * @param <R> The raw argument type.
292      * @param parameter The parameter whose argument is missing.
293      * @return A Mono that fails with a {@link ResultException} if the parameter is not
294      *         allowed to be missing, and is empty otherwise.
295      * @implNote The error case uses lazy instantiation. Use without 
296      *           {@link Mono#defer(java.util.function.Supplier)}.
297      */
298     @SideEffectFree
299     private <R extends @NonNull Object> Mono<R> handleMissingArgument( 
300             final Parameter<?> parameter ) {
301 
302         if ( parameter.required() ) {
303             return Mono.error( () -> new ResultException( 
304                     new CommandFailureArgumentMissing( parameter ) 
305             ) );
306         } else {
307             return Mono.empty();
308         }
309 
310     }

330     /**
331      * Parses an argument.
332      *
333      * @param <R> The raw argument type.
334      * @param <T> The parsed argument type.
335      * @param parameter The parameter specification.
336      * @param getter The function to use to get the raw argument.
337      * @param parser The parser to use.
338      * @return A Mono that issues the parsed argument. May fail with a 
339      *         {@link ResultException} if the raw value is invalid or missing 
340      *         (and required), and may be empty if no value.
341      */
342     @SideEffectFree
343     @SuppressWarnings( { "conditional", "return" } ) // Weird Optional stuff
344     private <R extends @NonNull Object, T extends @NonNull Object> Mono<T> parseArgument(
345             final Parameter<T> parameter,
346             final Function<String, Mono<R>> getter,
347             final ArgumentParser<R, T> parser 
348     ) {
349 
350         return getter.apply( parameter.name() )
351                 .switchIfEmpty( handleMissingArgument( parameter ) )
352                 .map( parser::validateRaw )
353                 // Note flatMap automatically packs exceptions thrown by parse()
354                 .flatMap( raw -> parser.parse( this, raw ) )
355                 .switchIfEmpty( Mono.justOrEmpty( parameter.defaultValue() ) )
356                 .onErrorMap( InvalidArgumentException.class, 
357                         e -> wrapInvalidParam( parameter, e ) 
358                 );
359 
360     }

387     /**
388      * Parses an argument.
389      *
390      * @param <T> The parsed argument type.
391      * @param parameter The parameter to parse.
392      * @return A Mono that issues the parsed argument. May fail with a 
393      *         {@link ResultException} if the raw value is invalid or missing 
394      *         (and required), and may be empty if no value.
395      */
396     @SideEffectFree
397     @SuppressWarnings( { "JavadocMethod", "unchecked" } ) // Ignore undeclared exception
398     private <T extends @NonNull Object> Mono<T> parseArgument( final Parameter<T> parameter ) {
399 
400         // Note: Cannot use <T> directly in the instanceof because Generics
401         // is horribly limited and can't even realize that an ArgumentParser<?, T> that is an 
402         // instance of AttachmentParser is, by definition, an instance of AttachmentParser<T> 
403         // (same for other types).
404         final var parser = parameter.parser();
405         if ( parser instanceof AttachmentParser<?> p ) {
406             return parseArgument( parameter, this::getAttachmentArgument, 
407                     ( AttachmentParser<T> ) p );
408         } else if ( parser instanceof StringParser<?> p ) {
409             return parseArgument( parameter, this::getStringArgument, ( StringParser<T> ) p );
410         } else if ( parser instanceof BooleanParser<?> p ) {
411             return parseArgument( parameter, this::getBooleanArgument, ( BooleanParser<T> ) p );
412         } else if ( parser instanceof IntegerParser<?> p ) {
413             return parseArgument( parameter, this::getIntegerArgument, ( IntegerParser<T> ) p );
414         } else if ( parser instanceof FloatParser<?> p ) {
415             return parseArgument( parameter, this::getFloatArgument, ( FloatParser<T> ) p );
416         } else if ( parser instanceof SnowflakeParser<?> p ) {
417             return parseArgument( parameter, n -> getSnowflakeArgument( n, p.type() ), 
418                     ( SnowflakeParser<T> ) p );
419         } else if ( parser instanceof UserArgumentParser<?> p ) {
420             return parseArgument( parameter, this::getUserArgument, 
421                     ( UserArgumentParser<T> ) p );
422         } else if ( parser instanceof RoleArgumentParser<?> p ) {
423             return parseArgument( parameter, this::getRoleArgument, 
424                     ( RoleArgumentParser<T> ) p );
425         } else if ( parser instanceof ChannelArgumentParser<?, ?> p ) {
426             return parseArgument( parameter, ( ChannelArgumentParser<?, T> ) p );
427         } else if ( parser instanceof MessageArgumentParser<?> p ) {
428             return parseArgument( parameter, this::getMessageArgument, 
429                     ( MessageArgumentParser<T> ) p );
430         } else { // Should never happen
431             throw new IllegalArgumentException( "Unrecognized parser type: " + parser.getClass() );
432         }
433 
434     }

436     /**
437      * Processes an argument.
438      *
439      * @param <T> The parsed argument type.
440      * @param parameter The parameter specification.
441      * @return A Mono that issues the parameter name and the processed argument. May fail 
442      *         with a {@link ResultException} if the raw value is invalid or is missing 
443      *         (and is required).
444      */
445     @SideEffectFree
446     @SuppressWarnings( { 
447             "return", // Bug with type inference in checker
448             "optional.parameter" // TODO: Weird interaction with generics?
449     } ) 
450     private <T extends @NonNull Object> Mono<? extends Entry<String, ? extends Argument<T>>>
451             processArgument( final Parameter<T> parameter ) {
452 
453         return parseArgument( parameter )
454                 .singleOptional()
455                 .map( v -> new Argument<>( parameter, v.orElse( null ) ) )
456                 .map( a -> Map.entry( parameter.name(), a ) )
457                 .doOnError( t -> {
458                     if ( t instanceof ResultException ex ) {
459                         LOGGER.trace( "Arg processing aborted: {}", ex.getResult() );
460                     } else {
461                         LOGGER.error( "Failed to process argument", t );
462                     }
463                 } );
464 
465     }

506      * @throws IllegalArgumentException if there is no parameter with the given name.

508     @Pure
509     private Argument<?> getArgument( final String name ) 
510             throws IllegalStateException, IllegalArgumentException {
511 
512         if ( arguments == null ) {
513             throw new IllegalStateException( "Context not loaded yet" );
514         }
515 
516         final var arg = arguments.get( name );
517         if ( arg == null ) {
518             throw new IllegalArgumentException( String.format( "No parameter named '%s'", name ) );
519         } else {
520             return arg;
521         }
522 
523     }

534     @Override
535     public <T extends @NonNull Object> @Nullable T getArgument( 
536             final @FindDistinct Parameter<? extends T> parameter ) throws IllegalArgumentException {
537 
538         final var argument = getArgument( parameter.name() );
539         if ( argument.parameter() == parameter ) {
540             // Type is determined by the original parameter, so if the given parameter is the
541             // same as the original parameter, the type trivially is the same
542             @SuppressWarnings( "unchecked" )
543             final Argument<T> arg = ( Argument<T> ) argument;
544             return arg.value();
545         } else {
546             throw new IllegalArgumentException( 
547                     "Parameter does not match definition: " + parameter );
548         }

623     /**
624      * Performs the initialize operation.
625      *
626      * @param observations The observation registry to use.
627      */
628     private void doInitialize( final ObservationRegistry observations ) {
629 
630         LOGGER.trace( "Initializing context" );
631 
632         // Prepare load
633         // cache() for idempotence and to prevent cancelling
634         this.loadResult = Mono.just( observations ).flatMap( this::doLoad ).cache();
635 
636         // Initialize reply manager
637         this.reply = new ReplyManagerWrapper( makeReplyManager() );
638 
639         // Create delegate access validator
640         this.access = accessManager.validator( this );
641 
642         LOGGER.trace( "Context initialized" );
643 
644     }

646     @Override
647     public Mono<Void> initialize( final ObservationRegistry observations ) {
648 
649         if ( initialized.getAndSet( true ) ) {
650             return initializeLatch.await(); // Already initializing
651         }
652 
653         LOGGER.trace( "Initializing context" );
654 
655         // Prepare load
656         // cache() for idempotence and to prevent cancelling
657         this.loadResult = Mono.just( observations ).flatMap( this::doLoad ).cache();
658         this.reply = new ReplyManagerWrapper( makeReplyManager() );
659 
660         return Mono.fromRunnable( () -> doInitialize( observations ) )
661                 .then()
662                 .doOnSuccess( v -> initializeLatch.countDown() )
663                 .doOnError( initializeLatch::fail )
664                 .doOnError( t -> LOGGER.error( "Failed to initialize", t ) )
665                 .checkpoint( METRIC_NAME_INITIALIZE )
666                 .name( METRIC_NAME_INITIALIZE )
667                 .transform( this::addTags )
668                 .tap( Micrometer.observation( observations ) )
669                 .cache(); // Prevent cancelling
670 
671     }

673     /**
674      * Performs the {@link #load()} operation.
675      *
676      * @param observations The registry to use for observations.
677      * @return The result.
678      * @see #load()
679      */
680     @SuppressWarnings( "assignment" ) // Not detecting null type bounds for some reason
681     public Mono<CommandResult> doLoad( final ObservationRegistry observations ) {
682 
683         LOGGER.trace( "Loading context" );
684 
685         final var init = Mono.defer( () -> initArgs() 
686                 .checkpoint( METRIC_NAME_ARGUMENT_INIT )
687                 .name( METRIC_NAME_ARGUMENT_INIT )
688                 .transform( this::addTags )
689                 .tap( Micrometer.observation( observations ) )
690         );
691 
692         final var parse = Mono.defer( () -> Flux.fromIterable( command.parameters() )
693                 .flatMap( p -> processArgument( p )
694                         .checkpoint( METRIC_NAME_ARGUMENT_PARSE_ONE )
695                         .name( METRIC_NAME_ARGUMENT_PARSE_ONE )
696                         .transform( this::addTags )
697                         .tag( METRIC_TAG_PARAMETER, p.name() )
698                         .tap( Micrometer.observation( observations ) )
699                 )
700                 .collectMap( Entry::getKey, Entry::getValue )
701                 .doOnNext( args -> {
702                     this.arguments = args;
703                 } )
704                 .checkpoint( METRIC_NAME_ARGUMENT_PARSE_ALL )
705                 .name( METRIC_NAME_ARGUMENT_PARSE_ALL )
706                 .transform( this::addTags )
707                 .tap( Micrometer.observation( observations ) )
708         );
709 
710         return init.then( parse )
711                 .checkpoint( METRIC_NAME_LOAD )
712                 .name( METRIC_NAME_LOAD )
713                 .transform( this::addTags )
714                 .tap( Micrometer.observation( observations ) )
715                 .doOnSuccess( v -> LOGGER.trace( "Context loaded" ) )
716                 .doOnError( t -> {
717                     if ( t instanceof ResultException ex ) {
718                         LOGGER.trace( "Load aborted: {}", ex.getResult() );
719                     } else {
720                         LOGGER.error( "Failed to load", t );
721                     }
722                 } )
723                 .then( Mono.empty().cast( CommandResult.class ) )
724                 .onErrorResume( ResultException.class, e -> Mono.just( e.getResult() ) );
725 
726     }
727 
728     @Override
729     public Mono<CommandResult> load() {
730 
731         if ( loadResult == null ) {
732             throw new IllegalStateException( "Called load() before initialize()" );
733         } else {
734             return loadResult;
735         }
736 
737     }
