18  /**
19   * Default implementation of an interaction-based command.
20   *
21   * @param <H> The handler type.
22   * @param scope The scope that the command is defined in.
23   * @param callable If the command may be directly invoked by users.
24   * @param parent The parent of the command.
25   * @param name The name of the command.
26   * @param aliases The aliases that may also invoke the command.
27   * @param displayName The display name of the command.
28   * @param description The description of the command.
29   * @param parameters The command parameters, in the order that they should be provided
30   *                   by the user.
31   * @param requiredGroup The group that a user must have access for in order to invoke 
32   *                      this command.
33   * @param skipGroupCheckOnInteraction Whether group access checking should be skipped.
34   * @param requireParentGroups Whether a user invoking this command must also have access 
35   *                            to the groups necessary to invoke its parent command(s).
36   * @param nsfw Whether this command can only be invoked in a NSFW channel.
37   * @param privateReply Whether this command's response is sent in a way that only the 
38   *                     invoking user can see.
39   * @param ephemeralReply The type of ephemeral response to use, if any.
40   * @param inheritSettings Whether the command settings should be inherited from the parent 
41   *                        command (ignoring the values provided by this command).
42   * @param invokeParent Whether to invoke the parent as part of normal execution.
43   * @param handlers The handler to use for processing an invocation of the command.
44   * @version 1.0
45   * @since 1.0
46   */
47  public record CommandImpl<H extends Handlers>(
48          Scope scope,
49          boolean callable,
50          Invocation parent,
51          @MatchesRegex( Command.NAME_REGEX ) String name,
52          Set<@MatchesRegex( Command.NAME_REGEX ) String> aliases,
53          @MatchesRegex( Command.DISPLAY_NAME_REGEX ) String displayName,
54          @MatchesRegex( Command.DESCRIPTION_REGEX ) String description,
55          List<Parameter<?>> parameters,
56          Group requiredGroup,
57          boolean skipGroupCheckOnInteraction,
58          boolean requireParentGroups,
59          boolean nsfw,
60          boolean privateReply,
61          EphemeralType ephemeralReply,
62          boolean inheritSettings,
63          boolean invokeParent,
64          H handlers
65  ) implements Command<H> {
66  
67      /**
68       * Initializes a new instance.
69       *
70       * @param scope The scope that the command is defined in.
71       * @param callable If the command may be directly invoked by users.
72       * @param parent The parent of the command.
73       * @param name The name of the command.
74       * @param aliases The aliases that may also invoke the command.
75       * @param displayName The display name of the command.
76       * @param description The description of the command.
77       * @param parameters The command parameters, in the order that they should be provided
78       *                   by the user.
79       * @param requiredGroup The group that a user must have access for in order to invoke 
80       *                      this command.
81       * @param skipGroupCheckOnInteraction Whether group access checking should be skipped.
82       * @param requireParentGroups Whether a user invoking this command must also have access 
83       *                            to the groups necessary to invoke its parent command(s).
84       * @param nsfw Whether this command can only be invoked in a NSFW channel.
85       * @param privateReply Whether this command's response is sent in a way that only the 
86       *                     invoking user can see.
87       * @param ephemeralReply The type of ephemeral response to use, if any.
88       * @param inheritSettings Whether the command settings should be inherited from the parent 
89       *                        command (ignoring the values provided by this command).
90       * @param invokeParent Whether to invoke the parent as part of normal execution.
91       * @param handlers The handler to use for processing an invocation of the command.
92       */
93      @SideEffectFree
94      @SuppressWarnings( "checkstyle:parameternumber" )
95      public CommandImpl(
96              final Scope scope,
97              final boolean callable,
98              final Invocation parent,
99              final @MatchesRegex( Command.NAME_REGEX ) String name,
100             final Set<@MatchesRegex( Command.NAME_REGEX ) String> aliases,
101             final @MatchesRegex( Command.DISPLAY_NAME_REGEX ) String displayName,
102             final @MatchesRegex( Command.DESCRIPTION_REGEX ) String description,
103             final List<Parameter<?>> parameters,
104             final Group requiredGroup,
105             final boolean skipGroupCheckOnInteraction,
106             final boolean requireParentGroups,
107             final boolean nsfw,
108             final boolean privateReply,
109             final EphemeralType ephemeralReply,
110             final boolean inheritSettings,
111             final boolean invokeParent,
112             final H handlers
113     ) {
114 
115         this.scope = Objects.requireNonNull( scope );
116         this.callable = callable;
117         this.parent = CommandUtils.validateParent( parent );
118         this.name = CommandUtils.validateName( name );
119         this.aliases = CommandUtils.validateAliases( aliases );
120         this.displayName = CommandUtils.validateDisplayName( displayName );
121         this.description = CommandUtils.validateDescription( description );
122         this.parameters = CommandUtils.validateParameters( parameters );
123         this.requiredGroup = CommandUtils.validateGroup( requiredGroup );
124         this.skipGroupCheckOnInteraction = skipGroupCheckOnInteraction;
125         this.requireParentGroups = requireParentGroups;
126         this.nsfw = nsfw;
127         this.privateReply = privateReply;
128         this.ephemeralReply = Objects.requireNonNull( ephemeralReply );
129         this.inheritSettings = inheritSettings;
130         this.invokeParent = invokeParent;
131         this.handlers = CommandUtils.validateHandlers( handlers );
132 
133     }
