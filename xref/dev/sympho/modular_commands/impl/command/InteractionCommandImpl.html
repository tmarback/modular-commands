16  /**
17   * Default implementation of an interaction-based command.
18   *
19   * @param scope The scope that the command is defined in.
20   * @param callable If the command may be directly invoked by users.
21   * @param parent The parent of the command.
22   * @param name The name of the command.
23   * @param displayName The display name of the command.
24   * @param description The description of the command.
25   * @param parameters The command parameters, in the order that they should be provided
26   *                   by the user.
27   * @param requiredDiscordPermissions he built-in permissions that a user should have in
28   *                                   order to execute the command.
29   * @param requireParentPermissions Whether a user invoking this command must also have 
30   *                                 the permissions to invoke its parent command.
31   * @param nsfw Whether this command can only be invoked in a NSFW channel.
32   * @param botOwnerOnly Whether this command can only be invoked by the owner of the bot.
33   * @param serverOwnerOnly Whether this command can only be invoked by the owner of the server.
34   * @param privateReply Whether this command's response is sent in a way that only the 
35   *                     invoking user can see.
36   * @param inheritSettings Whether the command settings should be inherited from the parent 
37   *                        command (ignoring the values provided by this command).
38   * @param invokeParent Whether to invoke the parent as part of normal execution.
39   * @param invocationHandler The handler to use for processing an invocation of the command.
40   * @param resultHandlers The handlers to use for processing the result, in order.
41   * @version 1.0
42   * @since 1.0
43   */
44  public record InteractionCommandImpl(
45          Scope scope,
46          boolean callable,
47          Invocation parent,
48          String name,
49          String displayName,
50          String description,
51          List<Parameter<?>> parameters,
52          PermissionSet requiredDiscordPermissions,
53          boolean requireParentPermissions,
54          boolean nsfw,
55          boolean botOwnerOnly,
56          boolean serverOwnerOnly,
57          boolean privateReply,
58          boolean inheritSettings,
59          boolean invokeParent,
60          InteractionInvocationHandler invocationHandler,
61          List<? extends InteractionResultHandler> resultHandlers
62  ) implements InteractionCommand {
63  
64      /**
65       * Initializes a new instance.
66       *
67       * @param scope The scope that the command is defined in.
68       * @param callable If the command may be directly invoked by users.
69       * @param parent The parent of the command.
70       * @param name The name of the command.
71       * @param displayName The display name of the command.
72       * @param description The description of the command.
73       * @param parameters The command parameters, in the order that they should be provided
74       *                   by the user.
75       * @param requiredDiscordPermissions he built-in permissions that a user should have in
76       *                                   order to execute the command.
77       * @param requireParentPermissions Whether a user invoking this command must also have 
78       *                                 the permissions to invoke its parent command.
79       * @param nsfw Whether this command can only be invoked in a NSFW channel.
80       * @param botOwnerOnly Whether this command can only be invoked by the owner of the bot.
81       * @param serverOwnerOnly Whether this command can only be invoked by the owner of the server.
82       * @param privateReply Whether this command's response is sent in a way that only the 
83       *                     invoking user can see.
84       * @param inheritSettings Whether the command settings should be inherited from the parent 
85       *                        command (ignoring the values provided by this command).
86       * @param invokeParent Whether to invoke the parent as part of normal execution.
87       * @param invocationHandler The handler to use for processing an invocation of the command.
88       * @param resultHandlers The handlers to use for processing the result, in order.
89       */
90      @SideEffectFree
91      @SuppressWarnings( "checkstyle:parameternumber" )
92      public InteractionCommandImpl(
93              final Scope scope,
94              final boolean callable,
95              final Invocation parent,
96              final String name,
97              final String displayName,
98              final String description,
99              final List<Parameter<?>> parameters,
100             final PermissionSet requiredDiscordPermissions,
101             final boolean requireParentPermissions,
102             final boolean nsfw,
103             final boolean botOwnerOnly,
104             final boolean serverOwnerOnly,
105             final boolean privateReply,
106             final boolean inheritSettings,
107             final boolean invokeParent,
108             final InteractionInvocationHandler invocationHandler,
109             final List<? extends InteractionResultHandler> resultHandlers
110     ) {
111 
112         this.scope = Objects.requireNonNull( scope );
113         this.callable = callable;
114         this.parent = CommandUtils.validateParent( parent );
115         this.name = CommandUtils.validateName( name );
116         this.displayName = CommandUtils.validateDisplayName( displayName );
117         this.description = CommandUtils.validateDescription( description );
118         this.parameters = CommandUtils.validateParameters( parameters );
119         this.requiredDiscordPermissions = CommandUtils.validateDiscordPermissions( 
120                 requiredDiscordPermissions );
121         this.requireParentPermissions = requireParentPermissions;
122         this.nsfw = nsfw;
123         this.botOwnerOnly = botOwnerOnly;
124         this.serverOwnerOnly = serverOwnerOnly;
125         this.privateReply = privateReply;
126         this.inheritSettings = inheritSettings;
127         this.invokeParent = invokeParent;
128         this.invocationHandler = CommandUtils.validateInvocationHandler( invocationHandler );
129         this.resultHandlers = CommandUtils.validateResultHandlers( resultHandlers );
130 
131     }
