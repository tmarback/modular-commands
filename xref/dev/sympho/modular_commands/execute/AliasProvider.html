3   import java.util.Arrays;
4   import java.util.Collection;
5   import java.util.Map;
6   import java.util.Map.Entry;
7   import java.util.function.UnaryOperator;
8   import java.util.stream.Collectors;
9   
10  import org.checkerframework.checker.nullness.qual.Nullable;
11  import org.slf4j.LoggerFactory;
12  
13  import dev.sympho.modular_commands.api.command.Invocation;
14  import dev.sympho.modular_commands.utils.StringSplitter.Async.Iterator;
15  
16  /**
17   * Provides aliases that apply to message-based commands.
18   *
19   * @version 1.0
20   * @since 1.0
21   */
22  @FunctionalInterface
23  public interface AliasProvider extends UnaryOperator<Iterator> {
24  
25      /**
26       * Applies aliases to the given iterator over received raw tokens.
27       *
28       * @param iter The arg iterator to apply aliases to.
29       * @return The iterator over the equivalent invocation with any aliases applied.
30       * @implSpec Implementations should <b>not</b> consume all elements in the given iterator
31       *           before checking for aliases, both due to performance concerns and due to
32       *           interfering with argument parsing. More specifically, the value of
33       *           {@link Iterator#remainder()} after the aliased portion should be exactly
34       *           the same before or after applying the alias.
35       */
36      @Override
37      Iterator apply( Iterator iter );
38  
39      /**
40       * Creates an alias provider with the given aliases.
41       *
42       * @param aliases The aliases to use in the created provider.
43       * @return The constructed provider.
44       */
45      static AliasProvider of( final Collection<? extends Entry<Invocation, Invocation>> aliases ) {
46  
47          final var map = aliases.stream().collect( 
48                  Collectors.toUnmodifiableMap( Entry::getKey, Entry::getValue ) 
49          );
50  
51          return iter -> {
52  
53              var current = Invocation.of();
54              @Nullable Invocation result = null;
55      
56              while ( iter.hasNext() ) {
57      
58                  final var next = current.child( iter.peek() );
59                  final var alias = map.get( next );
60                  if ( alias == null ) {
61                      break;
62                  }
63      
64                  current = next;
65                  result = alias;
66                  iter.next();
67      
68              }
69      
70              if ( result == null ) {
71                  return iter;
72              }
73      
74              LoggerFactory.getLogger( AliasProvider.class ).debug( 
75                      "Applying alias from {} to {}", current, result 
76              );
77      
78              final var splitter = iter.splitter();
79              final var alias = String.join( splitter.delimiter(), result.chain() );
80              final var aliased = String.join( splitter.delimiter(), alias, iter.remainder() );
81              return splitter.iterate( aliased );
82  
83          };
84  
85      }
86  
87      /**
88       * Creates an alias provider with the given aliases.
89       *
90       * @param aliases The aliases to use in the created provider.
91       * @return The constructed provider.
92       */
93      static AliasProvider of( final Map<Invocation, Invocation> aliases ) {
94  
95          return of( aliases.entrySet() );
96  
97      }
98  
99      /**
100      * Creates an alias provider with the given aliases.
101      *
102      * @param aliases The aliases to use in the created provider.
103      * @return The constructed provider.
104      */
105     @SafeVarargs
106     @SuppressWarnings( "varargs" )
107     static AliasProvider of( final Entry<Invocation, Invocation>... aliases ) {
108 
109         return of( Arrays.asList( aliases ) );
110 
111     }
112 
113     /**
114      * Creates an alias provider with no aliases.
115      *
116      * @return The constructed provider.
117      */
118     static AliasProvider none() {
119 
120         return iter -> iter;
121 
122     }
123     
124 }
