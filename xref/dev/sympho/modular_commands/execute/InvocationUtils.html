40      /**
41       * Extracts an invocation from a sequence of args by performing lookups on the given
42       * registry, while building the corresponding execution chain.
43       * 
44       * <p>After this method returns, the given {@code args} iterator will be positioned
45       * such that the next element is the first argument that did not match a subcommand
46       * (and thus the first proper argument).
47       * 
48       * <p>Matching is performed case-insensitive by converting received args to lowercase
49       * before attempting to match (command names are always lowercase).
50       *
51       * @param <H> The handler type.
52       * @param registry The registry to use for command lookups.
53       * @param args The invocation args.
54       * @param commandType The command type.
55       * @return The detected invocation and the corresponding execution chain.
56       */
57      @SideEffectFree
58      public static <H extends Handlers> Tuple2<Invocation, List<Command<? extends H>>> 
59              parseInvocation( final Registry registry, final SmartIterator<String> args, 
60              final Class<H> commandType ) {
61  
62          final List<Command<? extends H>> chain = new LinkedList<>();
63          Invocation current = Invocation.of();
64          while ( args.hasNext() ) {
65  
66              final var arg = args.peek().toLowerCase( Locale.ROOT ); // Ignore case
67              final var next = current.child( arg );
68              final var command = registry.findCommand( next, commandType );
69              if ( command == null ) {
70                  break;
71              } else {
72                  chain.add( command );
73                  current = current.child( command.name() );
74                  args.next();
75              }
76  
77          }
78          return Tuples.of( current, List.copyOf( chain ) );
79  
80      }

82      /**
83       * Extracts the command being invoked from an execution chain.
84       *
85       * @param <C> The command type.
86       * @param chain The execution chain.
87       * @return The command that was invoked (the last one in the chain).
88       */
89      @Pure
90      public static <C extends Command<?>> C getInvokedCommand( final List<? extends C> chain ) {
91  
92          return chain.get( chain.size() - 1 );
93  
94      }

96      /**
97       * Determines the command in the execution chain that should provide the
98       * invocation settings.
99       *
100      * @param <C> The command type.
101      * @param chain The command chain.
102      * @return The command to take settings from.
103      * @see Command#inheritSettings()
104      */
105     @Pure
106     public static <C extends @NonNull Command<?>> C getSettingsSource( final List<C> chain ) {
107 
108         return Lists.reverse( chain ).stream()
109                 .filter( Predicate.not( Command::inheritSettings ) )
110                 .findFirst()
111                 .orElse( chain.get( 0 ) );
112 
113     }

115     /**
116      * Determines the total set of groups required for an execution chain.
117      *
118      * @param chain The execution chain.
119      * @return The required groups.
120      */
121     @SideEffectFree
122     public static List<Group> accumulateGroups( 
123             final List<? extends Command<?>> chain ) {
124 
125         // https://github.com/typetools/checker-framework/issues/4048
126         @SuppressWarnings( "type.argument" )
127         final var needParent = Lists.reverse( chain ).stream()
128                 .takeWhile( Command::requireParentGroups )
129                 .count();
130 
131         // Need to include the parent of the last command to require parent, unless all
132         // of them do (clamp at chain size)
133         final var take = Math.min( needParent + 1, chain.size() );
134 
135         return chain.stream()
136                 .skip( chain.size() - take )
137                 .map( Command::requiredGroup )
138                 .toList();
139 
140     }

142     /**
143      * Determines if a parameter is always satisfied.
144      *
145      * @param parameter The parameter.
146      * @return If the parameter is always satisfied.
147      */
148     private static boolean satisfied( final Parameter<?> parameter ) {
149 
150         return parameter.required() || parameter.defaultValue() != null;
151 
152     }

154     /**
155      * Determines the sequence that command handlers should be invoked in for
156      * the given chain.
157      *
158      * @param <C> The command type.
159      * @param chain The invocation chain.
160      * @return The commands, in the order that their handlers must be executed.
161      *         Some of the commands in the chain may be missing if they do not
162      *         need to be executed.
163      * @throws InvalidChainException if the command chain is incompatible.
164      * @see Command#invokeParent()
165      */
166     @SideEffectFree
167     public static <C extends @NonNull Command<?>> List<C> handlingOrder( final List<C> chain ) {
168 
169         final List<C> commands = new LinkedList<>();
170         final var it = chain.listIterator( chain.size() );
171 
172         C source = it.previous();
173         commands.add( source );
174 
175         final C target = source;
176         final Set<String> satisfiedParameters = source.parameters().stream()
177                 .filter( InvocationUtils::satisfied )
178                 .map( Parameter::name )
179                 .collect( Collectors.toSet() );
180         
181         @SuppressWarnings( "rawtypes" ) // Generics is dumb with Class<>
182         final Map<String, Class<? extends Parameter>> parameterTypes = source.parameters().stream()
183                 .collect( Collectors.toMap( Parameter::name, Parameter::getClass ) );
184 
185         while ( it.hasPrevious() && source.invokeParent() ) {
186 
187             source = it.previous();
188             // Validate command compatibility
189             for ( final var p : source.parameters() ) {
190 
191                 final var receivedType = parameterTypes.get( p.name() );
192                 // Validate type
193                 if ( receivedType != null && receivedType != p.getClass() ) {
194                     throw new InvalidChainException( source, target, String.format( 
195                             "Parameter %s is of type %s in parent but type %s in child", 
196                             p.name(), p.getClass().getSimpleName(), receivedType.getSimpleName()
197                     ) );
198                 }
199                 // Validate presence
200                 if ( p.required() && !satisfiedParameters.contains( p.name() ) ) {
201                     throw new InvalidChainException( source, target, String.format( 
202                             "Parameter %s is required in parent but may not be present in child", 
203                             p.name()
204                     ) );
205                 }
206 
207             }
208 
209             commands.add( 0, source );
210 
211         }
212 
213         return List.copyOf( commands );
214 
215     }

217     /**
218      * Determines if a command has handlers compatible with the given type.
219      *
220      * @param <H> The handler type.
221      * @param command The command to check.
222      * @param type The handler type.
223      * @return The command, or {@code null} if the command's handlers are not
224      *         compatible.
225      */
226     @Pure
227     @SuppressWarnings( "unchecked" )
228     public static <H extends Handlers> @Nullable Command<? extends H> checkType(
229             final Command<?> command, final Class<H> type
230     ) {
231 
232         return type.isInstance( command.handlers() ) 
233                 ? ( Command<? extends H> ) command 
234                 : null;
235 
236     }
