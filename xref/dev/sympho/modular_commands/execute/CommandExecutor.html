1   package dev.sympho.modular_commands.execute;
2   
3   import dev.sympho.modular_commands.api.registry.Registry;
4   import discord4j.core.GatewayDiscordClient;
5   import reactor.core.Disposable;
6   import reactor.core.publisher.Mono;
7   import reactor.util.annotation.Nullable;
8   
9   /**
10   * Executor that receives events and invokes triggered commands as appropriate.
11   *
12   * @version 1.0
13   * @since 1.0
14   */
15  public class CommandExecutor {
16  
17      /** The event processing pipeline. */
18      private final Mono<Void> pipeline;
19  
20      /** The current processing task. */
21      private @Nullable Disposable live;
22  
23      /**
24       * Creates a new instance.
25       *
26       * @param client The client to receive events from.
27       * @param registry The registry to use to look up commands.
28       * @param builder The builder to use for constructing the event processing pipeline.
29       */
30      protected CommandExecutor( final GatewayDiscordClient client, final Registry registry,
31              final PipelineBuilder<?, ?, ?, ?, ?> builder ) {
32  
33          pipeline = builder.buildPipeline( client, registry );
34  
35      }
36  
37      /* Public interface methods */
38  
39      /**
40       * Start receiving events and executing commands.
41       *
42       * @return {@code true} if the call started the executor, 
43       *         {@code false} if it was already started.
44       */
45      public synchronized boolean start() {
46  
47          if ( live != null ) {
48              return false;
49          }
50  
51          live = pipeline.subscribe();
52          return true;
53  
54      }
55  
56      /**
57       * Stops receiving events and executing commands.
58       *
59       * @return {@code true} if the call stopped the executor, 
60       *         {@code false} if it was already stopped.
61       */
62      public synchronized boolean stop() {
63  
64          if ( live == null ) {
65              return false;
66          }
67  
68          live.dispose();
69          live = null;
70          return true;
71  
72      }
73      
74  }
