1   package dev.sympho.modular_commands.execute;
2   
3   import org.checkerframework.dataflow.qual.SideEffectFree;
4   
5   import dev.sympho.modular_commands.api.permission.AccessValidator;
6   import dev.sympho.modular_commands.api.permission.Group;
7   import discord4j.core.object.entity.Guild;
8   import discord4j.core.object.entity.User;
9   import discord4j.core.object.entity.channel.MessageChannel;
10  import reactor.core.publisher.Mono;
11  
12  /**
13   * A manager that provides access validators for different execution contexts.
14   *
15   * @version 1.0
16   * @since 1.0
17   */
18  @FunctionalInterface
19  public interface AccessManager {
20  
21      /**
22       * Creates an access validator under the context of the given guild, channel, and caller.
23       *
24       * @param guild The guild of the current execution.
25       * @param channel The channel of the current execution.
26       * @param caller The caller that invoked the execution.
27       * @return The appropriate access validator.
28       */
29      AccessValidator validator( Mono<Guild> guild, Mono<MessageChannel> channel, User caller );
30  
31      /**
32       * Creates a manager for which all validators always allow access.
33       * That is, a manager that allows access for any user to any group (effectively disabling
34       * group checking).
35       *
36       * @return The manager.
37       */
38      @SideEffectFree
39      static AccessManager alwaysAllow() {
40  
41          return ( guild, channel, caller ) -> group -> Mono.just( true );
42  
43      }
44  
45      /**
46       * Creates a manager for which all validators always deny access.
47       * That is, a manager that denies access for any user to any group (effectively disabling
48       * any functionality that requires group membership).
49       *
50       * @return The manager.
51       */
52      @SideEffectFree
53      static AccessManager alwaysDeny() {
54  
55          return ( guild, channel, caller ) -> group -> Mono.just( false );
56  
57      }
58  
59      /**
60       * Creates a manager that issues validators that perform simple group memebership checks
61       * (that is, a user has equivalent access to a group if and only if they belong to that
62       * group).
63       *
64       * @return The manager.
65       */
66      @SideEffectFree
67      static AccessManager basic() {
68  
69          return ( guild, channel, caller ) -> group -> group.belongs( guild, channel, caller );
70  
71      }
72  
73      /**
74       * Creates a manager that issues validators that generally perform group membership checks,
75       * but also allows a given group to override any permission in the system (that is, users
76       * that are members of the given group have equivalent permissions to any other group).
77       *
78       * @param overrideGroup The group that overrides any permission check.
79       * @return The manager.
80       */
81      @SideEffectFree
82      static AccessManager overridable( final Group overrideGroup ) {
83  
84          return ( guild, channel, caller ) -> group -> group.belongs( guild, channel, caller )
85                  .filter( Boolean::booleanValue ) // Make empty if not allowed
86                  .switchIfEmpty( Mono.defer( // If not allowed, check override
87                          () -> overrideGroup.belongs( guild, channel, caller ) 
88                  ) )
89                  .defaultIfEmpty( false );
90  
91      }
92      
93  }
