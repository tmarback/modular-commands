1   package dev.sympho.modular_commands.execute;
2   
3   import dev.sympho.modular_commands.api.command.context.CommandContext;
4   import dev.sympho.modular_commands.api.command.result.CommandResult;
5   import io.micrometer.observation.ObservationRegistry;
6   import reactor.core.publisher.Mono;
7   
8   /**
9    * A command context whose values (particularly arguments) are lazy-loaded (that is, are not
10   * loaded until requested by {@link #initialize(ObservationRegistry)} and {@link #load()}).
11   * 
12   * <p>This interface is not involved in the creation of commands in any way. It is only used
13   * internally to decouple initialization/parsing/validation from instantiation and from
14   * eachother.
15   *
16   * @version 1.0
17   * @since 1.0
18   */
19  public interface LazyContext extends CommandContext {
20  
21      /**
22       * Partially initializes internal state, making the context minimally ready for
23       * handling to start.
24       * 
25       * <p>The only parts of the context API that are guaranteed to be ready to use
26       * before this method is called are the ones that retrieve pieces of the invocation
27       * context, that is {@link #getEvent()}, {@link #getCaller()}, {@link #getChannel()}, 
28       * etc, as well as this method itself. All other methods have undefined behavior.
29       * 
30       * <p>After this method is called and the returned mono completes <i>successfully</i>,
31       * the following pieces of the API also become ready to be used:
32       * 
33       * <ul>
34       *   <li>Reply manager</li>
35       *   <li>Access and group checking</li>
36       *   <li>{@link #load()}</li>
37       * </ul>
38       * 
39       * <p>Calling any part of the API other than those listed will continue to result in
40       * undefined behavior until {@link #load()} is called and successfully completes.
41       * 
42       * <p>This method is idempotent; if it is called multiple times, the context will still
43       * be initialized only once, and all the returned Monos will only complete once it has
44       * finished loading.
45       *
46       * @param observations The registry to use for observations.
47       * @return A Mono that completes once the context is initialized.
48       */
49      Mono<Void> initialize( ObservationRegistry observations );
50  
51      /**
52       * Loads remaining internal state, making the context fully ready for use. It must only 
53       * be called after {@link #initialize(ObservationRegistry)} completes <i>successfully</i>, 
54       * otherwise its behavior is undefined.
55       * 
56       * <p>Until this method is called and the returned mono completes <i>successfully</i>,
57       * all methods other than this one (and those specified in 
58       * {@link #initialize(ObservationRegistry)} have undefined behavior.
59       * 
60       * <p>This method is idempotent; if it is called multiple times, the context will still
61       * be loaded only once, and all the returned Monos will only complete once it has
62       * finished loading, with the same result.
63       *
64       * @return A Mono that completes empty once internal values are successfully loaded. 
65       *         If a situation where the invocation should be terminated is encountered,
66       *         it emits the appropriate failure result.
67       * @apiNote This is split from {@link #initialize(ObservationRegistry)} for performance 
68       *          reasons; only a very limited subset of the API is necessary for early processing 
69       *          of the command, and things such as argument parsing can be relatively expensive,
70       *          thus it is beneficial to delay the latter until later in the processing
71       *          pipeline once they are actually needed.
72       */
73      Mono<CommandResult> load();
74      
75  }
