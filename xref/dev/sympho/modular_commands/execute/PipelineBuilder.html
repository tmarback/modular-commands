39  /**
40   * Type responsible for building a command processing pipeline.
41   *
42   * @param <E> The type of event that triggers commands.
43   * @param <CTX> The type of command context.
44   * @param <H> The type of handlers.
45   * @param <I> The iterator type used to traverse the received arguments.
46   * @version 1.0
47   * @since 1.0
48   * @apiNote Note that, for the purposes of this class, there is a distinction between <i>args</i>
49   *          and <i>arguments</i>:
50   * 
51   *          <ul>
52   *              <li><i>args</i> is the raw sequence of strings provided by the triggering event
53   *                  that identifies the command to execute. It may or may not also contain, after
54   *                  the command identifiers, strings that should be parsed into command 
55   *                  <i>arguments</i>.</li>
56   *              <li><i>arguments</i> are the values that satisfy the defined parameters of a
57   *                  command. They may be obtain from <i>args</i> that appear after the command
58   *                  identification, or directly from the event in some way (or both).</li>
59   *          </ul>
60   */
61  public abstract class PipelineBuilder<E extends Event, 
62          CTX extends CommandContext & LazyContext, H extends Handlers, 
63          I extends SmartIterator<String>> {

84      /**
85       * Builds a new pipeline with the given client and registry.
86       *
87       * @param client The client to receive events from.
88       * @param registry The registry to use to look up commands.
89       * @return The built processing pipeline.
90       */
91      public Mono<Void> buildPipeline( final GatewayDiscordClient client, final Registry registry ) {
92  
93          final Flux<E> source = client.on( eventType() )
94                  .filter( this::eventFilter )
95                  .doOnNext( e -> {
96                      LOGGER.trace( "Received event: {}", e );
97                  } );
98          return buildPipeline( source, registry ).retry();
99  
100     }

102     /**
103      * Builds the processing pipeline.
104      *
105      * @param source The event source.
106      * @param registry The registry to use to look up commands.
107      * @return The built processing pipeline.
108      */
109     @SideEffectFree
110     private Mono<Void> buildPipeline( final Flux<E> source, final Registry registry ) {
111 
112         return source.flatMap( event -> parseEvent( event, registry )
113                 .flatMap( this::executeCommand )
114                 .doOnNext( ctx -> {
115                     final CTX context = ctx.getT2();
116                     final CommandResult result = ctx.getT3();
117 
118                     if ( result instanceof CommandErrorException r ) {
119                         final var cause = r.cause();
120                         LOGGER.error( String.format( "Exception while executing command %s", 
121                                 context.getInvocation() ), cause );
122                     } else if ( result instanceof CommandError r ) {
123                         LOGGER.error( "Error while executing command {}: {}", 
124                                 context.getInvocation(), r.message() );
125                     } else {
126                         LOGGER.debug( "Finished command execution {} with result {}", 
127                                 context.getInvocation(), result.getClass().getSimpleName() );
128                         LOGGER.trace( "{} => {}", context.getInvocation(), result );
129                     }
130                 } )
131                 .flatMap( this::handleResult )
132                 .doOnNext( c -> {
133                     LOGGER.warn( "Handling of result of command {} not complete", 
134                             c.getInvocation() );
135                 } )
136                 .onErrorResume( e -> {
137                     LOGGER.error( "Exception thrown within processing pipeline", e );
138                     return Mono.empty();
139                 } )
140         ).doOnError( e -> LOGGER.error( "Fatal error", e ) ).then();
141 
142     }

163     /**
164      * Determines whether a parsed arg list (as given by {@link #parse(Event)}) must always
165      * be fully matched to a command. If {@code true}, the presence of leftover args (as would
166      * be received by {@link #makeContext(Event, Command, Invocation, SmartIterator)}) triggers 
167      * an error for that event (stopping the processing of that particular event, but without 
168      * stopping the pipeline).
169      *
170      * @return {@code true} if the parsed args must be fully matched to a command.
171      *         {@code false} if additional args are allowed.
172      * @apiNote This check is not related to any user-facing functionality, but purely as an
173      *          internal sanity check. If an event provides arguments in a way other than the
174      *          parsed args, with said args being used only to identify the command to execute,
175      *          it would not make sense for extra args to be received.
176      * 
177      *          <p>For example, in a slash command, the comand name is provided separately from
178      *          the arguments, and any subcommands are already labeled as such, so there is no
179      *          need to evaluate any further args when looking up the command. At the same
180      *          time, since slash commands (including subcommands) have to be registered with 
181      *          Discord ahead of time to be callable by users, the only way that a slash command 
182      *          event would provide a subcommand that is not found is if the command was registered
183      *          to Discord without having a corresponding internal handler, without would indicate
184      *          an inconsistency in the command system (the <i>declared</i> commands do not match
185      *          the <i>implemented</i> commands), which should be reported to the administrator
186      *          as a (non-fatal) pipeline error.
187      */
188     // END BUGGED PARAGRAPH

325     /**
326      * Verifies that the invocation is valid for the scope that the command is defined in.
327      *
328      * @param payload The invocation payload.
329      * @return {@code true} if the invocation is valid and the command should be executed.
330      */
331     @Pure
332     private boolean checkScope( 
333             final Tuple4<E, List<Command<? extends H>>, Invocation, I> payload ) {
334 
335         final E event = payload.getT1();
336         final List<Command<? extends H>> chain = payload.getT2();
337 
338         final var command = InvocationUtils.getInvokedCommand( chain );
339 
340         return command.scope() == Scope.GLOBAL || getGuildId( event ).isPresent();
341 
342     }

362     /**
363      * Parses an invocation from an event.
364      * 
365      * <p>Invocations of a command in an incompatible scope or of a command that cannot
366      * be invoked by itself are pre-filtered and will return an empty Mono.
367      *
368      * @param event The event.
369      * @param registry The registry to use for command lookups.
370      * @return A Mono that issues the parsed command, if any. 
371      */
372     @SideEffectFree
373     private Mono<Tuple4<E, List<Command<? extends H>>, Invocation, I>> parseEvent( 
374             final E event, final Registry registry ) {
375 
376         return Mono.just( event )
377                 .map( e -> Tuples.of( e, parse( e ) ) )
378                 .filter( ctx -> ctx.getT2().hasNext() )
379                 .map( ctx -> {
380                     final E e = ctx.getT1();
381                     final I args = ctx.getT2();
382 
383                     LOGGER.trace( "Parsed args {}", args );
384 
385                     final var parsed = InvocationUtils.parseInvocation( 
386                             registry, args, commandType() );
387 
388                     final Invocation invocation = parsed.getT1();
389                     final List<Command<? extends H>> chain = parsed.getT2();
390 
391                     if ( fullMatch() && args.hasNext() ) {
392                         throw new IllegalStateException( 
393                             "No full match found: " + args.toStream().toList().toString()
394                             + " was leftover after " + invocation.toString()
395                         );
396                     }
397 
398                     LOGGER.trace( "Matched invocation {}", invocation );
399 
400                     return Tuples.of( e, chain, invocation, args );
401                 } )
402                 .filter( ctx -> !ctx.getT2().isEmpty() )
403                 .filter( this::checkScope )
404                 .filter( this::checkCallable )
405                 .name( "command-parse" )
406                 .metrics();
407 
408     }

410     /**
411      * Validates that an invocation was appropriate.
412      *
413      * @param event The event that triggered the invocation.
414      * @param chain The command chain.
415      * @return A Mono that completes empty if the invocation is appropriate,
416      *         otherwise issuing a failure result.
417      */
418     @SideEffectFree
419     private Mono<CommandResult> validateCommand( final E event, 
420             final List<? extends Command<? extends H>> chain ) {
421 
422         final var validator = getValidator();
423         final var access = accessValidator( event );
424 
425         return validator.validateSettings( event, chain )
426                 .switchIfEmpty( validator.validateAccess( access, chain ) )
427                 .name( "command-validate" ).metrics();
428 
429     }

431     /**
432      * Verifies that the invocation of a command was completely handled. That is,
433      * that the final result was not {@link CommandContinue continue}.
434      *
435      * @param result The final invocation result.
436      * @param chain The command chain.
437      * @param context The invocation context.
438      * @return {@code true} if execution was fully handled.
439      * @throws IncompleteHandlingException if the execution was not fully handled.
440      */
441     private boolean verifyHandled( final CommandResult result, 
442             final List<? extends Command<? extends H>> chain, final CTX context ) 
443             throws IncompleteHandlingException {
444 
445         if ( result instanceof CommandContinue ) {
446             throw new IncompleteHandlingException( chain, context.getInvocation() );
447         } else {
448             return true;
449         }
450 
451     }

453     /**
454      * Invokes a command chain.
455      *
456      * @param chain The command chain.
457      * @param context The invocation context.
458      * @return A Mono that issues the final result once invocation is complete.
459      */
460     private Mono<CommandResult> invokeCommand( final List<? extends Command<? extends H>> chain,
461             final CTX context ) {
462 
463         final var invocation = InvocationUtils.getInvokedCommand( chain ).invocation();
464         LOGGER.debug( "Invoking command {}", invocation );
465 
466         final var handlers = InvocationUtils.accumulateHandlers(
467                 chain, this::getInvocationHandler );
468         LOGGER.trace( "Handlers for {}: {}", invocation, handlers );
469 
470         var state = Results.contMono();
471         for ( final var handler : handlers ) {
472             state = state.flatMap( r -> {
473                 if ( r instanceof CommandContinue ) {
474                     LOGGER.trace( "Invoking command handler for {}", invocation );
475                     return handler.handleWrapped( context );
476                 } else {
477                     LOGGER.trace( "Skipping command handler for {}", invocation );
478                     return Mono.just( r );
479                 }
480             } );
481         }
482 
483         return state.filter( r -> verifyHandled( r, chain, context ) )
484                 .name( "command-invoke" ).metrics();
485 
486     }

488     /**
489      * Executes a command under an invocation.
490      *
491      * @param <C> The command type.
492      * @param payload The invocation payload.
493      * @return A Mono that issues the invocation result once execution has completed.
494      */
495     private <C extends Command<? extends H>> Mono<Tuple3<C, CTX, CommandResult>> executeCommand( 
496             final Tuple4<E, List<C>, Invocation, I> payload ) {
497 
498         final E event = payload.getT1();
499         final List<C> chain = payload.getT2();
500         final Invocation invocation = payload.getT3();
501         final I args = payload.getT4();
502 
503         final var command = InvocationUtils.getInvokedCommand( chain );
504         final CTX context = makeContext( event, command, invocation, args );
505 
506         return context.initialize()
507                 .then( validateCommand( event, chain ) )
508                 .switchIfEmpty( Mono.defer( () -> context.load() ) )
509                 .switchIfEmpty( Mono.fromRunnable( () -> {
510                     context.replyManager()
511                             .setPrivate( command.privateReply() )
512                             .setEphemeral( command.ephemeralReply() );
513                 } ) )
514                 .switchIfEmpty( Mono.defer( () -> invokeCommand( chain, context ) ) )
515                 .map( result -> Tuples.of( command, context, result ) )
516                 .name( "command-execute" ).metrics();
517 
518     }

520     /**
521      * Handles the result of an invocation.
522      *
523      * @param payload The invocation result.
524      * @return A Mono that completes without issuing any value if the result was fully
525      *         handled. If the result handling was not complete for any reason, it
526      *         issues the invocation context.
527      */
528     private Mono<CTX> handleResult( 
529             final Tuple3<Command<? extends H>, CTX, CommandResult> payload ) {
530 
531         final Command<? extends H> command = payload.getT1();
532         final CTX context = payload.getT2();
533         final CommandResult result = payload.getT3();
534 
535         var state = Mono.just( context );
536         for ( final ResultHandler<? super CTX> handler : getResultHandlers( command.handlers() ) ) {
537             state = state.filterWhen( c -> handler.handle( c, result ) );
538         }
539         state = state.filterWhen( c -> BaseHandler.get().handle( c, result ) );
540 
541         return state.name( "command-result" ).metrics();
542 
543     }
