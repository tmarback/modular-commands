35  /**
36   * Type responsible for building a command processing pipeline.
37   *
38   * @param <E> The type of event that triggers commands.
39   * @param <C> The type of commands executed by the pipeline.
40   * @param <CTX> The type of command context.
41   * @param <IH> The type of invocation handler.
42   * @param <RH> The type of result handler.
43   * @version 1.0
44   * @since 1.0
45   * @apiNote Note that, for the purposes of this class, there is a distinction between <i>args</i>
46   *          and <i>arguments</i>:
47   * 
48   *          <ul>
49   *              <li><i>args</i> is the raw sequence of strings provided by the triggering event
50   *                  that identifies the command to execute. It may or may not also contain, after
51   *                  the command identifiers, strings that should be parsed into command 
52   *                  <i>arguments</i>.</li>
53   *              <li><i>arguments</i> are the values that satisfy the defined parameters of a
54   *                  command. They may be obtain from <i>args</i> that appear after the command
55   *                  identification, or directly from the event in some way (or both).</li>
56   *          </ul>
57   */
58  public abstract class PipelineBuilder<E extends Event, C extends Command, 
59          CTX extends CommandContext & LazyContext, 
60          IH extends InvocationHandler, RH extends ResultHandler> {
61  
62      /** Logger. */
63      private static final Logger LOGGER = LoggerFactory.getLogger( PipelineBuilder.class );
64  
65      /* Public interface */
66  
67      /**
68       * Builds a new pipeline with the given client and registry.
69       *
70       * @param client The client to receive events from.
71       * @param registry The registry to use to look up commands.
72       * @return The built processing pipeline.
73       */
74      public Mono<Void> buildPipeline( final GatewayDiscordClient client, final Registry registry ) {
75  
76          final Flux<E> source = client.on( eventType() )
77                  .filter( this::eventFilter )
78                  .doOnNext( e -> {
79                      LOGGER.trace( "Received event: {}", e );
80                  } );
81          return buildPipeline( source, registry ).retry();
82  
83      }
84  
85      /**
86       * Builds the processing pipeline.
87       *
88       * @param source The event source.
89       * @param registry The registry to use to look up commands.
90       * @return The built processing pipeline.
91       */
92      @SideEffectFree
93      private Mono<Void> buildPipeline( final Flux<E> source, final Registry registry ) {
94  
95          return source.flatMap( event -> parseEvent( event, registry )
96                  .flatMap( this::executeCommand )
97                  .doOnNext( ctx -> {
98                      final CTX context = ctx.getT2();
99                      final CommandResult result = ctx.getT3();
100 
101                     if ( result instanceof CommandErrorException r ) {
102                         final var cause = r.cause();
103                         LOGGER.error( String.format( "Exception while executing command %s", 
104                                 context.getInvocation() ), cause );
105                     } else if ( result instanceof CommandError r ) {
106                         LOGGER.error( "Error while executing command {}: {}", 
107                                 context.getInvocation(), r.message() );
108                     } else {
109                         LOGGER.debug( "Finished command execution {} with result {}", 
110                                 context.getInvocation(), result.getClass().getSimpleName() );
111                         LOGGER.trace( "{} => {}", context.getInvocation(), result );
112                     }
113                 } )
114                 .flatMap( this::handleResult )
115                 .doOnNext( c -> {
116                     LOGGER.warn( "Handling of result of command {} not complete", 
117                             c.getInvocation() );
118                 } )
119                 .onErrorResume( e -> {
120                     LOGGER.error( "Exception thrown within processing pipeline", e );
121                     return Mono.empty();
122                 } )
123         ).then();
124 
125     }

146     /**
147      * Determines whether a parsed arg list (as given by {@link #parse(Event)}) must always
148      * be fully matched to a command. If {@code true}, the presence of leftover args (as would
149      * be received by {@link #makeContext(Event, Command, Invocation, List)}) triggers an error
150      * for that event (stopping the processing of that particular event, but without stopping
151      * the pipeline).
152      *
153      * @return {@code true} if the parsed args must be fully matched to a command.
154      *         {@code false} if additional args are allowed.
155      * @apiNote This check is not related to any user-facing functionality, but purely as an
156      *          internal sanity check. If an event provides arguments in a way other than the
157      *          parsed args, with said args being used only to identify the command to execute,
158      *          it would not make sense for extra args to be received.
159      * 
160      *          <p>For example, in a slash command, the comand name is provided separately from
161      *          the arguments, and any subcommands are already labeled as such, so there is no
162      *          need to evaluate any further args when looking up the command. At the same
163      *          time, since slash commands (including subcommands) have to be registered with 
164      *          Discord ahead of time to be callable by users, the only way that a slash command 
165      *          event would provide a subcommand that is not found is if the command was registered
166      *          to Discord without having a corresponding internal handler, without would indicate
167      *          an inconsistency in the command system (the <i>declared</i> commands do not match
168      *          the <i>implemented</i> commands), which should be reported to the administrator
169      *          as a (non-fatal) pipeline error.
170      */
171     // END BUGGED PARAGRAPH
172     @Pure
173     protected abstract boolean fullMatch();

283     /**
284      * Invokes a handler while wrapping any thrown exceptions into a result.
285      *
286      * @param handler The handler to use.
287      * @param context The context to use.
288      * @return The result of the invocation.
289      */
290     @SuppressWarnings( "checkstyle:illegalcatch" )
291     private Mono<CommandResult> invokeWrap( final IH handler, final CTX context ) {
292 
293         try {
294             return invoke( handler, context ).onErrorResume( e -> Results.exceptionMono( e ) );
295         } catch ( final Exception e ) {
296             return Results.exceptionMono( e );
297         }
298 
299     }

301     /**
302      * Verifies that the invocation is valid for the scope that the command is defined in.
303      *
304      * @param payload The invocation payload.
305      * @return {@code true} if the invocation is valid and the command should be executed.
306      */
307     @Pure
308     private boolean checkScope( final Tuple4<E, List<C>, Invocation, List<String>> payload ) {
309 
310         final E event = payload.getT1();
311         final List<C> chain = payload.getT2();
312 
313         final C command = InvocationUtils.getInvokedCommand( chain );
314 
315         return command.scope() == Scope.GLOBAL || getGuildId( event ).isPresent();
316 
317     }

325     @Pure
326     private boolean checkCallable( final Tuple4<E, List<C>, Invocation, List<String>> payload ) {
327 
328         final List<C> chain = payload.getT2();

336     /**
337      * Parses an invocation from an event.
338      * 
339      * <p>Invocations of a command in an incompatible scope or of a command that cannot
340      * be invoked by itself are pre-filtered and will return an empty Mono.
341      *
342      * @param event The event.
343      * @param registry The registry to use for command lookups.
344      * @return A Mono that issues the parsed command, if any. 
345      */
346     @SideEffectFree
347     private Mono<Tuple4<E, List<C>, Invocation, List<String>>> parseEvent( 
348             final E event, final Registry registry ) {
349 
350         return Mono.just( event )
351                 .map( e -> Tuples.of( e, parse( e ) ) )
352                 .filter( ctx -> !ctx.getT2().isEmpty() )
353                 .mapNotNull( ctx -> {
354                     final E e = ctx.getT1();
355                     final List<String> args = ctx.getT2();
356 
357                     LOGGER.trace( "Parsed args {}", args );
358 
359                     final List<C> chain = InvocationUtils.makeChain( 
360                             registry, args, commandType() );
361 
362                     final Invocation invocation = new Invocation( args.subList( 0, chain.size() ) );
363 
364                     final var remainder = args.subList( chain.size(), args.size() );
365                     if ( fullMatch() && !remainder.isEmpty() ) {
366                         throw new IllegalStateException( 
367                             "No full match found: " + remainder.toString() 
368                             + " was leftover from " + args.toString()
369                         );
370                     }
371 
372                     LOGGER.trace( "Matched invocation {}", invocation );
373 
374                     return Tuples.of( e, chain, invocation, remainder );
375                 } )
376                 .filter( ctx -> !ctx.getT2().isEmpty() )
377                 .filter( this::checkScope )
378                 .filter( this::checkCallable )
379                 .name( "command-parse" )
380                 .metrics();
381 
382     }

384     /**
385      * Validates that an invocation was appropriate.
386      *
387      * @param event The event that triggered the invocation.
388      * @param chain The command chain.
389      * @return A Mono that completes successfully if the invocation is appropriate,
390      *         otherwise issuing a {@link ResultException} error with an error result.
391      */
392     @SideEffectFree
393     private Mono<Void> validateCommand( final E event, final List<? extends Command> chain ) {
394 
395         final var validator = getValidator();
396 
397         return validator.validateSettings( event, chain )
398                 .thenEmpty( validator.validateDiscordPermissions( event, chain ) )
399                 .name( "command-validate" ).metrics();
400 
401     }

403     /**
404      * Verifies that the invocation of a command was completely handled. That is,
405      * that the final result was not {@link CommandContinue continue}.
406      *
407      * @param result The final invocation result.
408      * @param chain The command chain.
409      * @param context The invocation context.
410      * @return {@code true} if execution was fully handled.
411      * @throws IncompleteHandlingException if the execution was not fully handled.
412      */
413     private boolean verifyHandled( final CommandResult result, final List<C> chain, 
414             final CTX context ) throws IncompleteHandlingException {
415 
416         if ( result instanceof CommandContinue ) {
417             throw new IncompleteHandlingException( chain, context.getInvocation() );
418         } else {
419             return true;
420         }
421 
422     }

424     /**
425      * Invokes a command chain.
426      *
427      * @param chain The command chain.
428      * @param context The invocation context.
429      * @return A Mono that issues the final result once invocation is complete.
430      */
431     private Mono<CommandResult> invokeCommand( final List<C> chain, final CTX context ) {
432 
433         final var invocation = InvocationUtils.getInvokedCommand( chain ).invocation();
434         LOGGER.debug( "Invoking command {}", invocation );
435 
436         final List<IH> handlers = InvocationUtils.accumulateHandlers(
437                 chain, this::getInvocationHandler );
438         LOGGER.trace( "Handlers for {}: {}", invocation, handlers );
439 
440         var state = Results.contMono();
441         for ( final var handler : handlers ) {
442             state = state.flatMap( r -> {
443                 if ( r instanceof CommandContinue ) {
444                     LOGGER.trace( "Invoking command handler for {}", invocation );
445                     return invokeWrap( handler, context );
446                 } else {
447                     LOGGER.trace( "Skipping command handler for {}", invocation );
448                     return Mono.just( r );
449                 }
450             } );
451         }
452 
453         return state.filter( r -> verifyHandled( r, chain, context ) )
454                 .name( "command-invoke" ).metrics();
455 
456     }

458     /**
459      * Executes a command under an invocation.
460      *
461      * @param payload The invocation payload.
462      * @return A Mono that issues the invocation result once execution has completed.
463      */
464     private Mono<Tuple3<C, CTX, CommandResult>> executeCommand( 
465             final Tuple4<E, List<C>, Invocation, List<String>> payload ) {
466 
467         final E event = payload.getT1();
468         final List<C> chain = payload.getT2();
469         final Invocation invocation = payload.getT3();
470         final List<String> args = payload.getT4();
471 
472         final C command = InvocationUtils.getInvokedCommand( chain );
473         final CTX context = makeContext( event, command, invocation, args );
474 
475         return validateCommand( event, chain )
476                 .thenReturn( context )
477                 .flatMap( ctx -> ctx.load().thenReturn( ctx ) )
478                 .flatMap( ctx -> invokeCommand( chain, ctx ) )
479                 .onErrorResume( ResultException.class, e -> Mono.just( e.getResult() ) )
480                 .map( result -> Tuples.of( command, context, result ) )
481                 .name( "command-execute" ).metrics();
482 
483     }

485     /**
486      * Handles the result of an invocation.
487      *
488      * @param payload The invocation result.
489      * @return A Mono that completes without issuing any value if the result was fully
490      *         handled. If the result handling was not complete for any reason, it
491      *         issues the invocation context.
492      */
493     private Mono<CTX> handleResult( final Tuple3<C, CTX, CommandResult> payload ) {
494 
495         final C command = payload.getT1();
496         final CTX context = payload.getT2();
497         final CommandResult result = payload.getT3();
498 
499         var state = Mono.just( context );
500         for ( final RH handler : getResultHandlers( command ) ) {
501             state = state.filterWhen( c -> handle( handler, c, result ) );
502         }
503 
504         return state.name( "command-result" ).metrics();
505 
506     }
