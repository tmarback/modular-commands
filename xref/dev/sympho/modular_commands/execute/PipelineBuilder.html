46  /**
47   * Type responsible for building a command processing pipeline.
48   *
49   * @param <E> The type of event that triggers commands.
50   * @param <CTX> The type of command context.
51   * @param <H> The type of handlers.
52   * @param <I> The iterator type used to traverse the received arguments.
53   * @version 1.0
54   * @since 1.0
55   * @apiNote Note that, for the purposes of this class, there is a distinction between <i>args</i>
56   *          and <i>arguments</i>:
57   * 
58   *          <ul>
59   *              <li><i>args</i> is the raw sequence of strings provided by the triggering event
60   *                  that identifies the command to execute. It may or may not also contain, after
61   *                  the command identifiers, strings that should be parsed into command 
62   *                  <i>arguments</i>.</li>
63   *              <li><i>arguments</i> are the values that satisfy the defined parameters of a
64   *                  command. They may be obtain from <i>args</i> that appear after the command
65   *                  identification, or directly from the event in some way (or both).</li>
66   *          </ul>
67   */
68  public abstract class PipelineBuilder<E extends Event, 
69          CTX extends InstrumentedContext & LazyContext, H extends Handlers, 
70          I extends SmartIterator<String>> {

131     /**

133      *
134      * @param result The execution result.
135      * @return The tag value.
136      */
137     private static String tagResult( final CommandResult result ) {
138 
139         if ( result instanceof CommandSuccess ) {
140             return "success";
141         } else if ( result instanceof CommandFailure ) {
142             return "failure";
143         } else if ( result instanceof CommandError ) {
144             return "error";
145         } else {
146             return "unknown";
147         }

179     /**
180      * Builds a new pipeline with the given client and registry.
181      *
182      * @param client The client to receive events from.
183      * @param registry The registry to use to look up commands.
184      * @return The built processing pipeline.
185      */
186     public Mono<Void> buildPipeline( final GatewayDiscordClient client, final Registry registry ) {
187 
188         final Flux<E> source = client.on( eventType() )
189                 .filter( this::eventFilter )
190                 .doOnNext( e -> {
191                     LOGGER.trace( "Received event: {}", e );
192                 } );
193         return buildPipeline( source, registry ).retry();
194 
195     }
196 
197     /**
198      * Builds the processing pipeline.
199      *
200      * @param source The event source.
201      * @param registry The registry to use to look up commands.
202      * @return The built processing pipeline.
203      */
204     @SideEffectFree
205     private Mono<Void> buildPipeline( final Flux<E> source, final Registry registry ) {
206 
207         return source.flatMap( event -> parseEvent( event, registry )
208                     .switchIfEmpty( Mono.fromRunnable( 
209                             () -> addTag( METRIC_TAG_RESULT, "not_command" ) 
210                     ) )
211                     .flatMap( this::executeCommand )
212                     // TODO: https://github.com/reactor/reactor-core/issues/3366
213                     .contextWrite( Function.identity() )
214                     .doOnNext( ctx -> {
215                         final CTX context = ctx.getT2();
216                         final CommandResult result = ctx.getT3();
217 
218                         if ( result instanceof CommandErrorException r ) {
219                             final var cause = r.cause();
220                             LOGGER.error( String.format( "Exception while executing command %s", 
221                                     context.getInvocation() ), cause );
222                         } else if ( result instanceof CommandError r ) {
223                             LOGGER.error( "Error while executing command {}: {}", 
224                                     context.getInvocation(), r.message() );
225                         } else {
226                             LOGGER.debug( "Finished command execution {} with result {}", 
227                                     context.getInvocation(), result.getClass().getSimpleName() );
228                             LOGGER.trace( "{} => {}", context.getInvocation(), result );
229                         }
230 
231                         addTagResult( result );
232                     } )
233                     .flatMap( this::handleResult )
234                     // TODO: https://github.com/reactor/reactor-core/issues/3366
235                     .contextWrite( Function.identity() )
236                     .doOnError( e -> LOGGER.error( 
237                             "Exception thrown within processing pipeline", e 
238                     ) )
239                     .contextCapture()
240                     .checkpoint( METRIC_NAME_EVENT )
241                     .name( METRIC_NAME_EVENT )
242                     .transform( addTags( event ) )
243                     .tap( Micrometer.observation( observations ) )
244                     .onErrorComplete()
245                     .thenReturn( true ) // For metric tracking purposes
246             )
247             .doOnError( e -> LOGGER.error( "Fatal error", e ) )
248             .checkpoint( METRIC_NAME_PIPELINE )
249             .name( METRIC_NAME_PIPELINE )
250             .transform( tagType()::apply )
251             .tap( Micrometer.metrics( meters ) )
252             .retryWhen( Retry.backoff( MAX_RETRIES, MIN_BACKOFF )
253                     .maxBackoff( MAX_BACKOFF )
254                     .transientErrors( true )
255             )
256             .doOnError( e -> LOGGER.error( "Pipeline closed due to too many errors" ) )
257             .then();
258 
259     }

302     /**
303      * Determines whether a parsed arg list (as given by {@link #parse(Event)}) must always
304      * be fully matched to a command. If {@code true}, the presence of leftover args (as would
305      * be received by {@link #makeContext(Event, Command, Invocation, SmartIterator)}) triggers 
306      * an error for that event (stopping the processing of that particular event, but without 
307      * stopping the pipeline).
308      *
309      * @return {@code true} if the parsed args must be fully matched to a command.
310      *         {@code false} if additional args are allowed.
311      * @apiNote This check is not related to any user-facing functionality, but purely as an
312      *          internal sanity check. If an event provides arguments in a way other than the
313      *          parsed args, with said args being used only to identify the command to execute,
314      *          it would not make sense for extra args to be received.
315      * 
316      *          <p>For example, in a slash command, the comand name is provided separately from
317      *          the arguments, and any subcommands are already labeled as such, so there is no
318      *          need to evaluate any further args when looking up the command. At the same
319      *          time, since slash commands (including subcommands) have to be registered with 
320      *          Discord ahead of time to be callable by users, the only way that a slash command 
321      *          event would provide a subcommand that is not found is if the command was registered
322      *          to Discord without having a corresponding internal handler, without would indicate
323      *          an inconsistency in the command system (the <i>declared</i> commands do not match
324      *          the <i>implemented</i> commands), which should be reported to the administrator
325      *          as a (non-fatal) pipeline error.
326      */
327     // END BUGGED PARAGRAPH

511     /**
512      * Parses an invocation from an event.
513      * 
514      * <p>Invocations of a command in an incompatible scope or of a command that cannot
515      * be invoked by itself are pre-filtered and will return an empty Mono.
516      *
517      * @param event The event.
518      * @param registry The registry to use for command lookups.
519      * @return A Mono that issues the parsed command, if any. 
520      */
521     @SideEffectFree
522     private Mono<Tuple4<E, List<Command<? extends H>>, Invocation, I>> parseEvent( 
523             final E event, final Registry registry ) {
524 
525         return Mono.just( event )
526                 .map( e -> Tuples.of( e, parse( e ) ) )
527                 .filter( ctx -> ctx.getT2().hasNext() )
528                 .map( ctx -> {
529                     final E e = ctx.getT1();
530                     final I args = ctx.getT2();
531 
532                     final var parsed = InvocationUtils.parseInvocation( 
533                             registry, args, commandType() );
534 
535                     final Invocation invocation = parsed.getT1();
536                     final List<Command<? extends H>> chain = parsed.getT2();
537 
538                     if ( fullMatch() && args.hasNext() ) {
539                         throw new IllegalStateException( 
540                             "No full match found: " + args.toStream().toList().toString()
541                             + " was leftover after " + invocation.toString()
542                         );
543                     }
544 
545                     LOGGER.trace( "Matched invocation {}", invocation );
546 
547                     return Tuples.of( e, chain, invocation, args );
548                 } )
549                 .filter( ctx -> !ctx.getT2().isEmpty() )
550                 .filter( this::checkScope )
551                 .filter( this::checkCallable )
552                 // TODO: https://github.com/reactor/reactor-core/issues/3366
553                 .contextWrite( Function.identity() )
554                 .doOnNext( r -> addTag( METRIC_TAG_RESULT, "found" ) )
555                 .switchIfEmpty( Mono.fromRunnable( 
556                         () -> addTag( METRIC_TAG_RESULT, "not_found" ) 
557                 ) )
558                 .checkpoint( METRIC_NAME_PARSE )
559                 .name( METRIC_NAME_PARSE )
560                 .transform( addTags( event ) )
561                 .tap( Micrometer.observation( observations ) );
562 
563     }

565     /**
566      * Validates that an invocation was appropriate.
567      *
568      * @param event The event that triggered the invocation.
569      * @param context The invocation context.
570      * @param chain The command chain.
571      * @return A Mono that completes empty if the invocation is appropriate,
572      *         otherwise issuing a failure result.
573      */
574     @SideEffectFree
575     private Mono<CommandResult> validateCommand( final E event, final CTX context,
576             final List<? extends Command<? extends H>> chain ) {
577 
578         final var validator = getValidator();
579         final var access = accessValidator( event );
580 
581         return validator.validateSettings( event, chain )
582                 // TODO: https://github.com/reactor/reactor-core/issues/3366
583                 .contextWrite( Function.identity() )
584                 .doOnNext( r -> addTag( METRIC_TAG_RESULT, "invalid" ) )
585                 .switchIfEmpty( validator.validateAccess( access, chain ) )
586                 // TODO: https://github.com/reactor/reactor-core/issues/3366
587                 .contextWrite( Function.identity() )
588                 .doOnNext( r -> addTag( METRIC_TAG_RESULT, "no_access" ) )
589                 .switchIfEmpty( Mono.fromRunnable( 
590                         () -> addTag( METRIC_TAG_RESULT, "allowed" ) 
591                 ) )
592                 .checkpoint( METRIC_NAME_VALIDATE )
593                 .name( METRIC_NAME_VALIDATE )
594                 .transform( context::addTags )
595                 .tap( Micrometer.observation( observations ) );
596 
597     }

599     /**
600      * Invokes a command chain.
601      *
602      * @param chain The command chain.
603      * @param context The invocation context.
604      * @return A Mono that issues the final result once invocation is complete.
605      */
606     private Mono<CommandResult> invokeCommand( 
607             final List<? extends @NonNull Command<? extends @NonNull H>> chain,
608             final CTX context ) {
609 
610         final var invocation = InvocationUtils.getInvokedCommand( chain ).invocation();
611         LOGGER.debug( "Invoking command {}", invocation );
612 
613         @SuppressWarnings( "type.argument" ) // @UnknownKeyFor bound weirdness
614         final var commands = InvocationUtils.handlingOrder( chain );
615         if ( LOGGER.isTraceEnabled() ) {
616             LOGGER.trace( "Execution order for {}: {}", invocation, commands.stream()
617                     .map( Command::id )
618                     .toList() 
619             );
620         }
621 
622         return Flux.fromIterable( commands )
623                 .concatMap( c -> getInvocationHandler( c.handlers() )
624                         .handleWrapped( context )
625                         // TODO: https://github.com/reactor/reactor-core/issues/3366
626                         .contextWrite( Function.identity() )
627                         .doOnNext( this::addTagResult )
628                         .switchIfEmpty( Mono.fromRunnable( 
629                                 () -> addTag( METRIC_TAG_RESULT, "continue" ) 
630                         ) )
631                         .checkpoint( c.id() )
632                         .name( METRIC_NAME_HANDLE )
633                         .transform( context::addTags )
634                         .tag( METRIC_TAG_HANDLER, c.id() )
635                         .tap( Micrometer.observation( observations ) )
636                 )
637                 .take( 1 ) // Ensures it stops at the first non-empty result
638                 .switchIfEmpty( Mono.error( 
639                         () -> new IncompleteHandlingException( chain, context.getInvocation() ) 
640                 ) )
641                 .single()
642                 // TODO: https://github.com/reactor/reactor-core/issues/3366
643                 .contextWrite( Function.identity() )
644                 .doOnNext( this::addTagResult )
645                 .checkpoint( METRIC_NAME_INVOKE )
646                 .name( METRIC_NAME_INVOKE )
647                 .transform( context::addTags )
648                 .tap( Micrometer.observation( observations ) );
649 
650     }

652     /**
653      * Executes a command under an invocation.
654      *
655      * @param <C> The command type.
656      * @param payload The invocation payload.
657      * @return A Mono that issues the invocation result once execution has completed.
658      * @throws IllegalStateException if some mismatch is detected.
659      */
660     private <C extends Command<? extends H>> Mono<Tuple3<C, CTX, CommandResult>> executeCommand( 
661             final Tuple4<E, List<C>, Invocation, I> payload ) {
662 
663         final E event = payload.getT1();
664         final List<C> chain = payload.getT2();
665         final Invocation invocation = payload.getT3();
666         final I args = payload.getT4();
667 
668         final var command = InvocationUtils.getInvokedCommand( chain );
669         final CTX context = makeContext( event, command, invocation, args );
670 
671         // Sanity check that the normalized invocation (e.g. the invocation after resolving 
672         // aliases) is the same as the canonical invocation
673         final var normalizedInvocation = chain.stream().map( Command::name ).toList();
674         if ( !normalizedInvocation.equals( command.invocation().chain() ) ) {
675             throw new IllegalStateException( String.format(
676                     "Normalized invocation is %s, but command %s has invocation %s",
677                     Invocation.of( normalizedInvocation ),
678                     command.id(),
679                     command.invocation()
680             ) );
681         }
682 
683         return context.initialize( observations )
684                 // TODO: https://github.com/reactor/reactor-core/issues/3366
685                 .contextWrite( Function.identity() )
686                 .then( Mono.defer( () -> validateCommand( event, context, chain ) ) )
687                 // TODO: https://github.com/reactor/reactor-core/issues/3366
688                 .contextWrite( Function.identity() )
689                 .switchIfEmpty( Mono.defer( () -> context.load() ) )
690                 // TODO: https://github.com/reactor/reactor-core/issues/3366
691                 .contextWrite( Function.identity() )
692                 .switchIfEmpty( Mono.defer( () -> invokeCommand( chain, context ) ) )
693                 // TODO: https://github.com/reactor/reactor-core/issues/3366
694                 .contextWrite( Function.identity() )
695                 .doOnNext( this::addTagResult )
696                 .map( result -> Tuples.of( command, context, result ) )
697                 .checkpoint( METRIC_NAME_EXECUTE )
698                 .name( METRIC_NAME_EXECUTE )
699                 .transform( context::addTags )
700                 .tap( Micrometer.observation( observations ) );
701 
702     }
703 
704     /**
705      * Handles the result of an invocation.
706      *
707      * @param payload The invocation result.
708      * @return A Mono that completes once handing is complete.
709      */
710     private Mono<Void> handleResult( 
711             final Tuple3<Command<? extends H>, CTX, CommandResult> payload ) {
712 
713         final Command<? extends H> command = payload.getT1();
714         final CTX context = payload.getT2();
715         final CommandResult result = payload.getT3();
716 
717         final var handlers = ListUtils.union( 
718                 getResultHandlers( command.handlers() ),
719                 List.of( BaseHandler.get() ) 
720         );
721 
722         return Flux.fromIterable( handlers )
723                 .concatMap( h -> h.handle( context, result ) )
724                 .filter( r -> r )
725                 .take( 1 ) // Stop once the first one signals complete
726                 .count()
727                 // TODO: https://github.com/reactor/reactor-core/issues/3366
728                 .contextWrite( Function.identity() )
729                 .filter( c -> c == 0 ) // No handler signaled complete
730                 .doOnNext( c -> {
731                     LOGGER.warn( "Handling of result of command {} not complete", 
732                             context.getInvocation() );
733                 } )
734                 .then()
735                 .checkpoint( METRIC_NAME_RESULT )
736                 .name( METRIC_NAME_RESULT )
737                 .transform( context::addTags )
738                 .tag( METRIC_TAG_RESULT, tagResult( result ) )
739                 .tap( Micrometer.observation( observations ) );
740 
741     }
