27  /**
28   * Executor that receives and handles commands through text messages.
29   *
30   * @version 1.0
31   * @since 1.0
32   */
33  public class MessageCommandExecutor extends BaseCommandExecutor<MessageCreateEvent, 
34          MessageContextImpl, MessageHandlers, Iterator> {
35  
36      /** Splitter to use for separating arguments in received messages. */
37      private final StringSplitter.Async splitter = new StringSplitter.Shell();
38  
39      /** Provides the prefixes that commands should have. */
40      private final PrefixProvider prefixProvider;
41  
42      /** Provides the aliases that should be applied. */
43      private final AliasProvider aliases;
44  
45      /**
46       * Creates a new instance.
47       *
48       * @param client The client to receive events from.
49       * @param registry The registry to use to look up commands.
50       * @param accessManager The access manager to use for access checks.
51       * @param meters The meter registry to use.
52       * @param observations The observation registry to use.
53       * @param prefixProvider The provider to get prefixes from.
54       * @param aliases Provides the aliases that should be applied.
55       */
56      public MessageCommandExecutor( final GatewayDiscordClient client, final Registry registry,
57              final AccessManager accessManager, 
58              final MeterRegistry meters, final ObservationRegistry observations,
59              final PrefixProvider prefixProvider, final AliasProvider aliases ) {
60  
61          super( client, registry, accessManager, meters, observations );
62  
63          this.prefixProvider = prefixProvider;
64          this.aliases = aliases;
65  
66      }
67  
68      @Override
69      protected Metrics.Tag.Type tagType() {
70          return Metrics.Tag.Type.MESSAGE;
71      }
72  
73      @Override
74      protected Class<MessageCreateEvent> eventType() {
75          return MessageCreateEvent.class;
76      }
77  
78      @Override
79      protected Class<MessageHandlers> commandType() {
80          return MessageHandlers.class;
81      }
82  
83      @Override
84      protected boolean fullMatch() {
85          return false;
86      }
87  
88      @Override
89      protected boolean eventFilter( final MessageCreateEvent event ) {
90          final var selfId = event.getClient().getSelfId();
91          return event.getMessage().getAuthor()
92                  .map( User::getId )
93                  .map( selfId::equals )
94                  .map( BooleanUtils::negate )
95                  .orElse( false );
96      }
97  
98      @Override
99      protected Iterator parse( final MessageCreateEvent event ) {
100 
101         final String message = event.getMessage().getContent();
102         if ( message.isEmpty() || Character.isWhitespace( message.codePointAt( 0 ) ) ) {
103             return splitter.emptyIterator();
104         }
105 
106         final String prefix = prefixProvider.getPrefix( event.getGuildId().orElse( null ) );
107         if ( message.startsWith( prefix ) ) {
108             final var iter = splitter.iterate( message.substring( prefix.length() ).trim() );
109             return aliases.apply( iter );
110         } else {
111             return splitter.emptyIterator();
112         }
113 
114     }
115 
116     @Override
117     protected MessageContextImpl makeContext( final MessageCreateEvent event,
118             final Command<? extends MessageHandlers> command, final Invocation invocation, 
119             final Iterator args ) {
120 
121         return new MessageContextImpl( event, invocation, command, args, accessManager );
122 
123     }
124 
125     @Override
126     protected Optional<Snowflake> getGuildId( final MessageCreateEvent event ) {
127         return event.getGuildId();
128     }
129 
130     @Override
131     protected Mono<Guild> getGuild( final MessageCreateEvent event ) {
132         return event.getGuild();
133     }
134 
135     @Override
136     protected Snowflake getChannelId( final MessageCreateEvent event ) {
137         return event.getMessage().getChannelId();
138     }
139 
140     @Override
141     protected Mono<MessageChannel> getChannel( final MessageCreateEvent event ) {
142         return event.getMessage().getChannel();
143     }
144 
145     @Override
146     protected User getCaller( final MessageCreateEvent event ) {
147 
148         final var author = event.getMessage().getAuthor();
149         if ( author.isPresent() ) {
150             return author.get();
151         } else {
152             throw new IllegalStateException( "Message with no author." );
153         }
154 
155     }
156 
157     @Override
158     @SuppressWarnings( "override.return" )
159     protected InvocationHandler<? super MessageContextImpl> getInvocationHandler( 
160             final MessageHandlers handlers ) {
161         return handlers.invocation();
162     }
163 
164     @Override
165     @SuppressWarnings( "override.return" )
166     protected List<? extends ResultHandler<? super MessageContextImpl>> getResultHandlers(
167             final MessageHandlers handlers ) {
168         return handlers.result();
169     }

171 }
