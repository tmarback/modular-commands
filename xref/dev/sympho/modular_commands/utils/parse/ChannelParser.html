1   package dev.sympho.modular_commands.utils.parse;
2   
3   import java.util.Objects;
4   
5   import org.checkerframework.checker.nullness.qual.NonNull;
6   import org.checkerframework.checker.nullness.qual.Nullable;
7   import org.checkerframework.dataflow.qual.Pure;
8   
9   import dev.sympho.modular_commands.api.command.context.CommandContext;
10  import dev.sympho.modular_commands.api.command.parameter.parse.InvalidArgumentException;
11  import dev.sympho.modular_commands.utils.OptionalUtils;
12  import discord4j.common.util.Snowflake;
13  import discord4j.core.object.entity.channel.Channel;
14  import reactor.core.publisher.Mono;
15  
16  /**
17   * A parser that extracts a channel from a string.
18   *
19   * @param <C> The channel type.
20   * @version 1.0
21   * @since 1.0
22   */
23  public class ChannelParser<C extends @NonNull Channel> extends MentionableParser<C> {
24  
25      /** The channel type. */
26      private final Class<C> type;
27  
28      /** Link URL parser. */
29      private final ChannelUrlParser<C> urlParser;
30  
31      /**
32       * Creates a new instance.
33       *
34       * @param type The type of channel.
35       */
36      @Pure
37      public ChannelParser( final Class<C> type ) {
38  
39          this.type = Objects.requireNonNull( type );
40          this.urlParser = new ChannelUrlParser<>( type );
41  
42      }
43  
44      /**
45       * Validates that a channel is of the expected type.
46       *
47       * @param channel The channel.
48       * @return The channel.
49       * @throws InvalidArgumentException if the channel does not match the expected type.
50       */
51      @Pure
52      private C validate( final Channel channel ) throws InvalidArgumentException {
53          
54          return ChannelUrlParser.validate( channel, type );
55  
56      }
57  
58      @Override
59      public @Nullable ChannelUrlParser<C> getUrlParser() {
60          return urlParser;
61      }
62  
63      @Override
64      protected Mono<C> getEntity( final CommandContext context, final Snowflake id ) {
65  
66          return context.getGuild().flatMap( g -> g.getChannelById( id ) ).map( this::validate );
67  
68      }
69  
70      @Override
71      public String parseMention( final String mention ) throws InvalidArgumentException {
72  
73          return OptionalUtils.castPresent( MentionableParser.parseMention( mention, "#" ) )
74                  .orElseThrow( () -> new InvalidArgumentException(
75                          "Not a valid channel mention: <%s>".formatted( mention ) )
76                  );
77  
78      }
79      
80  }
