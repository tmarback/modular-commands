18  /**
19   * A parser for URL-based arguments that are independent from the invocation context.
20   *
21   * @param <T> The parsed type.
22   * @version 1.0
23   * @since 1.0
24   */
25  public interface FunctorUrlParser<T extends @NonNull Object> extends UrlParser<T> {
26  
27      /**
28       * Creates a parser that delegates to other parsers, chosen by the given mapping function
29       * based on the URL to parse.
30       *
31       * @param <T> The parsed argument type.
32       * @param parserMapper The function to use to determine which parser to delegate to for
33       *                     a given URL.
34       * @return The parser.
35       */
36      static <T extends @NonNull Object> FunctorUrlParser<T> choice(
37              final Function<URL, @Nullable FunctorUrlParser<T>> parserMapper ) {
38  
39          return new Choice<>( parserMapper );
40  
41      }
42  
43      /**
44       * Creates a parser that delegates to the given parsers.
45       * 
46       * <p>The parser choice is defined as the first parser in the iteration order of the given
47       * collection for which {@link #supports(URL)} returns {@code true} for the URL being parsed.
48       * This implies that the iteration order matters if, and only if, there are URLs that may be
49       * supported by more than one of the parsers in the collection.
50       *
51       * @param <T> The parsed argument type.
52       * @param parsers The parsers to delegate to.
53       * @return The parser.
54       */
55      static <T extends @NonNull Object> FunctorUrlParser<T> choice(
56              final Collection<? extends FunctorUrlParser<T>> parsers ) {
57  
58          return choice( UrlParserUtils.toMapper( parsers ) );
59  
60      }
61  
62      /**
63       * Creates a parser that delegates to the given parsers.
64       * 
65       * <p>The parser choice is defined as the first parser in the given order for which 
66       * {@link #supports(URL)} returns {@code true} for the URL being parsed.
67       * This implies that the iteration order matters if, and only if, there are URLs that 
68       * may be supported by more than one of the parsers in the collection.
69       *
70       * @param <T> The parsed argument type.
71       * @param parsers The parsers to delegate to.
72       * @return The parser.
73       */
74      @SafeVarargs
75      @SuppressWarnings( "varargs" )
76      static <T extends @NonNull Object> FunctorUrlParser<T> choice( 
77              final FunctorUrlParser<T>... parsers ) {
78  
79          return choice( Arrays.asList( parsers ) );
80  
81      }
82  
83      /**
84       * Creates a parser that delegates to a parser depending on the host of the URL, using
85       * the given host-parser mappings.
86       *
87       * @param <T> The parsed argument type.
88       * @param parsers The host-parser mappings to delegate to.
89       * @return The parser.
90       */
91      static <T extends @NonNull Object> FunctorUrlParser<T> choiceHost(
92              final Map<String, ? extends FunctorUrlParser<T>> parsers ) {
93  
94          return choice( UrlParserUtils.toHostMapper( parsers ) );
95  
96      }
97  
98      /**
99       * Parses the given URL.
100      *
101      * @param url The URL to parse.
102      * @return The parsed value. May result in a {@link InvalidArgumentException} if the
103      *         URL is invalid.
104      * @throws InvalidArgumentException if the URL is invalid.
105      */
106     @SideEffectFree
107     Mono<T> parse( URL url ) throws InvalidArgumentException;
108 
109     @Override
110     default Mono<T> parse( final CommandContext context, final URL url ) 
111             throws InvalidArgumentException {
112 
113         return parse( url );
114 
115     }
116 
117     /**
118      * Returns a composed parser that first applies this parser to its input, and then applies
119      * the after parser to the result. If parsing with either parser throws an exception, it is 
120      * relayed to the caller of the composed parser.
121      *
122      * @param <V> The type of output of the {@code after} parser, and of the composed parser.
123      * @param after The parser to apply after this parser is applied.
124      * @return The composed parser.
125      */
126     @SideEffectFree
127     default <V extends @NonNull Object> UrlParser<V> then( final Functor<T, V> after ) {
128 
129         return new PostParser<>( this, after );
130 
131     }
132 
133     /**
134      * A composed parser that first applies this parser to one parser, and then applies the 
135      * results to a second parser. If parsing with either parser throws an exception, it is 
136      * relayed to the caller of the composed parser.
137      * 
138      * <p>Note that {@link #supports(URL) compatibility} is defined only by the first parser.
139      *
140      * @param <I> The intermediary type output by the first parser and consumed by the second.
141      * @param <T> The final output type.
142      * @param <P1> The type of the first parser.
143      * @param <P2> The type of the second parser.
144      * @since 1.0
145      */
146     class PostParser<
147                     I extends @NonNull Object, 
148                     T extends @NonNull Object, 
149                     P1 extends @NonNull FunctorUrlParser<I>,
150                     P2 extends @NonNull Functor<I, T>
151             > extends UrlParser.PostParser<I, T, P1, P2> implements FunctorUrlParser<T> {
152 
153         /**
154          * Creates a new instance.
155          *
156          * @param parser The first parser to apply.
157          * @param postParser The second parser to apply.
158          */
159         public PostParser( final P1 parser, final P2 postParser ) {
160 
161             super( parser, postParser );
162 
163         }
164 
165         @Override
166         public Mono<T> parse( final URL url ) throws InvalidArgumentException {
167 
168             return parser.parse( url )
169                     .flatMap( postParser::parse );
170 
171         }
172 
173     }
174 
175     /**
176      * Base for a parser that supports multiple URL types by delegating to one of a list of 
177      * parsers. Note that this includes {@link #supports(URL) compatibility checks}.
178      *
179      * @param <T> The parsed argument type.
180      * @param <P> The delegate parser type.
181      * @since 1.0
182      */
183     abstract class ChoiceBase<T extends @NonNull Object, P extends FunctorUrlParser<T>> 
184             extends UrlParser.ChoiceBase<T, P> implements FunctorUrlParser<T> {
185 
186         /**
187          * Creates a new instance.
188          *
189          * @param parserMapper The function to use to determine which parser to delegate to for
190          *                     a given URL.
191          */
192         protected ChoiceBase( final Function<URL, @Nullable P> parserMapper ) {
193 
194             super( parserMapper );
195 
196         }
197 
198         @Override
199         public Mono<T> parse( final URL url ) throws InvalidArgumentException {
200 
201             return getParser( url ).parse( url );
202 
203         }
204 
205     }
206 
207     /**
208      * Parser that supports multiple URL types by delegating to one of a list of parsers. Note
209      * that this includes {@link #supports(URL) compatibility checks}.
210      *
211      * @param <T> The parsed argument type.
212      * @since 1.0
213      * @apiNote This is a convenience subtype of {@link FunctorUrlParser.ChoiceBase ChoiceBase} 
214      *          with the second parameter already set.
215      */
216     class Choice<T extends @NonNull Object> extends ChoiceBase<T, FunctorUrlParser<T>> {
217 
218         /**
219          * Creates a new instance.
220          *
221          * @param parserMapper The function to use to determine which parser to delegate to for
222          *                     a given URL.
223          */
224         public Choice( final Function<URL, @Nullable FunctorUrlParser<T>> parserMapper ) {
225 
226             super( parserMapper );
227 
228         }
229 
230     }
