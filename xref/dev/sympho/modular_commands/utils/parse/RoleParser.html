1   package dev.sympho.modular_commands.utils.parse;
2   
3   import dev.sympho.modular_commands.api.command.context.CommandContext;
4   import dev.sympho.modular_commands.api.command.parameter.parse.InvalidArgumentException;
5   import dev.sympho.modular_commands.utils.OptionalUtils;
6   import discord4j.common.util.Snowflake;
7   import discord4j.core.object.entity.Guild;
8   import discord4j.core.object.entity.Role;
9   import reactor.core.publisher.Mono;
10  
11  /**
12   * A parser that extracts a role from a string.
13   *
14   * @version 1.0
15   * @since 1.0
16   */
17  public class RoleParser extends MentionableParser<Role> {
18  
19      /** Creates a new instance. */
20      public RoleParser() {}
21  
22      @Override
23      protected Mono<Role> getEntity( final CommandContext context, final Snowflake id ) {
24  
25          return context.getGuild().flatMap( g -> g.getRoleById( id ) );
26  
27      }
28  
29      @Override
30      public String parseMention( final String mention ) throws InvalidArgumentException {
31  
32          return OptionalUtils.castPresent( MentionableParser.parseMention( mention, "@&" ) )
33                  .orElseThrow( () -> new InvalidArgumentException(
34                          "Not a valid role mention: <%s>".formatted( mention ) )
35                  );
36  
37      }
38  
39      @Override
40      public Mono<Role> parse( final CommandContext context, final String raw )
41              throws InvalidArgumentException {
42  
43          if ( "@everyone".equals( raw ) || "@here".equals( raw ) ) {
44              return context.getGuild().flatMap( Guild::getEveryoneRole );
45          } else {
46              return super.parse( context, raw );
47          }
48  
49      }
50      
51  }
