3   import java.util.function.Function;
4   
5   import org.checkerframework.checker.nullness.qual.NonNull;
6   import org.checkerframework.dataflow.qual.SideEffectFree;
7   
8   import dev.sympho.modular_commands.api.command.context.CommandContext;
9   import dev.sympho.modular_commands.api.command.parameter.parse.InvalidArgumentException;
10  import discord4j.common.util.Snowflake;
11  import discord4j.core.object.entity.Message;
12  import discord4j.core.object.entity.Role;
13  import discord4j.core.object.entity.User;
14  import discord4j.core.object.entity.channel.Channel;
15  import reactor.core.publisher.Mono;
16  
17  /**
18   * Parser function that converts a string into one of the supported raw types.
19   *
20   * @param <R> The raw type.
21   * @version 1.0
22   * @since 1.0
23   * @apiNote You would normally never need to manually implement this, as this interface already
24   *          provides parsers for all the supported types through constants and static methods.
25   */
26  @FunctionalInterface
27  public interface RawParser<R extends @NonNull Object> extends Function<String, Mono<R>> {
28  
29      /**
30       * Parser for strings.
31       *
32       * @apiNote This is a trivial parser, but it exists for consistency reasons.
33       */
34      RawParser<String> STRING = Mono::just;
35  
36      /** Parser for booleans. */
37      @SuppressWarnings( "switch.expression" ) // Weird issue with the interning checker
38      RawParser<Boolean> BOOLEAN = raw -> switch ( raw.toLowerCase() ) {
39          case "true" -> Mono.just( true );
40          case "false" -> Mono.just( false );
41          default -> Mono.error( new InvalidArgumentException(
42              "Not a valid boolean: " + raw
43          ) );
44      };
45  
46      /** Parser for integers. */
47      RawParser<Long> INTEGER = raw -> {
48          try {
49              return Mono.just( Long.decode( raw ) );
50          } catch ( final NumberFormatException ex ) {
51              return Mono.error( new InvalidArgumentException( 
52                  "Not a valid integer: " + raw, ex
53               ) );
54          }
55      };
56  
57      /** Parser for floats. */
58      RawParser<Double> FLOAT = raw -> {
59          try {
60              return Mono.just( Double.parseDouble( raw ) );
61          } catch ( final NumberFormatException ex ) {
62              return Mono.error( new InvalidArgumentException( 
63                  "Not a valid number: " + raw, ex
64              ) );
65          }
66      };
67  
68      /** Parser for snowflakes. */
69      RawParser<Snowflake> SNOWFLAKE = raw -> {
70          try {
71              return Mono.just( Snowflake.of( raw ) );
72          } catch ( final NumberFormatException ex ) {
73              return Mono.error( new InvalidArgumentException( 
74                  "Not a valid snowflake: " + raw, ex
75               ) );
76          }
77      };
78  
79      /**
80       * Creates a parser for messages with the given context.
81       *
82       * @param context The context to use.
83       * @return The parser.
84       */
85      @SideEffectFree
86      static RawParser<Message> message( final CommandContext context ) {
87          return raw -> ParseUtils.MESSAGE.parse( context, raw );
88      }
89  
90      /**
91       * Creates a parser for users with the given context.
92       *
93       * @param context The context to use.
94       * @return The parser.
95       */
96      @SideEffectFree
97      static RawParser<User> user( final CommandContext context ) {
98          return raw -> ParseUtils.USER.parse( context, raw );
99      }
100 
101     /**
102      * Creates a parser for roles with the given context.
103      *
104      * @param context The context to use.
105      * @return The parser.
106      */
107     @SideEffectFree
108     static RawParser<Role> role( final CommandContext context ) {
109         return raw -> ParseUtils.ROLE.parse( context, raw );
110     }
111 
112     /**
113      * Creates a parser for channels with the given context.
114      *
115      * @param <C> The channel type.
116      * @param context The context to use.
117      * @param type The channel type.
118      * @return The parser.
119      */
120     @SideEffectFree
121     static <C extends @NonNull Channel> RawParser<C> channel( 
122             final CommandContext context, final Class<C> type ) {
123         // Can't use a static parser due to the generics
124         final var parser = ParseUtils.channel( type );
125         return raw -> parser.parse( context, raw );
126     }
127 
128     /**
129      * Parses the given string into a raw value.
130      *
131      * @param raw The original string.
132      * @return The pre-parsed raw value.
133      */
134     @SideEffectFree
135     Mono<R> parse( String raw );
136 
137     @Override
138     @SideEffectFree
139     default Mono<R> apply( final String raw ) {
140         return parse( raw );
141     }
142     
143 }
