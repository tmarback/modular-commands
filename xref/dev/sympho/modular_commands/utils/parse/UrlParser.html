20  /**
21   * A parser for URL-based arguments.
22   *
23   * @param <T> The parsed type.
24   * @version 1.0
25   * @since 1.0
26   */
27  public interface UrlParser<T extends @NonNull Object> extends ParserFunction<String, T> {
28  
29      /**
30       * Creates a parser that delegates to other parsers, chosen by the given mapping function
31       * based on the URL to parse.
32       *
33       * @param <T> The parsed argument type.
34       * @param parserMapper The function to use to determine which parser to delegate to for
35       *                     a given URL.
36       * @return The parser.
37       */
38      static <T extends @NonNull Object> UrlParser<T> choice(
39              final Function<URL, @Nullable UrlParser<T>> parserMapper ) {
40  
41          return new Choice<>( parserMapper );
42  
43      }
44  
45      /**
46       * Creates a parser that delegates to the given parsers.
47       * 
48       * <p>The parser choice is defined as the first parser in the iteration order of the given
49       * collection for which {@link #supports(URL)} returns {@code true} for the URL being parsed.
50       * This implies that the iteration order matters if, and only if, there are URLs that may be
51       * supported by more than one of the parsers in the collection.
52       *
53       * @param <T> The parsed argument type.
54       * @param parsers The parsers to delegate to.
55       * @return The parser.
56       */
57      static <T extends @NonNull Object> UrlParser<T> choice(
58              final Collection<? extends UrlParser<T>> parsers ) {
59  
60          return choice( UrlParserUtils.toMapper( parsers ) );
61  
62      }
63  
64      /**
65       * Creates a parser that delegates to the given parsers.
66       * 
67       * <p>The parser choice is defined as the first parser in the given order for which 
68       * {@link #supports(URL)} returns {@code true} for the URL being parsed.
69       * This implies that the iteration order matters if, and only if, there are URLs that 
70       * may be supported by more than one of the parsers in the collection.
71       *
72       * @param <T> The parsed argument type.
73       * @param parsers The parsers to delegate to.
74       * @return The parser.
75       */
76      @SafeVarargs
77      @SuppressWarnings( "varargs" )
78      static <T extends @NonNull Object> UrlParser<T> choice( 
79              final UrlParser<T>... parsers ) {
80  
81          return choice( Arrays.asList( parsers ) );
82  
83      }
84  
85      /**
86       * Creates a parser that delegates to a parser depending on the host of the URL, using
87       * the given host-parser mappings.
88       *
89       * @param <T> The parsed argument type.
90       * @param parsers The host-parser mappings to delegate to.
91       * @return The parser.
92       */
93      static <T extends @NonNull Object> UrlParser<T> choiceHost(
94              final Map<String, ? extends UrlParser<T>> parsers ) {
95  
96          return choice( UrlParserUtils.toHostMapper( parsers ) );
97  
98      }
99  
100     /**
101      * Parses the given string into a URL.
102      *
103      * @param raw The string to parse.
104      * @return The URL.
105      * @throws InvalidArgumentException if the given string is not a valid URL.
106      * @apiNote Mainly a wrapper to convert the URL exception into the invalid argument
107      *          type.
108      */
109     @SideEffectFree
110     static URL getUrl( final String raw ) throws InvalidArgumentException {
111 
112         try {
113             return new URL( raw );
114         } catch ( final MalformedURLException e ) {
115             throw new InvalidArgumentException( "Not a valid URL: " + raw, e );
116         }
117 
118     }
119 
120     /**
121      * Checks if the given URL is supported by this parser. If this returns {@code false},
122      * calling {@link #parse(CommandContext, URL)} with the same URL <i>will</i> result in
123      * an error.
124      * 
125      * <p>Note that the opposite isn't necessarily true; it is possible for this method to
126      * return {@code true} for a given URL while {@link #parse(CommandContext, URL)} results
127      * in an error. That just means that the basic structure of the URL was detected as being 
128      * <i>compatible</i> with this parser (for example, having a particular host and/or protocol),
129      * but was malformed or a variant that the parser doesn't support.
130      *
131      * @param url The URL to check.
132      * @return Whether the URL is compatible with this parser.
133      */
134     @Pure
135     boolean supports( URL url );
136 
137     /**
138      * Parses the given URL.
139      *
140      * @param context The execution context.
141      * @param url The URL to parse.
142      * @return The parsed value. May result in a {@link InvalidArgumentException} if the
143      *         URL is invalid.
144      * @throws InvalidArgumentException if the URL is invalid.
145      */
146     @SideEffectFree
147     Mono<T> parse( CommandContext context, URL url ) throws InvalidArgumentException;
148 
149     @Override
150     default Mono<T> parse( final CommandContext context, final String raw ) 
151             throws InvalidArgumentException {
152 
153         return parse( context, getUrl( raw ) );
154 
155     }
156 
157     /**
158      * Returns a composed parser that first applies this parser to its input, and then applies
159      * the after parser to the result. If parsing with either parser throws an exception, it is 
160      * relayed to the caller of the composed parser.
161      *
162      * @param <V> The type of output of the {@code after} parser, and of the composed parser.
163      * @param after The parser to apply after this parser is applied.
164      * @return The composed parser.
165      */
166     @SideEffectFree
167     default <V extends @NonNull Object> UrlParser<V> then( 
168             final ParserFunction<T, V> after ) {
169 
170         return new PostParser<>( this, after );
171 
172     }
173 
174     /**
175      * A composed parser that first applies this parser to one parser, and then applies the 
176      * results to a second parser. If parsing with either parser throws an exception, it is 
177      * relayed to the caller of the composed parser.
178      * 
179      * <p>Note that {@link #supports(URL) compatibility} is defined only by the first parser.
180      *
181      * @param <I> The intermediary type output by the first parser and consumed by the second.
182      * @param <T> The final output type.
183      * @param <P1> The type of the first parser.
184      * @param <P2> The type of the second parser.
185      * @since 1.0
186      */
187     class PostParser<
188                     I extends @NonNull Object, 
189                     T extends @NonNull Object, 
190                     P1 extends @NonNull UrlParser<I>,
191                     P2 extends @NonNull ParserFunction<I, T>
192             > extends ParserFunction.PostParser<String, I, T, P1, P2> implements UrlParser<T> {
193 
194         /**
195          * Creates a new instance.
196          *
197          * @param parser The first parser to apply.
198          * @param postParser The second parser to apply.
199          */
200         public PostParser( final P1 parser, final P2 postParser ) {
201 
202             super( parser, postParser );
203 
204         }
205 
206         @Override
207         public boolean supports( final URL url ) {
208             return parser.supports( url );
209         }
210 
211         @Override
212         public Mono<T> parse( final CommandContext context, final URL url ) 
213                 throws InvalidArgumentException {
214 
215             return parser.parse( context, url )
216                     .flatMap( i -> postParser.parse( context, i ) );
217 
218         }
219 
220     }

222     /**
223      * Base for a parser that supports multiple URL types by delegating to one of a list of 
224      * parsers. Note that this includes {@link #supports(URL) compatibility checks}.
225      *
226      * @param <T> The parsed argument type.
227      * @param <P> The delegate parser type.
228      * @since 1.0
229      */
230     abstract class ChoiceBase<T extends @NonNull Object, P extends UrlParser<T>> 
231             implements UrlParser<T> {
232             
233         /** The parsers to delegate to. */
234         private final Function<URL, @Nullable P> parserMapper;
235 
236         /**
237          * Creates a new instance.
238          *
239          * @param parserMapper The function to use to determine which parser to delegate to for
240          *                     a given URL.
241          */
242         protected ChoiceBase( final Function<URL, @Nullable P> parserMapper ) {
243 
244             this.parserMapper = parserMapper;
245 
246         }
247 
248         @Override
249         public boolean supports( final URL url ) {
250 
251             final var parser = parserMapper.apply( url );
252             return parser != null && parser.supports( url );
253 
254         }
255 
256         /**
257          * Retrives the parser to use for the given URL.
258          *
259          * @param url The URL being parsed.
260          * @return The parser to use.
261          * @throws InvalidArgumentException if none of the parsers support the given URL.
262          */
263         @Pure
264         protected P getParser( final URL url ) throws InvalidArgumentException {
265 
266             final var parser = parserMapper.apply( url );
267             if ( parser == null ) {
268                 throw new InvalidArgumentException( "Unsupported URL: " + url );
269             } else {
270                 return parser;
271             }
272         }
273 
274         @Override
275         public Mono<T> parse( final CommandContext context, final URL url ) 
276                 throws InvalidArgumentException {
277 
278             return getParser( url ).parse( context, url );
279 
280         }
281 
282     }

284     /**
285      * Parser that supports multiple URL types by delegating to one of a list of parsers. Note
286      * that this includes {@link #supports(URL) compatibility checks}.
287      *
288      * @param <T> The parsed argument type.
289      * @since 1.0
290      * @apiNote This is a convenience subtype of {@link ChoiceBase} with the second parameter
291      *          already set.
292      */
293     class Choice<T extends @NonNull Object> extends ChoiceBase<T, UrlParser<T>> {
294 
295         /**
296          * Creates a new instance.
297          *
298          * @param parserMapper The function to use to determine which parser to delegate to for
299          *                     a given URL.
300          */
301         public Choice( final Function<URL, @Nullable UrlParser<T>> parserMapper ) {
302 
303             super( parserMapper );
304 
305         }
306 
307     }
