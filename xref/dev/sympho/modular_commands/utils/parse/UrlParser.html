24  /**
25   * A parser for URL-based arguments.
26   *
27   * @param <T> The parsed type.
28   * @version 1.0
29   * @since 1.0
30   */
31  public interface UrlParser<T extends @NonNull Object> extends ParserFunction<String, T> {

33      /** Pattern that matches a masked link. */
34      @Regex( 1 ) Pattern MASKED_URL_PATTERN = Pattern.compile( "\\[.*\\]\\(([^\\[\\]]+?)\\)" );

36      /**

46              final Function<URL, @Nullable UrlParser<T>> parserMapper ) {

52      /**
53       * Creates a parser that delegates to the given parsers.
54       * 
55       * <p>The parser choice is defined as the first parser in the iteration order of the given
56       * collection for which {@link #supports(URL)} returns {@code true} for the URL being parsed.
57       * This implies that the iteration order matters if, and only if, there are URLs that may be
58       * supported by more than one of the parsers in the collection.
59       *
60       * @param <T> The parsed argument type.
61       * @param parsers The parsers to delegate to.
62       * @return The parser.
63       */
64      static <T extends @NonNull Object> UrlParser<T> choice(
65              final Collection<? extends UrlParser<T>> parsers ) {
66  
67          return choice( UrlParserUtils.toMapper( parsers ) );

107     /**
108      * Clears special URL formatting from the given string, if any exists.
109      * 
110      * <p>The following formatting types are supported:
111      * 
112      * <ul>
113      *   <li>Supressed links (links surrounded by {@code <>})</li>
114      *   <li>Masked links (links that display as clickable text)</li>
115      * </ul>
116      * 
117      * <p>Note that layered formats are supported if supported by Discord; for example, 
118      * {@code [Discord](<https://discord.com>)} is a link that is both supressed and masked.
119      *
120      * @param raw The string to strip.
121      * @return The stripped string.
122      */
123     @SideEffectFree
124     static String stripFormatting( String raw ) {
125 
126         // Check for masked links
127         final var match = MASKED_URL_PATTERN.matcher( raw.trim() );
128         if ( match.matches() ) {
129             raw = NullnessUtil.castNonNull( match.group( 1 ) );
130         }
131 
132         // Check for supressed links
133         if ( raw.startsWith( "<" ) && raw.endsWith( ">" ) ) {
134             raw = raw.substring( 1, raw.length() - 1 );
135         }
136         
137         return raw;
138 
139     }

141     /**
142      * Parses the given string into a URL.
143      *
144      * @param raw The string to parse.
145      * @return The URL.
146      * @throws InvalidArgumentException if the given string is not a valid URL.
147      * @apiNote Mainly a wrapper to convert the URL exception into the invalid argument
148      *          type.
149      */
150     @SideEffectFree
151     static URL getUrl( final String raw ) throws InvalidArgumentException {
152 
153         try {
154             return new URL( raw );
155         } catch ( final MalformedURLException e ) {
156             throw new InvalidArgumentException( "Not a valid URL: " + raw, e );
157         }
158 
159     }
160 
161     /**
162      * Attempts to parse a string as a URL.
163      * 
164      * <p>Some effort is made to detect URLs surrounded by special markup; see 
165      * {@link #stripFormatting(String)}.
166      *
167      * @param raw The string to parse.
168      * @param allowedProtocols The acceptable protocols.
169      * @return The parsed URL if one could be parsed, or {@code null} if the given string is
170      *         not a URL or has a protocol that is not contained in {@code allowedProtocols}.
171      * @throws InvalidArgumentException if the given string has a URL format (and an allowed
172      *                                  protocol) but is invalid in some way.
173      */
174     @SideEffectFree
175     static @Nullable URL parseUrl( final String raw, final Collection<String> allowedProtocols )
176             throws InvalidArgumentException {
177 
178         final var clean = stripFormatting( raw );
179 
180         final var sep = clean.indexOf( "://" );
181         if ( sep <= 0 ) {
182             return null; // Not a URL
183         }
184 
185         final var protocol = clean.substring( 0, sep );
186         if ( !allowedProtocols.contains( protocol ) ) {
187             return null; // Invalid protocol
188         }
189 
190         return getUrl( clean );
191 
192     }
193 
194     /**
195      * Attempts to parse a string as a URL.
196      * 
197      * <p>Some effort is made to detect URLs surrounded by special markup; see 
198      * {@link #stripFormatting(String)}.
199      *
200      * @param raw The string to parse.
201      * @param allowedProtocol The acceptable protocol.
202      * @return The parsed URL if one could be parsed, or {@code null} if the given string is
203      *         not a URL or does not have the {@code allowedProtocol}.
204      * @throws InvalidArgumentException if the given string has a URL format (and the allowed
205      *                                  protocol) but is invalid in some way.
206      */
207     @SideEffectFree
208     static @Nullable URL parseUrl( final String raw, final String allowedProtocol )
209             throws InvalidArgumentException {
210 
211         return parseUrl( raw, List.of( allowedProtocol ) );
212 
213     }

215     /**
216      * Checks if the given URL is supported by this parser. If this returns {@code false},
217      * calling {@link #parse(CommandContext, URL)} with the same URL <i>will</i> result in
218      * an error.
219      * 
220      * <p>Note that the opposite isn't necessarily true; it is possible for this method to
221      * return {@code true} for a given URL while {@link #parse(CommandContext, URL)} results
222      * in an error. That just means that the basic structure of the URL was detected as being 
223      * <i>compatible</i> with this parser (for example, having a particular host and/or protocol),
224      * but was malformed or a variant that the parser doesn't support.
225      *
226      * @param url The URL to check.
227      * @return Whether the URL is compatible with this parser.
228      */
229     @Pure
230     boolean supports( URL url );
231 
232     /**
233      * Parses the given URL.
234      *
235      * @param context The execution context.
236      * @param url The URL to parse.
237      * @return The parsed value. May result in a {@link InvalidArgumentException} if the
238      *         URL is invalid.
239      * @throws InvalidArgumentException if the URL is invalid.
240      */
241     @SideEffectFree
242     Mono<T> parse( CommandContext context, URL url ) throws InvalidArgumentException;
243 
244     @Override
245     default Mono<T> parse( final CommandContext context, final String raw ) 
246             throws InvalidArgumentException {
247 
248         return parse( context, getUrl( stripFormatting( raw ) ) );
249 
250     }

252     /**
253      * Returns a composed parser that first applies this parser to its input, and then applies
254      * the after parser to the result. If parsing with either parser throws an exception, it is 
255      * relayed to the caller of the composed parser.
256      *
257      * @param <V> The type of output of the {@code after} parser, and of the composed parser.
258      * @param after The parser to apply after this parser is applied.
259      * @return The composed parser.
260      */
261     @SideEffectFree
262     default <V extends @NonNull Object> UrlParser<V> then( 
263             final ParserFunction<T, V> after ) {
264 
265         return new PostParser<>( this, after );
266 
267     }

269     /**
270      * A composed parser that first applies this parser to one parser, and then applies the 
271      * results to a second parser. If parsing with either parser throws an exception, it is 
272      * relayed to the caller of the composed parser.
273      * 
274      * <p>Note that {@link #supports(URL) compatibility} is defined only by the first parser.
275      *
276      * @param <I> The intermediary type output by the first parser and consumed by the second.
277      * @param <T> The final output type.
278      * @param <P1> The type of the first parser.
279      * @param <P2> The type of the second parser.
280      * @since 1.0
281      */
282     class PostParser<
283                     I extends @NonNull Object, 
284                     T extends @NonNull Object, 
285                     P1 extends @NonNull UrlParser<I>,
286                     P2 extends @NonNull ParserFunction<I, T>
287             > extends ParserFunction.PostParser<String, I, T, P1, P2> implements UrlParser<T> {
288 
289         /**
290          * Creates a new instance.
291          *
292          * @param parser The first parser to apply.
293          * @param postParser The second parser to apply.
294          */
295         public PostParser( final P1 parser, final P2 postParser ) {
296 
297             super( parser, postParser );
298 
299         }
300 
301         @Override
302         public boolean supports( final URL url ) {
303             return parser.supports( url );
304         }
305 
306         @Override
307         public Mono<T> parse( final CommandContext context, final URL url ) 
308                 throws InvalidArgumentException {
309 
310             return parser.parse( context, url )
311                     .flatMap( i -> postParser.parse( context, i ) );
312 
313         }
314 
315     }

317     /**
318      * Base for a parser that supports multiple URL types by delegating to one of a list of 
319      * parsers. Note that this includes {@link #supports(URL) compatibility checks}.
320      *
321      * @param <T> The parsed argument type.
322      * @param <P> The delegate parser type.
323      * @since 1.0
324      */
325     abstract class ChoiceBase<T extends @NonNull Object, P extends UrlParser<T>> 
326             implements UrlParser<T> {
327             
328         /** The parsers to delegate to. */
329         private final Function<URL, @Nullable P> parserMapper;
330 
331         /**
332          * Creates a new instance.
333          *
334          * @param parserMapper The function to use to determine which parser to delegate to for
335          *                     a given URL.
336          */
337         protected ChoiceBase( final Function<URL, @Nullable P> parserMapper ) {
338 
339             this.parserMapper = parserMapper;
340 
341         }
342 
343         @Override
344         public boolean supports( final URL url ) {
345 
346             final var parser = parserMapper.apply( url );
347             return parser != null && parser.supports( url );
348 
349         }
350 
351         /**
352          * Retrives the parser to use for the given URL.
353          *
354          * @param url The URL being parsed.
355          * @return The parser to use.
356          * @throws InvalidArgumentException if none of the parsers support the given URL.
357          */
358         @Pure
359         protected P getParser( final URL url ) throws InvalidArgumentException {
360 
361             final var parser = parserMapper.apply( url );
362             if ( parser == null ) {
363                 throw new InvalidArgumentException( "Unsupported URL: " + url );
364             } else {
365                 return parser;
366             }
367         }
368 
369         @Override
370         public Mono<T> parse( final CommandContext context, final URL url ) 
371                 throws InvalidArgumentException {
372 
373             return getParser( url ).parse( context, url );
374 
375         }
376 
377     }
