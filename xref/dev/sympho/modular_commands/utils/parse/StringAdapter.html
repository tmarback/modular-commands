3   import org.checkerframework.checker.nullness.qual.NonNull;

5   import dev.sympho.modular_commands.api.command.context.CommandContext;
6   import dev.sympho.modular_commands.api.command.parameter.parse.ParserFunction;
7   import reactor.core.publisher.Mono;
8   
9   /**
10   * Adapter that converts an existing parser into a string parser, by first pre-parsing
11   * the string into another raw type.
12   *
13   * @param <R> The raw type of the original parser.
14   * @param <T> The parsed argument type.
15   * @param rawParser The parser to use to pre-parse into the raw type.
16   * @param valueParser The parser to use to parse the raw type.
17   * @version 1.0
18   * @since 1.0
19   */
20  public record StringAdapter<R extends @NonNull Object, T extends @NonNull Object>(
21          ParserFunction<String, R> rawParser,
22          ParserFunction<R, T> valueParser
23  ) implements ParserFunction<String, T> {
24  
25      @Override
26      public Mono<T> parse( final CommandContext context, final String raw ) {
27  
28          return rawParser.parse( context, raw )
29                  .flatMap( value -> valueParser.parse( context, value ) );
30  
31      }
32      
33  }
