1   package dev.sympho.modular_commands.utils.parse;
2   
3   import java.net.URL;
4   
5   import org.checkerframework.checker.nullness.qual.NonNull;
6   import org.checkerframework.checker.nullness.qual.Nullable;
7   import org.checkerframework.dataflow.qual.Pure;
8   import org.checkerframework.dataflow.qual.SideEffectFree;
9   
10  import dev.sympho.modular_commands.api.command.context.CommandContext;
11  import dev.sympho.modular_commands.api.command.parameter.parse.InvalidArgumentException;
12  import discord4j.core.GatewayDiscordClient;
13  import discord4j.core.object.entity.Entity;
14  import reactor.core.publisher.Mono;
15  
16  /**
17   * A parser for URLs of Discord entities.
18   *
19   * @param <E> The entity type.
20   * @version 1.0
21   * @since 1.0
22   */
23  public abstract class EntityUrlParser<E extends @NonNull Entity> implements UrlParser<E> {
24  
25      /**
26       * Checks if the given path (endpoint) is valid for the entity type of this parser.
27       *
28       * @param path The path.
29       * @return If the path is valid for this parser.
30       */
31      @Pure
32      protected abstract boolean validPath( String path );
33  
34      /**
35       * Parses the URL path.
36       *
37       * @param client The Discord client.
38       * @param path The path.
39       * @return The parsed entity, or {@code null} if the path is not valid.
40       */
41      @SideEffectFree
42      protected abstract @Nullable Mono<E> parsePath( GatewayDiscordClient client, String path );
43  
44      /**
45       * Gets the display name for this type.
46       *
47       * @return The name.
48       * @apiNote For use in error messages.
49       */
50      @Pure
51      protected abstract String typeName();
52  
53      /**
54       * Checks if the URL is a valid (generic) Discord URL.
55       *
56       * @param url The URL to check.
57       * @return If it is a valid Discord URL.
58       */
59      @Pure
60      private boolean baseValid( final URL url ) {
61  
62          return UrlParserUtils.isHttps( url )
63                  && UrlParserUtils.isHost( url, "discord.com" );
64  
65      }
66  
67      @Override
68      public boolean supports( final URL url ) {
69  
70          return baseValid( url ) && validPath( url.getPath() );
71  
72      }
73  
74      @Override
75      public Mono<E> parse( final CommandContext context, final URL url ) 
76              throws InvalidArgumentException {
77  
78          if ( !baseValid( url ) ) {
79              throw new InvalidArgumentException( "Not a valid Discord URL: %s".formatted( url ) );
80          }
81  
82          final var res = parsePath( context.getClient(), url.getPath() );
83          if ( res == null ) {
84              throw new InvalidArgumentException( "Not a valid %s URL: %s".formatted( 
85                      typeName(), url ) );
86          } else {
87              return res;
88          }
89  
90      }
91      
92  }
