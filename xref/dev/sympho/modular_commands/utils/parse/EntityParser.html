1   package dev.sympho.modular_commands.utils.parse;
2   
3   import org.checkerframework.checker.nullness.qual.NonNull;
4   import org.checkerframework.checker.nullness.qual.Nullable;
5   import org.checkerframework.dataflow.qual.Pure;
6   import org.checkerframework.dataflow.qual.SideEffectFree;
7   
8   import dev.sympho.modular_commands.api.command.context.CommandContext;
9   import dev.sympho.modular_commands.api.command.parameter.parse.InvalidArgumentException;
10  import dev.sympho.modular_commands.api.command.parameter.parse.ParserFunction;
11  import discord4j.common.util.Snowflake;
12  import discord4j.core.object.entity.Entity;
13  import reactor.core.publisher.Mono;
14  
15  /**
16   * A parser that extracts an entity from a string.
17   *
18   * @param <E> The entity type.
19   * @version 1.0
20   * @since 1.0
21   */
22  public abstract class EntityParser<E extends @NonNull Entity> implements ParserFunction<String, E> {
23  
24      /**
25       * Parses a snowflake ID from an argument string.
26       *
27       * @param raw The argument.
28       * @return The snowflake ID.
29       * @throws InvalidArgumentException if the given string is not a valid snowflake.
30       */
31      @SideEffectFree
32      public static Snowflake parseId( final String raw ) throws InvalidArgumentException {
33  
34          try {
35              return Snowflake.of( raw );
36          } catch ( final NumberFormatException e ) {
37              throw new InvalidArgumentException(
38                      String.format( "Value '%s' is not a valid snowflake ID.", raw ) );
39          }
40  
41      }
42  
43      /**
44       * Retrieves the parser to use for URLs, if supported.
45       *
46       * @return The URL parser, or {@code null} if URLs are not supported.
47       * @implSpec The default returns {@code null}.
48       */
49      @Pure
50      public @Nullable EntityUrlParser<E> getUrlParser() {
51          return null;
52      }
53  
54      /**
55       * Parses the entity ID from the raw argument.
56       *
57       * @param raw The received argument.
58       * @return The entity ID.
59       * @throws InvalidArgumentException if the given string is not a valid ID.
60       * @implSpec Supports only plain IDs by default.
61       */
62      @SideEffectFree
63      public Snowflake extractId( final String raw ) throws InvalidArgumentException {
64          return parseId( raw );
65      }
66  
67      /**
68       * Retrieves the entity with the given ID.
69       *
70       * @param context The execution context.
71       * @param id The entity ID.
72       * @return The entity. May be empty if not found.
73       */
74      @SideEffectFree
75      protected abstract Mono<E> getEntity( CommandContext context, Snowflake id );
76  
77      @Override
78      public Mono<E> parse( final CommandContext context, final String raw )
79              throws InvalidArgumentException {
80  
81          if ( getUrlParser() != null && raw.startsWith( "https://" ) ) {
82              return getUrlParser().parse( context, raw );
83          }
84  
85          return getEntity( context, extractId( raw ) )
86                  .onErrorMap( e -> new InvalidArgumentException( "Invalid.", e ) )
87                  .switchIfEmpty( 
88                      Mono.error( () -> new InvalidArgumentException( "Not found." ) )
89                  );
90  
91      }
92      
93  }
