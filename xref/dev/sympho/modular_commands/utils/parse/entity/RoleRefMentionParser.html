3   import org.checkerframework.checker.nullness.qual.Nullable;
4   
5   import dev.sympho.modular_commands.api.command.context.CommandContext;
6   import dev.sympho.modular_commands.api.command.parameter.parse.InvalidArgumentException;
7   import dev.sympho.modular_commands.utils.parse.entity.EntityRef.RoleRef;
8   import discord4j.common.util.Snowflake;
9   
10  /**
11   * A parser for user mentions to references.
12   * 
13   * <p>Note that this parser does not support {@code @everyone} and {@code @here} in
14   * {@link #parseId(String)}, as they do not have a unique ID. They are, however,
15   * supported by {@link #parseRef(CommandContext, String)} and 
16   * {@link #parse(CommandContext, String)}.
17   *
18   * @version 1.0
19   * @since 1.0
20   */
21  public class RoleRefMentionParser extends EntityRefMentionParser<RoleRef> {
22  
23      /** Creates a new instance. */
24      public RoleRefMentionParser() {}
25  
26      @Override
27      protected boolean prefixMatches( final String prefix ) {
28  
29          return "@&".equals( prefix );
30  
31      }
32  
33      @Override
34      protected RoleRef makeRef( final CommandContext context, final Snowflake id ) {
35  
36          return new RoleRef( context, id );
37  
38      }
39  
40      @Override
41      protected String typeName() {
42          return "role";
43      }
44  
45      @Override
46      public @Nullable RoleRef parseRef( final CommandContext context, final String raw ) 
47              throws InvalidArgumentException {
48  
49          if ( "@everyone".equals( raw ) || "@here".equals( raw ) ) {
50              return new RoleRef( context, RoleRef.everyoneId( context ) );
51          } else {
52              return super.parseRef( context, raw );
53          }
54  
55      }
56      
57  }
