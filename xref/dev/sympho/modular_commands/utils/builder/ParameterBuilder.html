4   
5   import org.checkerframework.checker.calledmethods.qual.CalledMethods;
6   import org.checkerframework.checker.nullness.qual.MonotonicNonNull;
7   import org.checkerframework.checker.nullness.qual.NonNull;
8   import org.checkerframework.checker.nullness.qual.Nullable;
9   import org.checkerframework.common.returnsreceiver.qual.This;
10  import org.checkerframework.common.value.qual.MatchesRegex;
11  import org.checkerframework.dataflow.qual.Deterministic;
12  import org.checkerframework.dataflow.qual.Pure;
13  import org.checkerframework.dataflow.qual.SideEffectFree;
14  
15  import dev.sympho.modular_commands.api.command.parameter.Parameter;
16  import dev.sympho.modular_commands.api.command.parameter.parse.ArgumentParser;
17  import dev.sympho.modular_commands.utils.ParameterUtils;

19  /**
20   * Base for a parameter builder.
21   *
22   * @param <T> The type of value received by the parameter.
23   * @version 1.0
24   * @since 1.0
25   */
26  @SuppressWarnings( "checkstyle:hiddenfield" )
27  public final class ParameterBuilder<T extends @NonNull Object> {
28  
29      /** The parameter name. */
30      protected @MonotonicNonNull @MatchesRegex( Parameter.NAME_REGEX ) String name;
31      
32      /** The parameter description. */
33      protected @MonotonicNonNull @MatchesRegex( Parameter.DESCRIPTION_REGEX ) String description;
34  
35      /** Whether the parameter is required. */
36      protected boolean required;
37      
38      /** The default parameter value. */
39      protected @Nullable T defaultValue;
40  
41      /** The parser to use for processing. */
42      protected @MonotonicNonNull ArgumentParser<?, T> parser;
43  
44      /**
45       * Constructs a new builder with default values.
46       */
47      @Pure
48      public ParameterBuilder() {
49  
50          this.name = null;
51          this.description = null;
52          this.required = false;
53          this.defaultValue = null;
54          this.parser = null;
55  
56      }
57  
58      /**
59       * Constructs a new builder that is a copy of the given builder.
60       *
61       * @param base The builder to copy.
62       */
63      @SideEffectFree
64      public ParameterBuilder( final ParameterBuilder<T> base ) {
65  
66          this.name = base.name;
67          this.description = base.description;
68          this.required = base.required;
69          this.defaultValue = base.defaultValue;
70          this.parser = base.parser;
71  
72      }
73  
74      /**
75       * Constructs a new builder that is initialized to make a copy of 
76       * the given parameter.
77       *
78       * @param <T> The argument type.
79       * @param base The parameter to copy.
80       * @return The initialized builder.
81       * @throws IllegalArgumentException if the given parameter has invalid values.
82       */
83      @SideEffectFree
84      public static <T extends @NonNull Object> @CalledMethods( {
85              "withName", "withDescription", 
86              "withRequired", "withDefault", 
87              "withParser" 
88      } ) ParameterBuilder<T> from( final Parameter<T> base ) throws IllegalArgumentException {
89  
90          return new ParameterBuilder<T>()
91                  .withName( base.name() )
92                  .withDescription( base.description() )
93                  .withRequired( base.required() )
94                  .withDefault( base.defaultValue() )
95                  .withParser( base.parser() );
96  
97      }
98  
99      /**
100      * Sets the name of the parameter to build.
101      *
102      * @param name The parameter name.
103      * @return This builder.
104      * @throws IllegalArgumentException if the given name is invalid.
105      * @see Parameter#name()
106      */
107     @Deterministic
108     public @This ParameterBuilder<T> withName( 
109             final @MatchesRegex( Parameter.NAME_REGEX ) String name ) 
110             throws IllegalArgumentException {
111 
112         this.name = ParameterUtils.validateName( name );
113         return this;
114 
115     }
116 
117     /**
118      * Sets the description of the parameter to build.
119      *
120      * @param description The parameter description.
121      * @return This builder.
122      * @throws IllegalArgumentException if the given name is invalid.
123      * @see Parameter#description()
124      */
125     @Deterministic
126     public @This ParameterBuilder<T> withDescription( 
127             final @MatchesRegex( Parameter.DESCRIPTION_REGEX ) String description ) 
128             throws IllegalArgumentException {
129 
130         this.description = ParameterUtils.validateDescription( description );
131         return this;
132 
133     }
134 
135     /**
136      * Sets the required flag of the parameter to build.
137      *
138      * @param required The required flag.
139      * @return This builder.
140      * @see Parameter#required()
141      */
142     @Deterministic
143     public @This ParameterBuilder<T> withRequired( final boolean required ) {
144 
145         this.required = required;
146         return this;
147 
148     }
149 
150     /**
151      * Sets the default value of the parameter to build.
152      *
153      * @param defaultValue The default value.
154      * @return This builder.
155      * @see Parameter#defaultValue()
156      */
157     @Deterministic
158     public @This ParameterBuilder<T> withDefault( final @Nullable T defaultValue ) {
159 
160         this.defaultValue = defaultValue;
161         return this;
162 
163     }
164 
165     /**
166      * Sets the parser to process arguments with.
167      *
168      * @param parser The parser to use.
169      * @return This builder.
170      * @see Parameter#parser()
171      */
172     @Deterministic
173     public @This ParameterBuilder<T> withParser( final ArgumentParser<?, T> parser ) {
174 
175         this.parser = Objects.requireNonNull( parser );
176         return this;
177 
178     }
179 
180     /**
181      * Builds the configured parameter.
182      *
183      * @return The built parameter.
184      * @throws IllegalStateException if the current configuration is invalid.
185      */
186     @SideEffectFree
187     public Parameter<T> build( 
188             @CalledMethods( { "withName", "withDescription", "withParser" } ) 
189             ParameterBuilder<T> this
190     ) throws IllegalStateException {
191 
192         if ( this.name == null ) {
193             throw new IllegalStateException( "Parameter name must be set before building." );
194         }
195         if ( this.description == null ) {
196             throw new IllegalStateException( "Parameter description must be set before building." );
197         }
198         if ( this.parser == null ) {
199             throw new IllegalStateException( "Parameter parser must be set before building." );
200         }
201 
202         return new Parameter<>( name, description, required, defaultValue, parser );
203 
204     }
205 
206     /**
207      * Indicates whether the given object is a builder of the same type and
208      * configuration as this one.
209      */
210     @Override
211     public boolean equals( final @Nullable Object o ) {
212 
213         if ( o instanceof ParameterBuilder<?> b ) {
214             return Objects.equals( this.name, b.name )
215                     && Objects.equals( this.description, b.description )
216                     && this.required == b.required
217                     && Objects.equals( this.defaultValue, b.defaultValue )
218                     && Objects.equals( this.parser, b.parser );
219         } else {
220             return false;
221         }
222 
223     }
224 
225     @Override
226     public int hashCode() {
227 
228         return Objects.hash( 
229                 this.name, this.description, 
230                 this.required, this.defaultValue, 
231                 this.parser 
232         );
233 
234     }
