3   import org.checkerframework.dataflow.qual.Pure;
4   import org.checkerframework.dataflow.qual.SideEffectFree;
5   
6   import dev.sympho.modular_commands.api.command.parameter.EntityParameter;
7   import discord4j.core.object.entity.Entity;
8   
9   /**
10   * Base for an entity parameter builder.
11   *
12   * @param <T> The type of entity received by the parameter.
13   * @param <P> The parameter type.
14   * @param <SELF> The self type.
15   * @see EntityParameter
16   * @version 1.0
17   * @since 1.0
18   */
19  abstract sealed class EntityParameterBuilder<
20              T extends Entity, 
21              P extends EntityParameter<T>,
22              SELF extends EntityParameterBuilder<T, P, SELF>
23          > extends ParameterBuilder<T, P, SELF> permits MentionableParameterBuilder {
24  
25      /**
26       * Constructs a new builder with default values.
27       */
28      @Pure
29      protected EntityParameterBuilder() {}
30  
31      /**
32       * Constructs a new builder that is a copy of the given builder.
33       *
34       * @param base The builder to copy.
35       */
36      @SideEffectFree
37      protected EntityParameterBuilder( final EntityParameterBuilder<? extends T, ?, ?> base ) {
38  
39          super( base );
40  
41      }
42  
43      /**
44       * Constructs a new builder that is initialized to make a copy of 
45       * the given parameter.
46       *
47       * @param base The parameter to copy.
48       */
49      @SideEffectFree
50      protected EntityParameterBuilder( final EntityParameter<T> base ) {
51  
52          super( base );
53  
54      }
55      
56  }
