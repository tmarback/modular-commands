1   package dev.sympho.modular_commands.utils.builder;

3   import org.checkerframework.dataflow.qual.Deterministic;
4   
5   /**
6    * Base type for a builder.
7    *
8    * @param <SELF> The self type to be returned in all chain methods.
9    * @version 1.0
10   * @since 1.0
11   * @apiNote Be very careful when using this. {@code SELF} <b>MUST</b> be the type of the
12   *          builder itself, or at least compatible with the return type of the chain
13   *          builder methods, else it will cause an exception when assigned to a variable
14   *          (and the compiler won't be able to warn about it).
15   */
16  public interface Builder<SELF extends Builder<SELF>> {
17  
18      /**
19       * Retrieves {@code this} cast to the self type.
20       *
21       * @return {@code this}.
22       * @apiNote This does <i>not</i> check that the cast is valid. Make sure that
23       *          {@code SELF} is set to the appropriate type.
24       */
25      @Deterministic
26      @SuppressWarnings( { "unchecked", "return" } )
27      default SELF self() {
28  
29          return ( SELF ) this;
30  
31      }
32      
33  }
