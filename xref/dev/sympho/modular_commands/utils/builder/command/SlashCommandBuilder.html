1   package dev.sympho.modular_commands.utils.builder.command;
2   
3   import org.checkerframework.dataflow.qual.SideEffectFree;
4   
5   import dev.sympho.modular_commands.api.command.SlashCommand;
6   import dev.sympho.modular_commands.api.command.handler.SlashInvocationHandler;
7   import dev.sympho.modular_commands.api.command.handler.SlashResultHandler;
8   import dev.sympho.modular_commands.impl.command.SlashCommandImpl;
9   
10  /**
11   * A builder for a slash command.
12   *
13   * @version 1.0
14   * @since 1.0
15   */
16  public final class SlashCommandBuilder extends CommandBuilder<SlashCommand, SlashInvocationHandler, 
17          SlashResultHandler, SlashCommandBuilder> {
18  
19      /**
20       * Constructs a new builder with default values.
21       */
22      @SideEffectFree
23      public SlashCommandBuilder() {}
24  
25      /**
26       * Constructs a new builder that is a copy of the given builder.
27       *
28       * @param base The builder to copy.
29      */
30      @SideEffectFree
31      public SlashCommandBuilder( final SlashCommandBuilder base ) {
32          super( base );
33      }
34  
35      /**
36       * Constructs a new builder that is initialized to make a copy of 
37       * the given command.
38       *
39       * @param base The command to copy.
40       * @throws IllegalArgumentException if the given command has invalid values.
41       */
42      @SideEffectFree
43      public SlashCommandBuilder( final SlashCommand base ) throws IllegalArgumentException {
44          super( base );
45      }
46  
47      @Override
48      public SlashCommand build() throws IllegalStateException {
49  
50          try {
51              return new SlashCommandImpl( 
52                  scope, callable, parent, buildName(), buildDisplayName(), 
53                  buildDescription(), parameters,
54                  requiredDiscordPermissions, requireParentPermissions, 
55                  nsfw, botOwnerOnly, serverOwnerOnly, privateReply, 
56                  inheritSettings, invokeParent,
57                  buildInvocationHandler(), resultHandlers );
58          } catch ( final IllegalArgumentException e ) {
59              throw new IllegalStateException( "Invalid parameter configuration.", e );
60          }
61  
62      }
63      
64  }
