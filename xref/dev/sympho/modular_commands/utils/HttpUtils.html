3   import java.nio.charset.Charset;
4   import java.util.function.Function;

6   import org.checkerframework.dataflow.qual.SideEffectFree;

8   import io.netty.handler.codec.http.HttpHeaderNames;
9   import io.netty.handler.codec.http.HttpStatusClass;
10  import io.netty.handler.codec.http.HttpUtil;
11  import reactor.netty.http.client.HttpClientResponse;
12  
13  /**
14   * Utilities for working with HTTP requests.
15   *
16   * @version 1.0
17   * @since 1.0
18   */
19  public final class HttpUtils {
20  
21      /** Do not instantiate. */
22      private HttpUtils() {}
23  
24      /**
25       * Retrives the charset used to encode an HTTP response.
26       * 
27       * <p>Also checks that the request succeeded, throwing a custom exception if it
28       * did not.
29       *
30       * @param <E> The exception type to throw if the request did not succeed.
31       * @param response The response metadata.
32       * @param errorMapper The function to use to generate an exception with a given error
33       *                    message.
34       * @return The encoding charset.
35       * @throws E if the request did not succeed.
36       */
37      @SideEffectFree
38      public static <E extends RuntimeException> Charset getCharset( 
39              final HttpClientResponse response, final Function<String, E> errorMapper ) throws E {
40  
41          final var status = response.status();
42          if ( status.codeClass() != HttpStatusClass.SUCCESS ) {
43              throw errorMapper.apply( "Request failed: " + status );
44          }
45  
46          final var contentType = response.responseHeaders()
47                  .get( HttpHeaderNames.CONTENT_TYPE );
48          return HttpUtil.getCharset( contentType );
49  
50      }
51  
52      /**
53       * Retrives the charset used to encode an HTTP response.
54       * 
55       * <p>Also checks that the request succeeded.
56       *
57       * @param response The response metadata.
58       * @return The encoding charset.
59       * @throws IllegalStateException if the request did not succeed.
60       */
61      @SideEffectFree
62      public static Charset getCharset( final HttpClientResponse response ) 
63              throws IllegalStateException {
64  
65          return getCharset( response, IllegalStateException::new );
66  
67      }
68      
69  }
