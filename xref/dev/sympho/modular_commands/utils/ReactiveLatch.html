1   package dev.sympho.modular_commands.utils;
2   
3   import java.util.concurrent.CountDownLatch;
4   import java.util.concurrent.atomic.AtomicLong;
5   import java.util.function.Function;
6   
7   import org.checkerframework.dataflow.qual.Pure;
8   import org.checkerframework.dataflow.qual.SideEffectFree;
9   import org.reactivestreams.Publisher;
10  
11  import reactor.core.publisher.Mono;
12  import reactor.core.publisher.Sinks;
13  
14  /**
15   * An equivalent to Java's {@link CountDownLatch} that uses Reactive Streams for non-blocking
16   * waiting. It can be used by either directly subscribing to {@link #await()} or by using an
17   * instance of this class with {@link Mono#delayUntil(Function)} (or the Flux equivalent).
18   * 
19   * <p>The latch can also be failed, propagating an error to those waiting on it instead of
20   * a completion.
21   *
22   * @see CountDownLatch
23   * @version 1.0
24   * @since 1.0
25   */
26  public class ReactiveLatch implements Function<Object, Publisher<Void>> {
27  
28      /** The backing sink. */
29      private final Sinks.Empty<Void> sink;
30  
31      /** The counter. */
32      private final AtomicLong count;
33  
34      /**
35       * Creates a new latch that opens after counting down a given number of times.
36       *
37       * @param needed How many times {@link #countDown()} needs to be called before
38       *               the latch opens.
39       * @throws IllegalArgumentException if the given value is not positive (>0).
40       */
41      @SideEffectFree
42      public ReactiveLatch( final long needed ) throws IllegalArgumentException {
43  
44          if ( needed <= 0 ) {
45              throw new IllegalArgumentException( "Latch must require a positive number." );
46          }
47  
48          this.sink = Sinks.empty();
49          this.count = new AtomicLong( needed );
50  
51      }
52  
53      /**
54       * Creates a new latch that opens after {@link #countDown()} is called once. This is
55       * effectively a binary latch or a switch.
56       */
57      @SideEffectFree
58      public ReactiveLatch() {
59  
60          this( 1L );
61  
62      }
63  
64      /**
65       * Creates a publisher that completes once this latch opens.
66       */
67      @Pure
68      @Override
69      public Publisher<Void> apply( final Object ignored ) {
70  
71          return await();
72  
73      }
74  
75      /**
76       * Creates a Mono that emits a completion signal when this latch enters the open state.
77       * 
78       * <p>Once the latch is open, subscribing to the returned Mono will complete instantly.
79       *
80       * @return The Mono to wait on.
81       */
82      @Pure
83      public Mono<Void> await() {
84  
85          return sink.asMono();
86  
87      }
88  
89      /**
90       * Counts down the latch. If the internal counter reaches zero, the latch opens.
91       * 
92       * <p>Has no effect if the latch is already opened or failed.
93       *
94       * @throws IllegalStateException if emitting the completion signal failed.
95       */
96      public void countDown() throws IllegalStateException {
97  
98          if ( count.decrementAndGet() == 0 ) { // Only the call that reaches zero emits
99              final var result = sink.tryEmitEmpty();
100             if ( result.isFailure() ) {
101                 throw new IllegalStateException( "Releasing latch failed: " + result );
102             }
103         }
104 
105     }
106 
107     /**
108      * Fails the latch, issuing an error signal to those waiting on it.
109      * 
110      * <p>Has no effect if the latch is already opened or failed.
111      *
112      * @param error The error to issue.
113      * @throws IllegalStateException if emitting the error signal failed.
114      */
115     public void fail( final Throwable error ) throws IllegalStateException {
116 
117         if ( count.getAndSet( 0 ) > 0 ) { // Only do it if not open yet
118             final var result = sink.tryEmitError( error );
119             if ( result.isFailure() ) {
120                 throw new IllegalStateException( "Issuing error on latch failed: " + result );
121             }
122         }
123 
124     }
125     
126 }
