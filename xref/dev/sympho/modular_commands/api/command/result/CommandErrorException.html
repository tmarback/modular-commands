1   package dev.sympho.modular_commands.api.command.result;
2   
3   import java.util.Objects;
4   
5   import org.checkerframework.dataflow.qual.Pure;
6   
7   /**
8    * An error result due to an exception being thrown.
9    *
10   * @version 1.0
11   * @since 1.0
12   */
13  public interface CommandErrorException extends CommandError {
14  
15      /**
16       * Retrieves the cause of the error.
17       *
18       * @return The cause.
19       */
20      @Pure
21      Throwable cause();
22  
23      /**
24       * {@inheritDoc}
25       *
26       * @implSpec The default behavior is to return the {@link Throwable#getMessage() message}
27       *           provided by the {@link #cause() exception}.
28       */
29      @Override
30      default String message() {
31  
32          return Objects.requireNonNullElse( cause().getMessage(), "An exception was thrown." );
33  
34      }
35      
36  }
