1   package dev.sympho.modular_commands.api.command.result;
2   
3   import discord4j.rest.util.Permission;
4   import discord4j.rest.util.PermissionSet;
5   
6   /**
7    * A failure result due to the user having insufficient permissions.
8    *
9    * @param missingDiscordPermissions The built-in permissions that the user is missing but are
10   *                                  required to run the command.
11   * @version 1.0
12   * @since 1.0
13   */
14  public record UserMissingPermissions( PermissionSet missingDiscordPermissions ) 
15          implements CommandFailureMessage {
16  
17      @Override
18      public String message() {
19  
20          final Iterable<String> missing = () -> missingDiscordPermissions().stream()
21                  .map( Permission::toString ).iterator();
22          return String.format( "Missing permissions: %s", String.join( ", ", missing ) );
23  
24      }
25      
26  }
