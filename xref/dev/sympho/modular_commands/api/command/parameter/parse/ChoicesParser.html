1   package dev.sympho.modular_commands.api.command.parameter.parse;
2   
3   import java.util.List;
4   import java.util.Map;
5   
6   import org.checkerframework.checker.nullness.qual.NonNull;
7   import org.checkerframework.checker.nullness.qual.Nullable;
8   import org.checkerframework.dataflow.qual.Pure;
9   
10  /**
11   * Parses received input arguments, potentially restricting the acceptable values to a
12   * set of choices.
13   *
14   * @param <P> The primitive type to be parsed.
15   * @param <T> The type of argument that is provided.
16   * @version 1.0
17   * @since 1.0
18   * @apiNote Implementations do not have to verify in parsing that the received raw value
19   *          is one of the valid choices.
20   *          The system will automatically do this verification before invoking the parser.
21   */
22  public sealed interface ChoicesParser<P extends @NonNull Object, T extends @NonNull Object> 
23          extends InputParser<P, T> permits StringParser, NumberParser {
24  
25      /** The maximum number of choices allowed. */
26      int MAX_CHOICES = 25;
27  
28      /**
29       * The allowed choices.
30       *
31       * @return The choices, or {@code null} if any value is allowed.
32       *         Must have between 1 and {@value #MAX_CHOICES} elements if not {@code null}.
33       */
34      @Pure
35      default @Nullable List<Choice<P>> choices() {
36          return null;
37      }
38  
39      /**
40       * Verifies that the given value is one of the allowed choices.
41       *
42       * @param value The value to validate.
43       * @return The value.
44       * @throws InvalidArgumentException If the value is not a valid choice.
45       */
46      @Pure
47      default P verifyChoice( final P value ) throws InvalidArgumentException {
48  
49          final var c = choices();
50          if ( c != null && !c.stream().map( Choice::value ).anyMatch( value::equals ) ) {
51              throw new InvalidArgumentException( "Not a valid choice" );
52          }
53          return value;
54  
55      }
56  
57      @Override
58      default P validateRaw( final P raw ) throws InvalidArgumentException {
59          return verifyChoice( InputParser.super.validateRaw( raw ) );
60      }
61  
62      /**
63       * A possible choice to be selected.
64       *
65       * @param <P> The raw value type.
66       * @param name The choice name.
67       * @param value The choice value.
68       * @since 1.0
69       */
70      record Choice<P extends @NonNull Object>(
71              String name,
72              P value
73      ) {
74  
75          /**
76           * Creates a new choice.
77           *
78           * @param <P> The raw value type.
79           * @param name The choice name.
80           * @param choice The choice value.
81           * @return The choice.
82           */
83          public static <P extends @NonNull Object> Choice<P> of( 
84                  final String name, final P choice ) {
85  
86              return new Choice<>( name, choice );
87      
88          }
89  
90          /**
91           * Creates a new choice mapping.
92           *
93           * @param <P> The raw value type.
94           * @param <V> The mapped value type.
95           * @param name The choice name.
96           * @param choice The choice value.
97           * @param value The mapped value.
98           * @return The choice.
99           */
100         public static <P extends @NonNull Object, V extends @NonNull Object> 
101                 Map.Entry<Choice<P>, V> of( final String name, final P choice, final V value ) {
102 
103             return Map.entry( of( name, choice ), value );
104     
105         }
106 
107     }
108     
109 }
