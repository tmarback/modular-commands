3   import java.nio.charset.Charset;
4   
5   import com.google.common.net.MediaType;
6   
7   import org.checkerframework.checker.nullness.qual.NonNull;
8   import org.checkerframework.dataflow.qual.SideEffectFree;
9   
10  import dev.sympho.modular_commands.api.command.context.CommandContext;
11  import discord4j.core.object.entity.Attachment;
12  import io.netty.handler.codec.http.HttpUtil;
13  import reactor.core.publisher.Mono;
14  import reactor.netty.ByteBufMono;
15  
16  /**
17   * Parses an argument from an attached text file.
18   *
19   * @param <T> The type of argument that is parsed.
20   * @version 1.0
21   * @since 1.0
22   */
23  @FunctionalInterface
24  public interface TextFileParser<T extends @NonNull Object> extends AttachmentDataParser<T> {
25  
26      /**
27       * @implSpec Verifies that the attachment is a text file (any type of MIME text file,
28       *           not necessarily .txt).
29       */
30      @Override
31      default void validate( final Attachment attachment ) throws InvalidArgumentException {
32  
33          if ( !attachment.getContentType()
34                  .map( MediaType::parse )
35                  .map( t -> t.is( MediaType.ANY_TEXT_TYPE ) )
36                  .orElse( false ) ) {
37              throw new InvalidArgumentException( "Attachment must be a text file" );
38          }
39  
40      }
41  
42      /**
43       * Parses the content of the attachment file.
44       *
45       * @param context The execution context.
46       * @param content The attachment content.
47       * @return A Mono that issues the parsed argument. If the content is invalid, it may
48       *         fail with a {@link InvalidArgumentException}.
49       * @throws InvalidArgumentException if the given content is not valid.
50       */
51      @SideEffectFree
52      Mono<T> parse( CommandContext context, String content ) throws InvalidArgumentException;
53  
54      /**
55       * @implSpec Parses the attachment file content using {@link #parse(CommandContext, String)}.
56       */
57      @Override
58      default Mono<T> parse( final CommandContext context, final Attachment attachment, 
59              final ByteBufMono body ) throws InvalidArgumentException {
60  
61          final Charset encoding = HttpUtil.getCharset( attachment.getContentType().orElse( null ) );
62          return body.asString( encoding ).flatMap( c -> parse( context, c ) );
63  
64      }
65      
66  }
