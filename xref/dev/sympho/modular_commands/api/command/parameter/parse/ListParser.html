3   import java.util.List;
4   import java.util.function.Function;
5   
6   import org.checkerframework.checker.nullness.qual.NonNull;
7   import org.checkerframework.common.value.qual.IntRange;
8   import org.checkerframework.dataflow.qual.Pure;
9   import org.checkerframework.dataflow.qual.SideEffectFree;
10  
11  import dev.sympho.modular_commands.api.command.context.CommandContext;
12  import dev.sympho.modular_commands.utils.StringSplitter;
13  import dev.sympho.modular_commands.utils.parse.TryParser;
14  import reactor.core.publisher.Flux;
15  import reactor.core.publisher.Mono;
16  
17  /**
18   * A parser that extracts lists of objects by splitting a string argument and parsing each item.
19   * 
20   * <p>Items appear in the resulting list in the same order that they appear in the raw argument.
21   *
22   * @param <T> The element type.
23   * @version 1.0
24   * @since 1.0
25   */
26  @FunctionalInterface
27  public interface ListParser<T extends @NonNull Object> extends StringParser<List<T>> {
28  
29      /** The default splitter ({@link StringSplitter.Shell}). */
30      StringSplitter.Async DEFAULT_SPLITTER = new StringSplitter.Shell();
31  
32      /**
33       * Applies pre-processing steps to the received argument before splitting, such as cleaning
34       * the string.
35       *
36       * @param arg The received argument.
37       * @return The pre-processed argument.
38       * @implNote The default is to trim the string, removing leading and trailing whitespace.
39       */
40      @SideEffectFree
41      default String preprocess( final String arg ) {
42          return arg.trim();
43      }
44  
45      /**
46       * Gives the splitter to use for splitting received strings.
47       *
48       * @return The splitter to use.
49       * @implSpec The default is {@link #DEFAULT_SPLITTER}.
50       */
51      @Pure
52      default StringSplitter.Async splitter() {
53          return DEFAULT_SPLITTER;
54      }
55  
56      /**
57       * Determines whether the parser should fail on the first element that encounters an
58       * error ({@code true}), or finish parsing all elements before reporting an
59       * {@link InvalidListException} with all encountered errors ({@code false}).
60       *
61       * @return Whether to fail as soon as any error is encountered.
62       * @implSpec The default is {@code true}.
63       */
64      default boolean failFast() {
65          return true;
66      }
67  
68      /**
69       * The minimum amount of items allowed (inclusive).
70       * 
71       * <p>Note that, while this value may technically be 0, the only way for an
72       * empty list to be received is through a message-based invocation, where 
73       * {@link #allowMerge() merging} does not occur, and the user uses quotations
74       * to provide an empty string, as in all other scenarios the argument would
75       * simply be missing. Due to that, it is strongly recommended to have an
76       * empty list as a default value (and not set the parameter as required)
77       * when setting the minimum to 0, in order to maintain consistency.
78       *
79       * @return The amount.
80       * @implSpec The default is 1. Must be between 0 and {@value Integer#MAX_VALUE}.
81       */
82      @Pure
83      @IntRange( from = 0, to = Integer.MAX_VALUE )
84      default int minItems() {
85          return 1;
86      }
87  
88      /**
89       * The maximum amount of items allowed (inclusive).
90       *
91       * @return The amount.
92       * @implSpec The default is {@value Integer#MAX_VALUE}. 
93       *           Must be between 1 and {@value Integer#MAX_VALUE}.
94       */
95      @Pure
96      @IntRange( from = 1, to = Integer.MAX_VALUE )
97      default int maxItems() {
98          return Integer.MAX_VALUE;
99      }
100 
101     /**
102      * @implSpec Defaults to {@code true}.
103      * @implNote Since lists are split by spaces by default, without trailing arg merging
104      *           they would need to always be specified with quotes, which is doubly
105      *           problematic since items within it may also need quotes. Thus, allowing
106      *           merge behavior is recommended.
107      */
108     @Override
109     default boolean allowMerge() {
110         return true;
111     }
112 
113     /**
114      * Parses an individual item in the list.
115      *
116      * @param context The invocation context.
117      * @param raw The raw item.
118      * @return The parsed item.
119      */
120     @SideEffectFree
121     Mono<T> parseItem( CommandContext context, String raw );
122 
123     /**
124      * Parses all the found elements.
125      *
126      * @param context The invocation context.
127      * @param raws The raw items.
128      * @return The parsed items.
129      */
130     private Flux<T> parseAll( final CommandContext context, final Flux<String> raws ) {
131 
132         if ( failFast() ) {
133             return raws.concatMap( i -> parseItem( context, i ) );
134         } else {
135             final var parser = TryParser.of( this::parseItem );
136             return raws.concatMap( i -> parser.parse( context, i ) )
137                     .collectList()
138                     .map( TryParser::split )
139                     .flatMap( r -> r.getT2().isEmpty() 
140                             ? Mono.just( r.getT1() ) 
141                             : Mono.error( new InvalidListException( r.getT2() ) )
142                     )
143                     .flatMapIterable( Function.identity() );
144         }
145 
146     }
147 
148     @Override
149     default Mono<List<T>> parseArgument( final CommandContext context, final String raw ) 
150             throws InvalidArgumentException {
151 
152         final var target = preprocess( raw );
153 
154         if ( target.isEmpty() ) {
155             return Mono.just( List.of() );
156         }
157 
158         return splitter().splitAsync( target )
159                 .index()
160                 .flatMap( v -> v.getT1() < maxItems() 
161                         ? Mono.just( v.getT2() ) 
162                         : Mono.error( new InvalidArgumentException( String.format( 
163                                 "Too many items, must be at most %d", maxItems()
164                         ) ) ) 
165                 )
166                 .transform( f -> parseAll( context, f ) )
167                 .collectList()
168                 .flatMap( args -> args.size() >= minItems()
169                         ? Mono.just( args )
170                         : Mono.error( new InvalidArgumentException( String.format(
171                                 "Not enough items, must be at least %d", minItems()
172                         ) ) ) 
173                 );
174 
175     }
