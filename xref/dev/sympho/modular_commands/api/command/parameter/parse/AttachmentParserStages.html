1   package dev.sympho.modular_commands.api.command.parameter.parse;
2   
3   import org.checkerframework.dataflow.qual.SideEffectFree;
4   
5   import dev.sympho.modular_commands.api.command.context.CommandContext;
6   import discord4j.core.object.entity.Attachment;
7   import reactor.core.publisher.Mono;
8   import reactor.netty.ByteBufMono;
9   
10  /**
11   * Components of an attachment argument parser.
12   *
13   * @version 1.0
14   * @since 1.0
15   */
16  public final class AttachmentParserStages {
17  
18      /** Do not instantiate. */
19      private AttachmentParserStages() {}
20  
21      /**
22       * Validates that the attachment is appropriate prior to fetching it.
23       *
24       * @since 1.0
25       * @apiNote Any validation that can be done using only the attachment object (such as
26       *          file type, image size, etc) should be done in a validator rather than 
27       *          within the {@link Parser parser} to avoid needlessly making network requests.
28       */
29      @FunctionalInterface
30      public interface Validator {
31  
32          /**
33           * Validates that the attachment is appropriate prior to fetching it.
34           *
35           * @param attachment The attachment to validate.
36           * @throws InvalidArgumentException if the attachment is not valid.
37           */
38          @SideEffectFree
39          void validate( Attachment attachment ) throws InvalidArgumentException;
40  
41      }
42  
43      /**
44       * Parses the response from fetching the attachment file into the corresponding value.
45       *
46       * @param <T> The result type.
47       * @since 1.0
48       */
49      @FunctionalInterface
50      public interface Parser<T> {
51  
52          /**
53           * Parses the response from fetching the attachment file into the corresponding value.
54           *
55           * @param context The execution context.
56           * @param attachment The attachment descriptor.
57           * @param body The response body.
58           * @return A Mono that issues the parsed argument. If the content is invalid, it may
59           *         fail with a {@link InvalidArgumentException}.
60           * @throws InvalidArgumentException if the given response is not valid.
61           */
62          @SideEffectFree
63          Mono<T> parse( CommandContext context, Attachment attachment, ByteBufMono body ) 
64                  throws InvalidArgumentException;
65  
66      }
67      
68  }
