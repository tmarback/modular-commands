3   import org.checkerframework.checker.nullness.qual.NonNull;
4   import org.checkerframework.dataflow.qual.Pure;
5   import org.checkerframework.dataflow.qual.SideEffectFree;
6   
7   import dev.sympho.modular_commands.api.command.context.CommandContext;
8   import reactor.core.publisher.Mono;
9   
10  /**
11   * Parses received arguments into their actual value.
12   *
13   * @param <T> The type of argument that is provided.
14   * @param <R> The type of raw argument that is received.
15   * @version 1.0
16   * @since 1.0
17   */
18  public sealed interface ArgumentParser<R extends @NonNull Object, T extends @NonNull Object> 
19          extends ParserFunction<R, T> permits AttachmentParser, InputParser {
20  
21      /**
22       * Validates the raw value before parsing.
23       *
24       * @param raw The raw value.
25       * @return The raw value.
26       * @throws InvalidArgumentException if the value is invalid.
27       * @apiNote This method is used to provide built-in validation in text-commands that would 
28       *          be done by Discord's servers in slash commands. Client code should place any
29       *          custom validation in {@link #parseArgument(CommandContext, Object)}.
30       */
31      @Pure
32      default R validateRaw( final R raw ) throws InvalidArgumentException {
33          return raw;
34      }
35  
36      /**
37       * Parses the given raw argument from the user into the corresponding value.
38       *
39       * @param context The execution context.
40       * @param raw The raw argument received from the user.
41       * @return A Mono that issues the parsed argument. If the raw value is invalid, it may
42       *         fail with a {@link InvalidArgumentException}.
43       * @throws InvalidArgumentException if the given argument is not a valid value.
44       */
45      @SideEffectFree
46      Mono<T> parseArgument( CommandContext context, R raw ) throws InvalidArgumentException;
47  
48      /**
49       * @apiNote Do not override.
50       */
51      @Override
52      default Mono<T> parse( CommandContext context, R raw ) throws InvalidArgumentException {
53          return parseArgument( context, validateRaw( raw ) );
54      }
55  
56  }
