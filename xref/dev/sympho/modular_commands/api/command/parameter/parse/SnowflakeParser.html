1   package dev.sympho.modular_commands.api.command.parameter.parse;

3   import org.checkerframework.checker.nullness.qual.NonNull;
4   import org.checkerframework.dataflow.qual.Pure;
5   
6   import discord4j.common.util.Snowflake;
7   
8   /**
9    * Parses snowflake-based input arguments.
10   * 
11   * <p>Note that no validation is performed on the raw value received ahead of time.
12   * The only benefit offered is that, for hybrid commands (text and slash) that need
13   * only a plain ID (instead of a proper entity object), this type can appear in the
14   * slash command as a proper type (user/role/channel) while avoiding the need to fetch
15   * the entity separately in the text command, losing only the validation.
16   *
17   * @param <T> The type of argument that is provided.
18   * @version 1.0
19   * @since 1.0
20   */
21  public non-sealed interface SnowflakeParser<T extends @NonNull Object> 
22          extends InputParser<Snowflake, T> {
23  
24      /**
25       * The ID type.
26       *
27       * @since 1.0
28       */
29      enum Type {
30  
31          /**
32           * An arbitrary snowflake ID. Note that this type is no different from using a
33           * {@link StringParser string parser} with {@link Snowflake#of(String)}.
34           */
35          ANY,
36  
37          /** A user ID. */
38          USER,
39  
40          /** A role ID. */
41          ROLE,
42          
43          /** A channel ID. */
44          CHANNEL
45  
46      }
47  
48      /**
49       * The ID type accepted.
50       * 
51       * <p>As no validation is performed, this only affects how the parameter is
52       * registered in a slash command. It has no functional difference for the handler.
53       *
54       * @return The ID type.
55       */
56      @Pure
57      Type type();
58      
59  }
