3   import java.util.Map;
4   
5   import org.checkerframework.checker.nullness.qual.NonNull;
6   import org.checkerframework.dataflow.qual.Pure;
7   import org.checkerframework.dataflow.qual.SideEffectFree;
8   
9   import dev.sympho.modular_commands.api.command.context.CommandContext;
10  import dev.sympho.modular_commands.api.exception.InvalidArgumentException;
11  import reactor.core.publisher.Mono;
12  
13  /**
14   * Specification for a parameter whose value may optionally be restricted
15   * to a set of predetermined values.
16   *
17   * @param <T> The parameter type.
18   * @version 1.0
19   * @since 1.0
20   */
21  public sealed interface ChoicesParameter<T extends @NonNull Object> extends Parameter<T> 
22          permits NumberParameter, StringParameter {
23  
24      /**
25       * The possible choices for the parameter value.
26       * 
27       * <p>If the returned map is empty, the parameter value is not
28       * restricted to a set of choices (but may have other restrictions).
29       *
30       * @return The choices.
31       */
32      @Pure
33      Map<String, T> choices();
34  
35      /**
36       * Parses the given raw argument from the user into the corresponding value.
37       *
38       * @param raw The raw argument received from the user.
39       * @return The value specified by the argument.
40       * @throws InvalidArgumentException if the given string is not a valid value.
41       */
42      @SideEffectFree
43      T parseValue( String raw ) throws InvalidArgumentException;
44  
45      @Override
46      default Mono<T> parse( CommandContext context, String raw ) throws InvalidArgumentException {
47  
48          final T value = parseValue( raw );
49          if ( choices().isEmpty() || choices().containsValue( value ) ) {
50              return Mono.just( value );
51          } else {
52              throw new InvalidArgumentException( this, 
53                      String.format( "Value '%s' is not a valid choice.", raw ) );
54          }
55  
56      }
57      
58  }
