1   package dev.sympho.modular_commands.api.command.parameter;
2   
3   import org.checkerframework.dataflow.qual.SideEffectFree;
4   
5   import dev.sympho.modular_commands.api.command.context.CommandContext;
6   import dev.sympho.modular_commands.api.exception.InvalidArgumentException;
7   import discord4j.common.util.Snowflake;
8   import discord4j.core.GatewayDiscordClient;
9   import discord4j.core.object.entity.Entity;
10  import reactor.core.publisher.Mono;
11  
12  /**
13   * Specification for a parameter that receives Discord entities.
14   *
15   * @param <T> The entity type.
16   * @version 1.0
17   * @since 1.0
18   */
19  public sealed interface EntityParameter<T extends Entity> extends Parameter<T>
20          permits MentionableParameter, MessageParameter {
21  
22      /**
23       * Parses the entity from a URL.
24       *
25       * @param client The client to get entities from.
26       * @param url The URL to parse from.
27       * @return The entity. May be empty if not found.
28       * @throws InvalidArgumentException if the URL is invalid, or if the parameter type does not
29       *                                  support URLs.
30       * @implSpec The default implementation throws an exception (no URL support).
31       */
32      @SideEffectFree
33      default Mono<T> fromUrl( final GatewayDiscordClient client, final String url ) 
34              throws InvalidArgumentException {
35          throw new InvalidArgumentException( this, "Links not supported." );
36      }
37  
38      /**
39       * Parses the entity ID from the raw argument.
40       *
41       * @param raw The received argument.
42       * @return The entity ID.
43       * @throws InvalidArgumentException if the given string is not a valid ID.
44       */
45      @SideEffectFree
46      default Snowflake parseId( final String raw ) throws InvalidArgumentException {
47  
48          try {
49              return Snowflake.of( raw );
50          } catch ( NumberFormatException e ) {
51              throw new InvalidArgumentException( this, 
52                      String.format( "Value '%s' is not a valid snowflake ID.", raw ) );
53          }
54  
55      }
56  
57      /**
58       * Retrieves the entity with the given ID.
59       *
60       * @param context The execution context.
61       * @param id The entity ID.
62       * @return The entity. May be empty if not found.
63       */
64      @SideEffectFree
65      Mono<T> getEntity( CommandContext context, Snowflake id );
66  
67      @Override
68      default Mono<T> parse( final CommandContext context, final String raw )
69              throws InvalidArgumentException {
70  
71          final Mono<T> result;
72          if ( raw.startsWith( "https://" ) ) {
73              result = fromUrl( context.getClient(), raw );
74          } else {
75              final var id = parseId( raw );
76              result = getEntity( context, id );
77          }
78          return result.onErrorMap( e -> new InvalidArgumentException( this, "Invalid.", e ) )
79                  .switchIfEmpty( 
80                      Mono.error( () -> new InvalidArgumentException( this, "Not found." ) )
81                  );
82  
83      }
84      
85  }
