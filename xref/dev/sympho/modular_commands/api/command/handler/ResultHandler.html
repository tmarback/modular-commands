1   package dev.sympho.modular_commands.api.command.handler;

3   import dev.sympho.modular_commands.api.command.context.AnyCommandContext;
4   import dev.sympho.modular_commands.api.command.result.CommandResult;
5   import reactor.core.publisher.Mono;
6   
7   /**
8    * A function that handles the result of a command.
9    *
10   * @version 1.0
11   * @since 1.0
12   */
13  public sealed interface ResultHandler
14          permits MessageResultHandler, SlashResultHandler {
15  
16      /**
17       * Handles the result of a command.
18       *
19       * @param context The context of the command.
20       * @param result The result of the command.
21       * @return A Mono that issues
22       *         {@code true} if the result should continue to be processed by subsequent handlers,
23       *         {@code false} if the result was fully handled and no longer needs to be processed.
24       *         The Mono may also be empty, in which case handling stops (as if {@code false}).
25       * @apiNote Unlike an invocation handler, this handler is <i>not</i> free to throw any 
26       *          exceptions. This is due to the fact that no further handling will be done if
27       *          a result handler encounters an error, other than logging the exception.
28       *          
29       */
30      Mono<Boolean> handle( AnyCommandContext context, CommandResult result );
31      
32  }
