1   package dev.sympho.modular_commands.api.command.handler;
2   
3   import java.util.function.BiFunction;
4   
5   import org.checkerframework.checker.nullness.qual.NonNull;
6   
7   import dev.sympho.modular_commands.api.command.context.CommandContext;
8   import dev.sympho.modular_commands.api.command.result.CommandResult;
9   import reactor.core.publisher.Mono;
10  
11  /**
12   * A function that handles the result of a command.
13   *
14   * @param <C> The context type.
15   * @version 1.0
16   * @since 1.0
17   */
18  @FunctionalInterface
19  public interface ResultHandler<C extends @NonNull CommandContext> 
20          extends BiFunction<C, CommandResult, Mono<Boolean>> {
21  
22      /**
23       * Handles the result of a command.
24       *
25       * @param context The context of the command.
26       * @param result The result of the command.
27       * @return A Mono that issues
28       *         {@code true} if the result should continue to be processed by subsequent handlers,
29       *         {@code false} if the result was fully handled and no longer needs to be processed.
30       *         The Mono may also be empty, in which case handling stops (as if {@code false}).
31       * @apiNote Unlike an invocation handler, this handler is <i>not</i> free to throw any 
32       *          exceptions. This is due to the fact that no further handling will be done if
33       *          a result handler encounters an error, other than logging the exception.
34       *          
35       */
36      Mono<Boolean> handle( C context, CommandResult result );
37  
38      /**
39       * @implSpec Delegates to {@link #handle(CommandContext, CommandResult)}. Do not override.
40       */
41      @Override
42      default Mono<Boolean> apply( final C context, final CommandResult result ) {
43          return handle( context, result );
44      }
45      
46  }
