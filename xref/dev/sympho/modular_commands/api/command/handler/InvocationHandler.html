8   /**
9    * A function that handles the execution of a command.
10   *
11   * @version 1.0
12   * @since 1.0
13   */
14  public sealed interface InvocationHandler 
15          permits MessageInvocationHandler, SlashInvocationHandler {
16  
17      // BEGIN BUGGED PARAGRAPH
18      /**
19       * Handles an invocation of the command.
20       * 
21       * <p>If this handler throws an exception, or the returned Mono results into an exception,
22       * the exception is automatically converted into an 
23       * {@link CommandErrorException exception result}.
24       *
25       * @param context The invocation context.
26       * @return The invocation result. It may result in an error (exception) if an
27       *         error occured during handling (same as if this handler threw an
28       *         exception directly).
29       * @throws Exception if an error occurred during handling.
30       * @apiNote A handler is allowed to throw checked exceptions in order to simplify 
31       *          error handling. If an exception occurs, the usual course of action would
32       *          be to return it as an {@link CommandErrorException exception result}.
33       *          Thus, it makes more sense to allow it to propagate up, then have the
34       *          centralized handler automatically wrap the exception into an error,
35       *          rather than clutter every handler with similar exception handling just
36       *          for that purpose.
37       * 
38       *          <p>If an exception requires handling to avoid leaving the system in an
39       *          inconsistent or dangerous state, it may be implemented either in the
40       *          handler itself, or in a {@link ResultHandler} that detects and handles
41       *          the resulting error.
42       * 
43       *          <p>The same logic applies to the returned Mono being allowed to result in
44       *          an error. Of course, in that case there is no way to allow checked
45       *          exceptions, so those would still need to be wrapped into an unchecked
46       *          exception or manually returned as an exception result.
47       */
48      // END BUGGED PARAGRAPH
49      Mono<CommandResult> handle( AnyCommandContext context ) throws Exception;
50      
51  }
