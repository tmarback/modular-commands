2   
3   import java.util.function.Function;
4   
5   import org.checkerframework.checker.nullness.qual.NonNull;
6   
7   import dev.sympho.modular_commands.api.command.context.CommandContext;
8   import dev.sympho.modular_commands.api.command.result.CommandErrorException;
9   import dev.sympho.modular_commands.api.command.result.CommandResult;
10  import dev.sympho.modular_commands.api.command.result.Results;
11  import dev.sympho.modular_commands.api.exception.ResultException;
12  import reactor.core.publisher.Mono;
13  
14  /**
15   * A function that handles the execution of a command.
16   *
17   * @param <C> The context type.
18   * @version 1.0
19   * @since 1.0
20   */
21  @FunctionalInterface
22  public interface InvocationHandler<C extends @NonNull CommandContext> 
23          extends Function<C, Mono<CommandResult>> {
24  
25      // BEGIN BUGGED PARAGRAPH
26      /**
27       * Handles an invocation of the command.
28       *
29       * @param context The invocation context.
30       * @return The invocation result. It may result in an error (exception) if an
31       *         error occured during handling (same as if this handler threw the
32       *         exception directly). This includes {@link ResultException}.
33       * @throws ResultException if execution finishes early. This is equivalent to returning 
34       *                         the result, but should only be used when returning directly 
35       *                         is impractical.
36       * @throws Exception if an error occurred during handling.
37       * @apiNote A handler is allowed to throw checked exceptions in order to simplify 
38       *          error handling. If an exception occurs, the usual course of action would
39       *          be to return it as an {@link CommandErrorException exception result}.
40       *          Thus, it makes more sense to allow it to propagate up, then have the
41       *          centralized handler automatically wrap the exception into an error,
42       *          rather than clutter every handler with similar exception handling just
43       *          for that purpose.
44       * 
45       *          <p>If an exception requires handling to avoid leaving the system in an
46       *          inconsistent or dangerous state, it may be implemented either in the
47       *          handler itself, or in a {@link ResultHandler} that detects and handles
48       *          the resulting error.
49       * 
50       *          <p>The same logic applies to the returned Mono being allowed to result in
51       *          an error. Of course, in that case there is no way to allow checked
52       *          exceptions, so those would still need to be wrapped into an unchecked
53       *          exception or manually returned as an exception result.
54       */
55      // END BUGGED PARAGRAPH
56      Mono<CommandResult> handle( C context ) throws ResultException, Exception;
57  
58      /**
59       * Handles an invocation of the command using {@link #handle(CommandContext)}, automatically
60       * converting any exception thrown in that method or issued in the resulting mono to an
61       * {@link CommandErrorException exception result} (with the exception of a 
62       * {@link ResultException}, which is unpacked into the contained result), and guaranteeing
63       * that the mono is not empty.
64       *
65       * @param context The invocation context.
66       * @return The invocation result. It is guaranteed to not be an error nor empty.
67       * @implSpec Do not override this.
68       */
69      @SuppressWarnings( "checkstyle:illegalcatch" )
70      default Mono<CommandResult> handleWrapped( final C context ) {
71  
72          try {
73              return handle( context )
74                      .onErrorResume( ResultException.class, e -> Mono.just( e.getResult() ) )
75                      .onErrorResume( e -> Results.exceptionMono( e ) )
76                      .defaultIfEmpty( Results.error( "No result issued!" ) );
77          } catch ( final ResultException e ) {
78              return Mono.just( e.getResult() );
79          } catch ( final Exception e ) {
80              return Results.exceptionMono( e );
81          }
82  
83      }
84  
85      /**
86       * @implSpec Delegates to {@link #handleWrapped(CommandContext)}. Do not override.
87       */
88      @Override
89      default Mono<CommandResult> apply( final C context ) {
90          return handleWrapped( context );
91      }
92      
93  }
