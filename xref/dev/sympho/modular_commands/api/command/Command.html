28  /**
29   * A command that can be invoked by a user.
30   * 
31   * <p>Irrespective of whether the command it is used with is compatible with interactions
32   * or not, all values must be compatible with the
33   * <a href="https://discord.com/developers/docs/interactions/application-commands#application-command-object">
34   * Discord API specification</a> for command parameters.
35   *
36   * @param <H> The handler type of this command. The subtype(s) of {@link Handlers} used determine
37   *            what methods of invocation are supported by this command. However, this type
38   *            parameter is used only to facilitate type checking, with actual support being
39   *            determined by the concrete type of the return value of {@link #handlers()}. See
40   *            the description of that method for details.
41   * @version 1.0
42   * @since 1.0
43   */
44  // END LONG LINES
45  @Value.Immutable
46  @Value.Style( 
47          visibility = Value.Style.ImplementationVisibility.PACKAGE, 
48          overshadowImplementation = true 
49  )
50  public interface Command<H extends @NonNull Handlers> {
51  
52      /**
53       * The scopes that a command may be defined in.
54       *
55       * @version 1.0
56       * @since 1.0
57       */
58      enum Scope { 
59  
60          /**
61           * A command that can be invoked in either guild channels or private channels.
62           */
63          GLOBAL, 
64  
65          /**
66           * A command that can only be invoked in guild channels.
67           */
68          GUILD 
69  
70      }

73  
74      /** Default for {@link #scope()} ({@link Scope#GLOBAL}). */
75      Scope DEFAULT_SCOPE = Scope.GLOBAL;
76  
77      /** Default for {@link #callable()}. */
78      boolean DEFAULT_CALLABLE = true;
79  
80      /** Default for {@link #requiredGroup()} ({@link Groups#EVERYONE}). */

82  
83      /** Default for {@link #skipGroupCheckOnInteraction()}. */
84      boolean DEFAULT_SKIP = true;
85

87      boolean DEFAULT_REQUIRE_PARENT_GROUPS = true;
88

90      boolean DEFAULT_NSFW = false;
91  
92      /** Default for {@link #repliesDefaultPrivate()}. */
93      boolean DEFAULT_PRIVATE = false;
94

96      boolean DEFAULT_DEFER = false;

99      boolean DEFAULT_INHERIT = false;

106     /** Pattern for valid slash command names in the Discord API. */
107     @Regex String NAME_REGEX = "(?U)^[-_\\p{L}\\p{N}\\p{sc=Deva}\\p{sc=Thai}]{1,32}+$";
108     /** Pattern for valid user/message command names in the Discord API. */
109     @Regex String DISPLAY_NAME_REGEX = "(?U)^[ -_\\p{L}\\p{N}\\p{sc=Deva}\\p{sc=Thai}]{1,32}+$";
110     /** Pattern for valid command descriptions in the Discord API. */
111     @Regex String DESCRIPTION_REGEX = "(?Us)^.{1,100}+$";

114      * The ID that uniquely identifies this command in the system.
115      *
116      * @return The command ID.
117      */
118     @Pure

121     /**
122      * The scope that the command is defined in.
123      *
124      * @return The command scope.
125      * @implSpec The default value is {@link #DEFAULT_SCOPE}.
126      */
127     @Pure

129     default Scope scope() {

133     /**
134      * Whether this command can be invoked by a user.
135      * 
136      * <p>A non-callable command is usually used as a parent for other commands that should
137      * not be itself invoked.
138      * 
139      * <p>Do note that slash commands cannot be invoked if they have children commands, and
140      * so in that case this parameter has no effect. This is a limitation from Discord itself.
141      *
142      * @return {@code true} if the command may be directly called by users. 
143      *         {@code false} otherwise.
144      * @implSpec The default value is {@value #DEFAULT_CALLABLE}.
145      */
146     @Pure
147     @Value.Default
148     default boolean callable() {
149         return DEFAULT_CALLABLE;
150     }
151 
152     /**
153      * The parent of the command.
154      * 
155      * <p>If the invocation returned by this is empty, this command has
156      * no parent, and is therefore a main command. Otherwise, it is a
157      * subcommand.
158      * 
159      * <p>Note that a parent can only be specified by its {@link #name() name}, not
160      * any aliases it may have (when supported). This implies that any command that
161      * may match the Invocation returned by this method through an alias is <i>not</i>
162      * considered a parent of this command.
163      *
164      * @return The parent of the command.
165      * @implSpec The default is {@link Invocation#EMPTY the empty invocation} (e.g. no parent).
166      */
167     @Pure
168     @Value.Default
169     default Invocation parent() {
170         return Invocation.EMPTY;
171     }
172 
173     /**
174      * The name of the command.
175      * 
176      * <p>This is the name used to match the command internally and for accessing
177      * text-based commands (slash and text commands). It is provided to the Discord
178      * API for slash commands, but <i>not</i> for user and message commands (for
179      * that, see {@link #displayName()}).
180      *
181      * @return The name of the command.
182      */
183     @Pure
184     @MatchesRegex( NAME_REGEX ) String name();
185 
186     /**
187      * The invocation that executes this command.
188      *
189      * @return The command invocation.
190      * @implSpec The invocation is determined by appending the {@link #name() name}
191      *           to the {@link #parent() parent}.
192      */
193     @SideEffectFree
194     @Value.NonAttribute
195     default Invocation invocation() {
196 
197         return parent().child( name() );
198 
199     }

201     /**
202      * The aliases that may also invoke the command, when invoking through a message.
203      * 
204      * <p>These aliases are only relevant to handling invocations. They may not be
205      * used to specify the command as a parent.
206      * 
207      * <p>Aliases must satisfy the same restrictions as the {@link #name() name}.
208      *
209      * @return The command aliases.
210      * @apiNote This setting is only meaningful for message-based invocations. By extension,
211      *          if the command does not support message invocations, this setting has no
212      *          effect.
213      * @implSpec The default is an empty set.
214      */
215     @Pure
216     @Value.Default
217     default Set<@MatchesRegex( NAME_REGEX ) String> aliases() {
218         return Collections.emptySet();
219     }
220 
221     /**
222      * The alias invocations that may also invoke the command.
223      *
224      * @return The command aliases as invocations.
225      * @see #aliases()
226      * @apiNote This value is only meaningful for message-based invocations. By extension,
227      *          if the command does not support message invocations, this setting has no
228      *          effect.
229      * @implSpec The invocation is determined by appending each alias to the 
230      *           {@link #parent() parent}. Do not override this.
231      */
232     @SideEffectFree
233     @Value.NonAttribute
234     default Set<Invocation> aliasInvocations() {
235 
236         return aliases().stream()
237                 .map( this.parent()::child )
238                 .collect( Collectors.toUnmodifiableSet() );
239 
240     }

262     /**
263      * The command parameters, in the order that they should be provided by the user.
264      * 
265      * <p>All parameters that are marked as {@link Parameter#required() required}
266      * <i>must</i> come <i>before</i> all parameters that aren't (optional parameters).
267      *
268      * @return The command parameters.
269      * @apiNote The restriction in parameter order is due to the fact that it would
270      *          not make any sense to have an optional parameter before a required one.
271      *          Since all parameters must be provided in sequence, that would effectively
272      *          make the optional parameter required.
273      */
274     @Pure
275     List<Parameter<?>> parameters();

277     /**
278      * The group that a user must have access for in order to invoke this command.
279      * 
280      * <p>Note that having access to a group is not necessarily the same as belonging
281      * to that group (although it is the basic threshold); the specific definition
282      * depends on the {@link AccessValidator} implementation used to verify access.
283      * 
284      * <p>For application commands, be aware that this is completely disjoint from Discord's
285      * own permission system, and so utilizing this feature may lead to user confusion due
286      * to discrepancies between the apparent configured permissions and the internal group
287      * checks. It is nonetheless left as an option for situations where the application
288      * should have control over who is allowed to run the command (instead of guild
289      * administrators), for example global debug commands that should be reserved for 
290      * the developers, and for complex permissions beyond what Discord can support
291      * (for example, subcommand-specific access restrictions). Make sure that
292      * {@link #skipGroupCheckOnInteraction() group checking is not skipped} when
293      * doing this.
294      * 
295      * <p>To completely skip group checking on invocations made through application
296      * commands (possibly saving a few function calls in such cases), use
297      * {@link #skipGroupCheckOnInteraction()}. It may also be useful in the case of 
298      * commands that support both text and interactions, where group checking may be 
299      * wanted only for the text case (where there is no built-in permission system 
300      * through Discord).
301      *
302      * @return The required group.
303      * @implSpec The default value is {@link #DEFAULT_GROUP}.
304      */
305     @Pure
306     @Value.Default
307     default Group requiredGroup() {
308         return DEFAULT_GROUP;
309     }

311     /**
312      * Whether group access checking should be skipped when this command is invoked
313      * through an interaction.
314      *
315      * @return Whether group access checking should be skipped when this command is 
316      *         invoked through an interaction.
317      * @see #requiredGroup()
318      * @apiNote This value is only meaningful for interaction-based invocations. By extension,
319      *          if the command does not support interaction invocations, this setting has no
320      *          effect.
321      * @implSpec The default value is {@value #DEFAULT_SKIP}, as application commands 
322      *           should generally avoid clashing with user-set permissions unless necessary.
323      */
324     @Pure
325     @Value.Default
326     default boolean skipGroupCheckOnInteraction() {
327         return DEFAULT_SKIP;
328     }

380     /**
381      * Whether the initial reply to the command should be deferred, indicating to the user
382      * that a reply is pending.
383      * 
384      * <p>Note that whether the deferral is publicly visible or not is defined by the
385      * {@link #repliesDefaultPrivate() default visibility of replies}. Note also that,
386      * if deferral is used, the visibility of the first reply will be set to the default
387      * to match the deferral, with the value of {@link CommandReplySpec#privately()} 
388      * being ignored if set. Further replies are not affected.
389      * 
390      * <p>If using a deferral, it is expected that a reply will be sent at some point.
391      * It is an error to set this to {@code true} and not send any replies.
392      *
393      * @return Whether this command's initial response is deferred.
394      * @apiNote The specific mechanism used for deferral may vary depending on how 
395      *          the command was invoked (message, slash command, etc).
396      * @implSpec The default value is {@value #DEFAULT_DEFER}.
397      */

399     @Value.Default

402     }

425     /**
426      * Whether the parent command should be invoked before this command is invoked.
427      * 
428      * <p>If the parent command may also define that its parent must be executed, and
429      * so on. Execution will start at the closest ancestor that sets this to false or
430      * does not have a parent (including itself).
431      * 
432      * <p>The parent(s) will be invoked with the same context object, which means that a
433      * parent may pass objects to a child through the context. It also means that the
434      * parameters to this command must be <i>compatible</i> with all executed ancestors,
435      * which means that
436      * 
437      * <ul>
438      *  <li>Any parameter that is defined in both this command and an executed parent (that
439      *      is, has the same name) must be of the same type.
440      *  <li>Any parameter that is <i>required</i> in an executed parent must be defined
441      *      in this command, and it must either be also required or have a default value.</li>
442      *  <li>The <i>order</i> of shared parameters does <i>not</i> need to be the same or 
443      *      similar in any way.</li>
444      * </ul>
445      * 
446      * <p>If the parameters of this command are not compatible with those of any executed
447      * parent as defined above, an error will be triggered at execution time.
448      * 
449      * <p>There are no compatibility requirements for ancestors that would not be executed
450      * as part of an execution of this command.
451      * 
452      * <p>Naturally, this flag has no effect on a command with no parent.
453      *
454      * @return Whether this command's parent should be executed as part of an invocation.
455      * @implSpec The default value is {@value #DEFAULT_INVOKE_PARENT}.
456      */
457     @Pure
458     @Value.Default
459     default boolean invokeParent() {
460         return DEFAULT_INVOKE_PARENT;
461     }

463     /**
464      * The handlers to use for processing an invocation of the command.
465      * 
466      * <p>Note that the subtype(s) of {@link Handlers} implemented by the returned instance 
467      * determine what methods of invocation are supported by this command. For example, if 
468      * it implements only {@link MessageHandlers}, it will only support message commands;
469      * if it implements both {@link MessageHandlers} and {@link SlashHandlers}, it will
470      * support both message commands and slash commands; and so on.
471      * 
472      * <p>Be careful that, while this interface does have a generic type parameter to constrain 
473      * the allowed return type for this method, it is used only to facilitate type checking
474      * (especially internally). The invocation support is determined by the actual type of
475      * the instance returned by this method. That is, if this method returns an object that
476      * implements both {@link MessageHandlers} and {@link SlashHandlers}, this command will
477      * support both message and slash commands, even if this instance is created as of type 
478      * {@code Command<MessageHandlers>}. There is no way around this due to the nature of
479      * type erasure.
480      *
481      * @return The handler.
482      */
483     @Pure
484     H handlers();
