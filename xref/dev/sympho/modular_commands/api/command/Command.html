6   import org.checkerframework.dataflow.qual.Pure;
7   
8   import dev.sympho.modular_commands.api.command.handler.InvocationHandler;
9   import dev.sympho.modular_commands.api.command.handler.ResultHandler;
10  import dev.sympho.modular_commands.api.command.parameter.Parameter;
11  import discord4j.rest.util.PermissionSet;

13  // BEGIN LONG LINES
14  /**
15   * A command that can be invoked by a user.
16   * 
17   * <p>Irrespective of whether the command it is used with is compatible with interactions
18   * or not, all values must be compatible with the
19   * <a href="https://discord.com/developers/docs/interactions/application-commands#application-command-object">
20   * Discord API specification</a> for command parameters.
21   *
22   * @version 1.0
23   * @since 1.0
24   */
25  // END LONG LINES
26  public sealed interface Command 
27          permits MessageCommand, SlashCommand {
28  
29      /**
30       * The scopes that a command may be defined in.
31       *
32       * @version 1.0
33       * @since 1.0
34       */
35      enum Scope { 
36  
37          /**
38           * A command that can be invoked in either guild channels or private channels.
39           */
40          GLOBAL, 
41  
42          /**
43           * A command that can only be invoked in guild channels.
44           */
45          GUILD 
46  
47      }
48  
49      /**
50       * The scope that the command is defined in.
51       *
52       * @return The command scope.
53       */
54      @Pure
55      Scope scope();
56  
57      /**
58       * Whether this command can be invoked by a user.
59       * 
60       * <p>A non-callable command is usually used as a parent for other commands that should
61       * not be itself invoked.
62       * 
63       * <p>Do note that slash commands cannot be invoked if they have children commands, and
64       * so in that case this parameter has no effect. This is a limitation from Discord itself.
65       *
66       * @return {@code true} if the command may be directly called by users. 
67       *         {@code false} otherwise.
68       */
69      @Pure
70      boolean callable();
71  
72      /**
73       * The parent of the command.
74       * 
75       * <p>If the invocation returned by this is empty, this command has
76       * no parent, and is therefore a main command. Otherwise, it is a
77       * subcommand.
78       * 
79       * <p>Note that a parent can only be specified by its {@link #name() name}, not
80       * any aliases it may have (when supported). This implies that any command that
81       * may match the Invocation returned by this method through an alias is <i>not</i>
82       * considered a parent of this command.
83       *
84       * @return The parent of the command.
85       */
86      @Pure
87      Invocation parent();
88  
89      /**
90       * The name of the command.
91       * 
92       * <p>This is the name used to match the command internally and for accessing
93       * text-based commands (slash and text commands). It is provided to the Discord
94       * API for slash commands, but <i>not</i> for user and message commands (for
95       * that, see {@link #displayName()}).
96       *
97       * @return The name of the command.
98       */
99      @Pure
100     String name();
101 
102     /**
103      * The invocation that executes this command.
104      *
105      * @return The command invocation.
106      * @implSpec The invocation is determined by appending the {@link #name() name}
107      *           to the {@link #parent() parent}.
108      */
109     @Pure
110     default Invocation invocation() {
111 
112         return parent().child( name() );
113 
114     }
115 
116     /**
117      * The display name of the command.
118      * 
119      * <p>This is displayed to the user in documentation-related functionality.
120      * For user and message commands, this is also the value provided to the
121      * Discord API as the command name instead of {@link #name()}.
122      *
123      * @return The display name of the command.
124      */
125     @Pure
126     String displayName();
127 
128     /**
129      * The description of the command.
130      *
131      * @return The description of the command.
132      */
133     @Pure
134     String description();
135 
136     /**
137      * The command parameters, in the order that they should be provided
138      * by the user.
139      * 
140      * <p>All parameters that are marked as {@link Parameter#required() required}
141      * <i>must</i> come <i>before</i> all parameters that aren't (optional parameters).
142      *
143      * @return The command parameters.
144      * @apiNote The restriction in parameter order is due to the fact that it would
145      *          not make any sense to have an optional parameter before a required one.
146      *          Since all parameters must be provided in sequence, that would effectively
147      *          make the optional parameter required.
148      */
149     @Pure
150     List<Parameter<?>> parameters();
151 
152 
153 
154     /**
155      * The built-in permissions that a user should have in order to execute the
156      * command.
157      * 
158      * <p>If the invoking channel is a private channel and the {@link #scope() scope}
159      * of the command allows being invoked there, this requirement is ignored.
160      *
161      * @return The built-in permissions required to run the command.
162      */
163     @Pure
164     PermissionSet requiredDiscordPermissions();
165 
166     /**
167      * Whether a user invoking this command must also have the permissions
168      * to invoke its parent command.
169      *
170      * @return Whether a user invoking this command must also have the permissions
171      *         to invoke the parent command.
172      * @apiNote This value is only meaningful for subcommands.
173      */
174     @Pure
175     boolean requireParentPermissions();
176 
177     /**
178      * Whether this command can only be invoked in a NSFW channel.
179      * 
180      * <p>A private channel (if the {@link #scope() scope}) of the command allows invocation
181      * in one) always satisfies this condition.
182      * 
183      * <p>Conversely, a guild channel that cannot be marked as NSFW (such as an announcement
184      * channel) never satisfies this condition.
185      *
186      * @return Whether this command can only be invoked in a NSFW channel.
187      */
188     @Pure
189     boolean nsfw();
190 
191     /**
192      * Whether this command can only be invoked by the owner of the bot.
193      *
194      * @return Whether this command can only be invoked by the owner of the bot.
195      */
196     @Pure
197     boolean botOwnerOnly();
198 
199     /**
200      * Whether this command can only be invoked by the owner of the server.
201      * 
202      * <p>If the invoking channel is a private channel and the {@link #scope() scope}
203      * of the command allows being invoked there, this requirement is ignored.
204      *
205      * @return Whether this command can only be invoked by the owner of the server.
206      */
207     @Pure
208     boolean serverOwnerOnly();
209 
210     /**
211      * Whether this command's response is sent in a way that only the invoking user
212      * can see.
213      *
214      * @return Whether this command's response is sent in a way that only the invoking user
215      *         can see.
216      * @apiNote This setting only affects the default configuration of the reply manager
217      *          provided by the execution context. The specific mechanism used for it
218      *          depends on how the command was invoked (message, slash command, etc).
219      */
220     @Pure
221     boolean privateReply();
222 
223 
224     /**
225      * Whether the command settings should be inherited from the parent command
226      * (ignoring the values provided by this command).
227      * 
228      * <p>The settings are the values provided by the following methods:
229      * 
230      * <ul>
231      *  <li>{@link #nsfw()}</li>
232      *  <li>{@link #botOwnerOnly()}</li>
233      *  <li>{@link #serverOwnerOnly()}</li>
234      *  <li>{@link #privateReply()}</li>
235      * </ul>
236      *
237      * @return Whether settings should be inherited from the parent command.
238      * @apiNote This value is only meaningful for subcommands.
239      */
240     @Pure
241     boolean inheritSettings();
242 
243     /**
244      * Whether the parent command should be invoked before this command is invoked.
245      * 
246      * <p>If the parent command may also define that its parent must be executed, and
247      * so on. Execution will start at the closest ancestor that sets this to false or
248      * does not have a parent (including itself).
249      * 
250      * <p>The parent(s) will be invoked with the same context object, which means that a
251      * parent may pass objects to a child through the context. It also means that the
252      * parameters to this command must be <i>compatible</i> with all executed ancestors,
253      * which means that
254      * 
255      * <ul>
256      *  <li>Any parameter that is defined in both this command and an executed parent (that
257      *      is, has the same name) must be of the same type.
258      *  <li>Any parameter that is <i>required</i> in an executed parent must be defined
259      *      in this command, and it must either be also required or have a default value.</li>
260      *  <li>The <i>order</i> of shared parameters does <i>not</i> need to be the same or 
261      *      similar in any way.</li>
262      * </ul>
263      * 
264      * <p>If the parameters of this command are not compatible with those of any executed
265      * parent as defined above, an error will be triggered at execution time.
266      * 
267      * <p>There are no compatibility requirements for ancestors that would not be executed
268      * as part of an execution of this command.
269      * 
270      * <p>Naturally, this flag has no effect on a command with no parent.
271      *
272      * @return Whether this command's parent should be executed as part of an invocation.
273      */
274     @Pure
275     boolean invokeParent();
276 
277     /**
278      * The handler to use for processing an invocation of the command.
279      *
280      * @return The handler.
281      * @apiNote Why require the use of a handler object instead of using a plain method, you ask?
282      *          Because the Java compiler is dumb and doesn't automatically deal with widening
283      *          parameter type overloads, which leads to a lot of boilerplate to get this handler
284      *          structure working, and I have to make the separate handler classes anyway, so I'd
285      *          rather not duplicate all of that.
286      *          <br>Making Command extend CommandHandler would also be a solution, except that
287      *          sealed classes don't work across packages without using modules. So yeah.
288      */
289     @Pure
290     InvocationHandler invocationHandler();
291 
292     /**
293      * The handlers to use for processing the result, in order.
294      * 
295      * <p>They are given priority over any handlers defined at the registry
296      * or global level.
297      *
298      * @return The result handlers for this command.
299      * @implSpec The default is to have no handlers.
300      */
301     @Pure
302     default List<? extends ResultHandler> resultHandlers() {
303         return Collections.emptyList();
304     }
305     
306 }
