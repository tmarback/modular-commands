1   package dev.sympho.modular_commands.api.command.context;
2   
3   import java.util.List;
4   import java.util.Map;
5   
6   import org.checkerframework.checker.nullness.qual.Nullable;
7   import org.checkerframework.dataflow.qual.Pure;
8   
9   import dev.sympho.bot_utils.event.MessageCreateEventContext;
10  import discord4j.core.object.entity.Member;
11  import discord4j.core.object.entity.channel.MessageChannel;
12  import reactor.core.publisher.Mono;
13  
14  /**
15   * The execution context of an command invoked through a message.
16   *
17   * @version 1.0
18   * @since 1.0
19   */
20  public interface MessageCommandContext extends CommandContext, MessageCreateEventContext {
21  
22      @Override
23      default Mono<MessageChannel> channel() {
24          return MessageCreateEventContext.super.channel();
25      }
26  
27      @Override
28      default @Nullable Member callerMember() {
29          return event().getMember().orElse( null );
30      }
31  
32      /**
33       * Retrieves the raw argument string as it was received, before being split.
34       * 
35       * <p>Does not include the command name(s) (see {@link CommandContext#invocation()} for that),
36       * only the arguments that are later used for {@link #rawArgs()}.
37       *
38       * @return The received arguments before splitting.
39       */
40      @Pure
41      String argString();
42  
43      /**
44       * Retrieves the raw arguments received before parsing, in the order that they were received.
45       *
46       * @return The raw arguments received. The returned list is unmodifiable.
47       */
48      @Pure
49      List<String> rawArgs();
50  
51      /**
52       * Retrieves the raw arguments received, keyed by the corresponding parameter name.
53       *
54       * @return The raw arguments received. The returned map is unmodifiable.
55       */
56      @Pure
57      Map<String, String> rawArgMap();
58      
59  }
