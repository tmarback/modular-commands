1   package dev.sympho.modular_commands.api.command.context;
2   
3   import org.checkerframework.dataflow.qual.Pure;
4   
5   import discord4j.common.util.Snowflake;
6   import discord4j.core.event.domain.message.MessageCreateEvent;
7   import discord4j.core.object.entity.Message;
8   
9   /**
10   * The execution context of an command invoked through a message.
11   *
12   * @version 1.0
13   * @since 1.0
14   */
15  public interface MessageCommandContext extends CommandContext {
16  
17      @Override
18      MessageCreateEvent getEvent();
19  
20      /**
21       * Retrieves the message that invoked the command.
22       *
23       * @return The invoking message.
24       */
25      @Pure
26      default Message getMessage() {
27          return getEvent().getMessage();
28      }
29  
30      /**
31       * Retrieves the ID of the message that invoked the command.
32       *
33       * @return The invoking message ID.
34       */
35      @Pure
36      default Snowflake getMessageId() {
37          return getMessage().getId();
38      }
39      
40  }
