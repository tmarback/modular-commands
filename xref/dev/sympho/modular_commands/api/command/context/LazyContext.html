3   import dev.sympho.modular_commands.execute.ResultException;
4   import reactor.core.publisher.Mono;
5   
6   /**
7    * A command context whose values (particularly arguments) are lazy-loaded (that is, are not
8    * loaded until requested by {@link #load()}).
9    * 
10   * <p>This interface is not involved in the creation of commands in any way. It is only used
11   * internally to decouple parsing/validation from instantiation.
12   *
13   * @version 1.0
14   * @since 1.0
15   */
16  public non-sealed interface LazyContext extends CommandContext {
17  
18      /**
19       * Loads internal state, making the context ready for use.
20       * 
21       * <p>Until this method is called and the returned mono is completed <i>successfully</i>,
22       * the values returned by {@link #getArgument(String, Class)} are undefined.
23       *
24       * @return A mono that completes once internal values are loaded. It may also error out
25       *         with a {@link ResultException} to terminate the invocation with a result.
26       * @throws ResultException if execution should be terminated.
27       */
28      Mono<Void> load() throws ResultException;
29      
30  }
