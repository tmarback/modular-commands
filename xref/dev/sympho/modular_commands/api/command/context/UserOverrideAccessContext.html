1   package dev.sympho.modular_commands.api.command.context;
2   
3   import org.checkerframework.checker.nullness.qual.Nullable;
4   
5   import discord4j.common.util.Snowflake;
6   import discord4j.core.GatewayDiscordClient;
7   import discord4j.core.object.entity.Guild;
8   import discord4j.core.object.entity.User;
9   import reactor.core.publisher.Mono;
10  
11  /**
12   * An access context that copies an external context but overrides the user 
13   * (and associated values).
14   *
15   * @param <C> The context type.
16   * @version 1.0
17   * @since 1.0
18   */
19  class UserOverrideAccessContext<C extends AccessContext> implements AccessContext {
20  
21      /** The original context. */
22      protected final C base;
23      /** The user to override to. */
24      protected final User user;
25  
26      /**
27       * Creates a new instance.
28       *
29       * @param base The original context.
30       * @param user The user to override to.
31       */
32      UserOverrideAccessContext( final C base, final User user ) {
33  
34          this.user = user;
35          this.base = base;
36  
37      }
38  
39      @Override
40      public GatewayDiscordClient getClient() {
41          return base.getClient();
42      }
43  
44      @Override
45      public User getUser() {
46          return user;
47      }
48  
49      @Override
50      public Mono<Guild> getGuild() {
51          return base.getGuild();
52      }
53  
54      @Override
55      public @Nullable Snowflake getGuildId() {
56          return base.getGuildId();
57      }
58  
59      @Override
60      public boolean isPrivate() {
61          return base.isPrivate();
62      }
63  
64  }
