99      /**
100      * Retrieves the user that called the command as a guild
101      * member of the given guild.
102      *
103      * @param guildId The ID of the target guild.
104      * @return The calling user as a guild member of the given guild.
105      * @implNote The default implementation will re-use the member instance
106      *           {@link #getCallerMember() provided by the event} if appropriate (that is, if the
107      *           given guild is the same that the command was invoked from) to avoid making an
108      *           API request.
109      */
110     @Pure
111     default Mono<Member> getCallerMember( final Snowflake guildId ) {
112 
113         return Objects.requireNonNullElse( getCallerMember(), getCaller() )
114                 .asMember( guildId );
115 
116     }

142     /**
143      * Retrieves one of the arguments to the command.
144      *
145      * @param <T> The type of the argument.
146      * @param name The name of the corresponding parameter.
147      * @param argumentType The type of the argument.
148      * @return The argument value, or {@code null} if the received argument is empty
149      *         (omitted by the caller or an empty parsing result) and does not have 
150      *         a default value.
151      * @throws IllegalArgumentException if there is no parameter with the given name.
152      * @throws ClassCastException if the given argument type does not match the type of the
153      *                            argument with the given name.
154      * @see #getArgument(Parameter, Class)
155      * @apiNote This method will never return {@code null} if the parameter provides a default 
156      *          value. If it is marked as required, it will return {@code null} if and only if
157      *          the parser result was empty (which implies that it will never return {@code null} 
158      *          if the parser is guaranteed to never give an empty result).
159      */
160     @Pure
161     <T extends @NonNull Object> @Nullable T getArgument( String name, Class<T> argumentType )
162             throws IllegalArgumentException, ClassCastException;
163 
164     /**
165      * Retrieves one of the arguments to the command.
166      *
167      * @param <T> The type of the argument.
168      * @param parameter The corresponding parameter.
169      * @param argumentType The type of the argument.
170      * @return The argument value, or {@code null} if the received argument is empty
171      *         (omitted by the caller or an empty parsing result) and does not have 
172      *         a default value.
173      * @throws IllegalArgumentException if there is no parameter with a matching name.
174      * @throws ClassCastException if the given argument type does not match the type of the
175      *                            argument with the given name.
176      * @see #getArgument(String, Class)
177      * @apiNote This is functionally equivalent to {@link #getArgument(String, Class)},
178      *          but allows access directly from the parameter instance and provides
179      *          compile-time type checking.
180      */
181     @Pure
182     default <T extends @NonNull Object> @Nullable T getArgument( 
183             Parameter<? extends T> parameter, Class<T> argumentType )
184             throws IllegalArgumentException, ClassCastException {
185 
186         return getArgument( parameter.name(), argumentType );
187 
188     }

190     /**
191      * Retrieves one of the arguments to the command.
192      *
193      * @param <T> The type of the argument.
194      * @param parameter The corresponding parameter.
195      * @return The argument value, or {@code null} if the received argument is empty
196      *         (omitted by the caller or an empty parsing result) and does not have 
197      *         a default value.
198      * @throws IllegalArgumentException if the given parameter is not present in the
199      *                                  invoked command.
200      * @apiNote This is functionally equivalent to {@link #getArgument(Parameter, Class)}.
201      *          However, it has a stronger requirement on the {@code parameter} argument
202      *          in that it must be the <i>same instance</i> (i.e., according to {@code ==}) 
203      *          that was used to define the parameter in the original command, instead of
204      *          just needing to match the name. This is necessary as the lack of the class
205      *          parameter means there is no other way to ensure type safety. On the other
206      *          hand, this variant makes it possible to use arguments that have their own
207      *          type parameters in a type-safe manner.
208      */
209     @Pure
210     <T extends @NonNull Object> @Nullable T getArgument( Parameter<? extends T> parameter )
211             throws IllegalArgumentException;
212 
213     /**
214      * Retrieves one of the arguments to the command expecting that it is non-null,
215      * i.e. that it either has a default value or is never empty (is required and 
216      * the parser never has an empty result).
217      *
218      * @param <T> The type of the argument.
219      * @param name The name of the corresponding parameter.
220      * @param argumentType The type of the argument.
221      * @return The argument value.
222      * @throws IllegalArgumentException if there is no parameter with the given name.
223      * @throws ClassCastException if the given argument type does not match the type of the
224      *                            argument with the given name.
225      * @throws NullPointerException if the argument was empty (not received or empty parsing 
226      *                              result) and does not have a default value.
227      * @see #requireArgument(Parameter, Class)
228      * @apiNote An NPE thrown by this method indicates either a mismatched configuration 
229      *          (code expects the parameter to be required or default but it was not
230      *          configured as such) or that the parser function unexpectedly returned
231      *          an empty result.
232      */
233     @Pure
234     default <T extends @NonNull Object> T requireArgument( String name, Class<T> argumentType )
235             throws IllegalArgumentException, ClassCastException, NullPointerException {
236 
237         return Objects.requireNonNull( getArgument( name, argumentType ) );
238 
239     }

298     /**
299      * Places a context object for subsequent handlers, optionally replacing any existing
300      * values under the same key.
301      *
302      * @param key The object key.
303      * @param obj The object to store.
304      * @param replace If {@code true}, the object will be placed unconditionally, replacing
305      *                any existing value in that key. Otherwise, it will only be placed if there
306      *                are no values with the given key.
307      * @return {@code true} if the given object was placed in the context. If {@code replace}
308      *         is {@code false} and there is already an object at the given key, returns
309      *         {@code false}.
310      * @apiNote This method is <i>not</i> thread-safe.
311      */
312     boolean setContext( String key, @Nullable Object obj, boolean replace );
313 
314     /**
315      * Unconditionally places a context object for subsequent handlers.
316      *
317      * @param key The object key.
318      * @param obj The object to store.
319      * @see #setContext(String, Object, boolean)
320      * @apiNote This method is equivalent to 
321      *          {@link #setContext(String, Object, boolean) setContext(key, obj, true)}.
322      */
323     default void setContext( String key, @Nullable Object obj ) {
324         setContext( key, obj, true );
325     }

342     /**
343      * Retrieves a non-null context object set by {@link #setContext(String, Object, boolean)}.
344      *
345      * @param <T> The type of the object.
346      * @param key The object key.
347      * @param type The object class.
348      * @return The context object.
349      * @throws IllegalArgumentException If there is no context object with the given key.
350      * @throws ClassCastException if the context object with the given key is not compatible
351      *                            with the given type (not the same or a subtype).
352      * @throws NullPointerException if the context object was {@code null}.
353      */
354     @Pure
355     default <T> T requireContext( final String key, final Class<? extends T> type )

482     /**
483      * Determines whether the given user belongs to the given group in the context of
484      * this invocation (guild and channel).
485      *
486      * @param user The user to check for.
487      * @param group The group to check for.
488      * @return A Mono that emits {@code true} if the given user belongs to the given
489      *         group under this invocation context, or {@code false} otherwise.
490      */
491     @SideEffectFree
492     default Mono<Boolean> belongs( final User user, final Group group ) {
493 
494         return group.belongs( user, this )
495                 .defaultIfEmpty( false ); // Just to be safe
496 
497     }
498 
499     /**
500      * Determines whether the given user belongs to the given group in the context of
501      * this invocation (guild and channel).
502      *
503      * @param user The ID of the user to check for.
504      * @param group The group to check for.
505      * @return A Mono that emits {@code true} if the given user belongs to the given
506      *         group under this invocation context, or {@code false} otherwise.
507      */
508     @SideEffectFree
509     default Mono<Boolean> belongs( final Snowflake user, final Group group ) {
510 
511         return getClient().getUserById( user ).flatMap( u -> belongs( u, group ) );
512 
513     }
