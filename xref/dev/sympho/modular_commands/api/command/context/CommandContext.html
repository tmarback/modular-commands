120     /**
121      * Retrieves one of the arguments to the command.
122      *
123      * @param <T> The type of the argument.
124      * @param name The name of the corresponding parameter.
125      * @param argumentType The type of the argument.
126      * @return The argument value, or {@code null} if the argument was not given by
127      *         the caller and does not have a default value.
128      * @throws IllegalArgumentException if there is no parameter with the given name.
129      * @throws ClassCastException if the given argument type does not match the type of the

131      * @see #getArgument(Parameter, Class)
132      * @apiNote This method will never return {@code null} if the parameter is marked
133      *          as required or provides a default value.

135     @Pure

137             throws IllegalArgumentException, ClassCastException;

139     /**
140      * Retrieves one of the arguments to the command.
141      *
142      * @param <T> The type of the argument.
143      * @param parameter The corresponding parameter.
144      * @param argumentType The type of the argument.
145      * @return The argument value, or {@code null} if the argument was not given by
146      *         the caller and does not have a default value.
147      * @throws IllegalArgumentException if there is no parameter with a matching name.
148      * @throws ClassCastException if the given argument type does not match the type of the
149      *                            argument with the given name.
150      * @see #getArgument(String, Class)
151      * @apiNote This is functionally equivalent to {@link #getArgument(String, Class)},
152      *          but allows access directly from the parameter instance and provides
153      *          compile-time type checking.
154      */
155     @Pure
156     default <T extends @NonNull Object> @Nullable T getArgument( 
157             Parameter<? extends T> parameter, Class<T> argumentType )
158             throws IllegalArgumentException, ClassCastException {
159 
160         return getArgument( parameter.name(), argumentType );
161 
162     }

164     /**
165      * Retrieves one of the arguments to the command.
166      *
167      * @param <T> The type of the argument.
168      * @param parameter The corresponding parameter.
169      * @return The argument value, or {@code null} if the argument was not given by
170      *         the caller and does not have a default value.
171      * @throws IllegalArgumentException if the given parameter is not present in the
172      *                                  invoked command.
173      * @apiNote This is functionally equivalent to {@link #getArgument(Parameter, Class)}.
174      *          However, it has a stronger requirement on the {@code parameter} argument
175      *          in that it must be the <i>same instance</i> (i.e., according to {@code ==}) 
176      *          that was used to define the parameter in the original command, instead of
177      *          just needing to match the name. This is necessary as the lack of the class
178      *          parameter means there is no other way to ensure type safety. On the other
179      *          hand, this variant makes it possible to use arguments that have their own
180      *          type parameters in a type-safe manner.
181      */
182     @Pure
183     <T extends @NonNull Object> @Nullable T getArgument( Parameter<? extends T> parameter )
184             throws IllegalArgumentException;
185 
186     /**
187      * Retrieves one of the arguments to the command, assuming it is non-null
188      * (so either required or with a default value).
189      *
190      * @param <T> The type of the argument.
191      * @param name The name of the corresponding parameter.
192      * @param argumentType The type of the argument.
193      * @return The argument value.
194      * @throws IllegalArgumentException if there is no parameter with the given name.
195      * @throws ClassCastException if the given argument type does not match the type of the
196      *                            argument with the given name.
197      * @throws NullPointerException if the argument was not received and does not have a
198      *                              default value.
199      * @see #requireArgument(Parameter, Class)
200      * @apiNote An NPE thrown by this method indicates a mismatched configuration 
201      *          (code expects the parameter to be required or default but it was not
202      *          configured as such).
203      */
204     @Pure
205     default <T extends @NonNull Object> T requireArgument( String name, Class<T> argumentType )
206             throws IllegalArgumentException, ClassCastException, NullPointerException {
207 
208         return Objects.requireNonNull( getArgument( name, argumentType ) );
209 
210     }

267     /**
268      * Places a context object for subsequent handlers, optionally replacing any existing
269      * values under the same key.
270      *
271      * @param key The object key.
272      * @param obj The object to store.
273      * @param replace If {@code true}, the object will be placed unconditionally, replacing
274      *                any existing value in that key. Otherwise, it will only be placed if there
275      *                are no values with the given key.
276      * @return {@code true} if the given object was placed in the context. If {@code replace}
277      *         is {@code false} and there is already an object at the given key, returns
278      *         {@code false}.
279      * @apiNote This method is <i>not</i> thread-safe.
280      */
281     boolean setContext( String key, @Nullable Object obj, boolean replace );
282 
283     /**
284      * Unconditionally places a context object for subsequent handlers.
285      *
286      * @param key The object key.
287      * @param obj The object to store.
288      * @see #setContext(String, Object, boolean)
289      * @apiNote This method is equivalent to 
290      *          {@link #setContext(String, Object, boolean) setContext(key, obj, true)}.
291      */
292     default void setContext( String key, @Nullable Object obj ) {
293         setContext( key, obj, true );
294     }

296     /**
297      * Retrieves a context object set by {@link #setContext(String, Object, boolean)}.
298      *
299      * @param <T> The type of the object.
300      * @param key The object key.
301      * @param type The object class.
302      * @return The context object.
303      * @throws IllegalArgumentException if there is no context object with the given key.
304      * @throws ClassCastException if the context object with the given key is not compatible
305      *                            with the given type (not the same or a subtype).
306      */
307     @Pure
308     <T> @Nullable T getContext( String key, Class<? extends T> type )
309             throws IllegalArgumentException, ClassCastException;
310 
311     /**
312      * Retrieves a non-null context object set by {@link #setContext(String, Object, boolean)}.
313      *
314      * @param <T> The type of the object.
315      * @param key The object key.
316      * @param type The object class.
317      * @return The context object.
318      * @throws IllegalArgumentException If there is no context object with the given key.
319      * @throws ClassCastException if the context object with the given key is not compatible
320      *                            with the given type (not the same or a subtype).
321      * @throws NullPointerException if the context object was {@code null}.
322      */
323     @Pure
324     default <T> T requireContext( final String key, final Class<? extends T> type )
325             throws IllegalArgumentException, ClassCastException, NullPointerException {
326 
327         return Objects.requireNonNull( getContext( key, type ) );

411     /**
412      * Determines whether the given user belongs to the given group in the context of
413      * this invocation (guild and channel).
414      *
415      * @param user The user to check for.
416      * @param group The group to check for.
417      * @return A Mono that emits {@code true} if the given user belongs to the given
418      *         group under this invocation context, or {@code false} otherwise.
419      */
420     @SideEffectFree
421     default Mono<Boolean> belongs( final User user, final Group group ) {
422 
423         return group.belongs( getGuild(), getChannel(), user )
424                 .defaultIfEmpty( false ); // Just to be safe
425 
426     }
427 
428     /**
429      * Determines whether the given user belongs to the given group in the context of
430      * this invocation (guild and channel).
431      *
432      * @param user The ID of the user to check for.
433      * @param group The group to check for.
434      * @return A Mono that emits {@code true} if the given user belongs to the given
435      *         group under this invocation context, or {@code false} otherwise.
436      */
437     @SideEffectFree
438     default Mono<Boolean> belongs( final Snowflake user, final Group group ) {
439 
440         return getClient().getUserById( user ).flatMap( u -> belongs( u, group ) );
441 
442     }
