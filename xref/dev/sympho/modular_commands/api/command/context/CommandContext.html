1   package dev.sympho.modular_commands.api.command.context;
2   
3   import org.checkerframework.checker.nullness.qual.Nullable;
4   import org.checkerframework.dataflow.qual.Pure;
5   
6   import dev.sympho.modular_commands.api.command.Command;
7   import dev.sympho.modular_commands.api.command.Invocation;
8   import discord4j.common.util.Snowflake;
9   import discord4j.core.event.domain.Event;
10  import discord4j.core.object.entity.Guild;
11  import discord4j.core.object.entity.Member;
12  import discord4j.core.object.entity.User;
13  import discord4j.core.object.entity.channel.MessageChannel;
14  import reactor.core.publisher.Mono;
15  
16  /**
17   * The execution context of an invoked command.
18   *
19   * @version 1.0
20   * @since 1.0
21   */
22  public sealed interface CommandContext permits LazyContext, 
23          MessageCommandContext, InteractionCommandContext {
24  
25      /**
26       * Retrieves the event that triggered the command.
27       *
28       * @return The trigger event.
29       */
30      @Pure
31      Event getEvent();
32  
33      /**
34       * Retrieves the invocation that triggered the command.
35       * 
36       * <p>This may be different from the command's declared {@link Command#invocation()}
37       * if it was invoked using an alias (when supported).
38       *
39       * @return The trigger invocation.
40       */
41      @Pure
42      Invocation getInvocation();
43  
44      /**
45       * Retrieves the user that called the command.
46       *
47       * @return The calling user.
48       */
49      @Pure
50      User getCaller();
51  
52      /**
53       * Retries the user that called the command as a guild
54       * member as provided by the triggering event, if
55       * present.
56       *
57       * @return The calling user as a guild member, or 
58       */
59      @Pure
60      @Nullable Member getCallerMember();
61  
62      /**
63       * Retrieves the channel that the command was invoked in.
64       *
65       * @return The invoking channel.
66       */
67      Mono<MessageChannel> getChannel();
68  
69      /**
70       * Retrieves the ID of the channel that the command was invoked in.
71       *
72       * @return The invoking channel's ID.
73       */
74      @Pure
75      Snowflake getChannelId();
76  
77      /**
78       * Retrieves the guild that the command was invoked in, if there
79       * is one.
80       *
81       * @return The invoking guild.
82       */
83      Mono<Guild> getGuild();
84  
85      /**
86       * Retrieves the ID of the guild that the command was invoked in, if there
87       * is one.
88       *
89       * @return The invoking guild's ID, or {@code null} if the command was
90       *         invoked in a private channel.
91       */
92      @Pure
93      @Nullable Snowflake getGuildId();
94  
95      /**
96       * Retrieves one of the arguments to the command.
97       *
98       * @param <T> The type of the argument.
99       * @param name The name of the corresponding parameter.
100      * @param argumentType The type of the argument.
101      * @return The argument value, or {@code null} if the argument was not given by
102      *         the caller and does not have a default value.
103      * @throws IllegalArgumentException if there is no parameter with the given name.
104      * @throws ClassCastException if the given argument type does not match the type of the
105      *                            argument with the given name.
106      * @apiNote This method will never return {@code null} if the parameter is marked
107      *          as required or provides a default value.
108      */
109     @Pure
110     <T> @Nullable T getArgument( String name, Class<? extends T> argumentType )
111             throws IllegalArgumentException, ClassCastException;
112 
113     /**
114      * Places a context object for subsequent handlers, optionally replacing any existing
115      * values under the same key.
116      *
117      * @param key The object key.
118      * @param obj The object to store.
119      * @param replace If {@code true}, the object will be placed unconditionally, replacing
120      *                any existing value in that key. Otherwise, it will only be placed if there
121      *                are no values with the given key.
122      * @return {@code true} if the given object was placed in the context. If {@code replace}
123      *         is {@code false} and there is already an object at the given key, returns
124      *         {@code false}.
125      * @apiNote This method is <i>not</i> thread-safe.
126      */
127     boolean setContext( String key, @Nullable Object obj, boolean replace );
128 
129     /**
130      * Unconditionally places a context object for subsequent handlers.
131      *
132      * @param key The object key.
133      * @param obj The object to store.
134      * @see #setContext(String, Object, boolean)
135      * @apiNote This method is equivalent to 
136      *          {@link #setContext(String, Object, boolean) setContext(key, obj, true)}.
137      */
138     default void setContext( String key, @Nullable Object obj ) {
139         setContext( key, obj, true );
140     }
141 
142     /**
143      * Retrieves a context object set by {@link #setContext(String, Object, boolean)}.
144      *
145      * @param <T> The type of the object.
146      * @param key The object key.
147      * @param type The object class.
148      * @return The context object.
149      * @throws IllegalArgumentException If there is no context object with the given key.
150      * @throws ClassCastException If the context object with the given key is not compatible
151      *                            with the given type (not the same or a subtype).
152      */
153     @Pure
154     <T> @Nullable T getContext( String key, Class<? extends T> type )
155             throws IllegalArgumentException, ClassCastException;
156     
157 }
