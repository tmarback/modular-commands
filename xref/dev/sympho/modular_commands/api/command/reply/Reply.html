1   package dev.sympho.modular_commands.api.command.reply;
2   
3   import org.checkerframework.dataflow.qual.Pure;
4   import org.checkerframework.dataflow.qual.SideEffectFree;
5   
6   import discord4j.common.util.Snowflake;
7   import discord4j.core.object.entity.Message;
8   import discord4j.core.spec.InteractionReplyEditSpec;
9   import discord4j.core.spec.MessageEditSpec;
10  import reactor.core.publisher.Mono;
11  
12  /**
13   * A reply made by a command.
14   *
15   * @version 1.0
16   * @since 1.0
17   */
18  public interface Reply {
19  
20      /**
21       * The index of the reply in the sequence of replies sent by the command.
22       *
23       * @return The index of the reply.
24       * @see ReplyManager#get(int)
25       */
26      @Pure
27      int index();
28  
29      /**
30       * The ID of the reply message.
31       *
32       * @return The message ID.
33       */
34      @Pure
35      Snowflake id();
36  
37      /**
38       * Retrieves the reply message.
39       *
40       * @return The reply message.
41       */
42      @SideEffectFree
43      Mono<Message> message();
44  
45      /**
46       * Edits the reply message.
47       *
48       * @param spec The edit specification.
49       * @return The edited message.
50       */
51      Mono<Message> edit( CommandReplyEditSpec spec );
52  
53      /**
54       * Edits the reply message.
55       *
56       * @return An edit builder Mono that can be configured with the target edit then subscribed
57       *         to apply the edit.
58       */
59      default CommandReplyEditMono edit() {
60  
61          return CommandReplyEditMono.of( this );
62  
63      }
64  
65      /**
66       * Edits the reply message.
67       *
68       * @param spec The edit specification.
69       * @return The edited message.
70       * @see ReplySpec#from(MessageEditSpec)
71       * @see #edit(CommandReplyEditSpec)
72       */
73      default Mono<Message> edit( final MessageEditSpec spec ) {
74  
75          return edit( ReplySpec.from( spec ) );
76  
77      }
78  
79      /**
80       * Edits the reply message.
81       *
82       * @param spec The edit specification.
83       * @return The edited message.
84       * @see ReplySpec#from(InteractionReplyEditSpec)
85       * @see #edit(CommandReplyEditSpec)
86       */
87      default Mono<Message> edit( final InteractionReplyEditSpec spec ) {
88  
89          return edit( ReplySpec.from( spec ) );
90  
91      }
92  
93      /**
94       * Deletes the reply message.
95       *
96       * @return A mono that completes when the message has been deleted.
97       */
98      Mono<Void> delete();
99      
100 }
