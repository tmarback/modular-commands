3   import org.immutables.value.Value;

5   import dev.sympho.modular_commands.utils.SpecStyle;
6   import discord4j.core.object.entity.Message;
7   import discord4j.discordjson.MetaEncodingEnabled;
8   import reactor.core.CoreSubscriber;
9   import reactor.core.publisher.Mono;
10  
11  /**
12   * Specification for editing a command reply that can be directly subscribed to to execute.
13   *
14   * @version 1.0
15   * @since 1.0
16   */
17  @SpecStyle
18  @MetaEncodingEnabled
19  @Value.Immutable( builder = false )
20  @SuppressWarnings( { "immutables:subtype", "immutables:incompat" } )
21  abstract class CommandReplyEditMonoGenerator extends Mono<Message> 
22          implements CommandReplyEditSpecGenerator {
23  
24      /**
25       * The backing reply.
26       *
27       * @return The reply.
28       */
29      abstract Reply reply();
30  
31      @Override
32      @SuppressWarnings( "argument" )
33      public void subscribe( final CoreSubscriber<? super Message> actual ) {
34          reply().edit( CommandReplyEditSpec.copyOf( this ) ).subscribe( actual );
35      }
36  
37      @Override
38      public abstract String toString();
39      
40  }
