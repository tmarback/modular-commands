1   package dev.sympho.modular_commands.api.command;
2   
3   import java.util.Collections;
4   import java.util.List;
5   import java.util.Set;
6   import java.util.stream.Collectors;
7   
8   import org.checkerframework.dataflow.qual.Pure;
9   
10  import dev.sympho.modular_commands.api.command.handler.MessageInvocationHandler;
11  import dev.sympho.modular_commands.api.command.handler.MessageResultHandler;
12  
13  /**
14   * A message-based command.
15   *
16   * @version 1.0
17   * @since 1.0
18   */
19  public non-sealed interface MessageCommand extends Command {
20  
21      /**
22       * The aliases that may also invoke the command.
23       * 
24       * <p>These aliases are only relevant to handling invocations. They may not be
25       * used to specify the command as a parent.
26       * 
27       * <p>Aliases must satisfy the same restrictions as the {@link #name() name}.
28       *
29       * @return The command aliases.
30       */
31      @Pure
32      Set<String> aliases();
33  
34      /**
35       * The alias invocations that may also invoke the command.
36       *
37       * @return The command aliases as invocations.
38       * @see #aliases()
39       * @implSpec The invocation is determined by appending each alias to the 
40       *           {@link #parent() parent}.
41       */
42      @Pure
43      default Set<Invocation> aliasInvocations() {
44  
45          return aliases().stream()
46                  .map( this.parent()::child )
47                  .collect( Collectors.toUnmodifiableSet() );
48  
49      }
50  
51      @Override
52      MessageInvocationHandler invocationHandler();
53  
54      @Override
55      default List<? extends MessageResultHandler> resultHandlers() {
56          return Collections.emptyList();
57      }
58      
59  }
