1   package dev.sympho.modular_commands.api.permission;
2   
3   import org.checkerframework.dataflow.qual.SideEffectFree;
4   
5   import dev.sympho.modular_commands.api.command.context.AccessContext;
6   import dev.sympho.modular_commands.api.command.context.ChannelAccessContext;
7   import discord4j.core.object.entity.User;
8   import reactor.core.publisher.Mono;
9   
10  /**
11   * A group whose membership is defined at the guild level or higher (i.e. the channel is 
12   * not taken into account).
13   *
14   * @version 1.0
15   * @since 1.0
16   * @apiNote Despite the name, implementations of this interface do not need to take the guild
17   *          into account for determining membership; they only cannot use the channel.
18   */
19  @FunctionalInterface
20  public interface GuildGroup extends Group {
21  
22      /**
23       * Determines whether the user in the given context belongs to this group.
24       *
25       * @param context The access context being checked.
26       * @return A Mono that issues {@code true} if the user belongs to this group under
27       *         the context, or {@code false} otherwise.
28       * @implSpec Implementations should prefer using the monos provided by the context directly
29       *           instead of obtaining them through other means (such as through the client), as
30       *           the context is optimized when possible to reuse existing instances instead of
31       *           issuing new API calls.
32       */
33      @SideEffectFree
34      Mono<Boolean> belongs( AccessContext context );
35  
36      /**
37       * @implSpec Delegates to {@link #belongs(AccessContext)}.
38       */
39      @Override
40      default Mono<Boolean> belongs( final ChannelAccessContext context ) {
41          return belongs( ( AccessContext ) context );
42      }
43  
44      /**
45       * Determines whether the given user belongs to this group in the context of the 
46       * given guild and channel.
47       *
48       * @param user The user to check for.
49       * @param context The access context being checked.
50       * @return A Mono that issues {@code true} if the user belongs to this group under
51       *         the given context, or {@code false} otherwise.
52       * @apiNote This is equivalent to using {@link AccessContext#asUser(User)} and
53       *          delegating to {@link #belongs(AccessContext)}. The user in the given
54       *          context is ignored.
55       */
56      @SideEffectFree
57      default Mono<Boolean> belongs( final User user, final AccessContext context ) {
58  
59          return belongs( context.asUser( user ) );
60  
61      }
62      
63  }
