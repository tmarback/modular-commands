1   package dev.sympho.modular_commands.api.permission;
2   
3   import org.checkerframework.dataflow.qual.SideEffectFree;
4   
5   import discord4j.core.object.entity.Guild;
6   import discord4j.core.object.entity.User;
7   import discord4j.core.object.entity.channel.MessageChannel;
8   import reactor.core.publisher.Mono;
9   
10  /**
11   * Specifies a group that users may belong to.
12   * 
13   * <p>Note that group membership is not static, but rather may change between guilds
14   * and/or channels. This allows the definition of flexible permission groups that refer 
15   * to different groups of people in different servers/channels.
16   *
17   * @version 1.0
18   * @since 1.0
19   */
20  @FunctionalInterface
21  public interface Group {
22      
23      /**
24       * Determines whether the given user belongs to this group in the context of the 
25       * given guild and channel.
26       *
27       * @param guild The guild in which group membership is being checked for. It may be
28       *              empty in the case of a private channel.
29       * @param channel The channel in which group membership is being checked for.
30       * @param user The user to check group membership for.
31       * @return A Mono that issues {@code true} if the user belongs to this group under
32       *         the given guild and channel, or {@code false} otherwise.
33       */
34      @SideEffectFree
35      Mono<Boolean> belongs( Mono<Guild> guild, Mono<MessageChannel> channel, User user );
36  
37  }
