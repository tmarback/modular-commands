1   package dev.sympho.modular_commands.api.registry;
2   
3   import java.util.Collection;
4   
5   import org.checkerframework.checker.nullness.qual.Nullable;
6   import org.checkerframework.dataflow.qual.Pure;
7   import org.checkerframework.dataflow.qual.SideEffectFree;
8   
9   import dev.sympho.modular_commands.api.command.Command;
10  import dev.sympho.modular_commands.api.command.Invocation;
11  import dev.sympho.modular_commands.api.command.handler.Handlers;
12  
13  /**
14   * A registry that stores commands that may be invoked by users.
15   *
16   * @version 1.0
17   * @since 1.0
18   */
19  public interface Registry {
20  
21      /**
22       * Retrieves the best command known to this registry that has the given parent and name,
23       * and is compatible with the given type.
24       *
25       * @param <H> The handler type that must be supported.
26       * @param invocation The invocation of the command.
27       * @param type The handler type.
28       * @return The best matching command known to this registry, or {@code null} if none were
29       *         found.
30       */
31      @Pure
32      <H extends Handlers> @Nullable Command<? extends H> findCommand( 
33              Invocation invocation, Class<H> type );
34  
35      /**
36       * Retrieves all commands known to this registry that are compatible with the given type.
37       *
38       * @param <H> The handler type that must be supported.
39       * @param type The handler type.
40       * @return The compatible commands known to this registry.
41       */
42      @SideEffectFree
43      <H extends Handlers> Collection<Command<? extends H>> getCommands( Class<H> type );
44  
45      /**
46       * Retrieves the command with the given ID that is registered to this registry.
47       *
48       * @param id The ID of the command to retrieve.
49       * @return The command that was registered with the given ID, or {@code null} if there
50       *         is no such command in this registry.
51       */
52      @Pure
53      @Nullable Command<?> getCommand( String id );
54  
55      /**
56       * Registers a command into this registry.
57       *
58       * @param id The ID to register the command under. This is different from the command name
59       *           and display name in that it is not used for command handling or any user-facing
60       *           functionality, but rather only as a way to internally identify the command.
61       * @param command The command to register.
62       * @return {@code true} if the command was registered successfully. {@code false} if there
63       *         is already a command in this registry with the given ID, or if there is already
64       *         a command known to this registry with the given signature and this registry does
65       *         not allow overriding. If {@code false}, the registration failed and the call had
66       *         no effect.
67       * @throws IllegalArgumentException if the given command has an invalid configuration.
68       */
69      boolean registerCommand( String id, Command<?> command ) throws IllegalArgumentException;
70  
71      /**
72       * Removes a command from this registry that was registered with the given ID.
73       *
74       * @param id The ID of the command to remove.
75       * @return The removed command, or {@code null} if there was no command in this registry
76       *         registered with the given ID.
77       */
78      @Nullable Command<?> removeCommand( String id );
79      
80  }
