57      /**
58       * Sends a message on the channel that the command was received.
59       *
60       * @param context The command context.
61       * @param message The message to send.
62       * @return A success result once the message is sent.
63       */
64      private static Mono<CommandResult> sendMessage( final CommandContext context,
65              final String message ) {
66  
67          LOGGER.info( "Sending message: {}", message );
68          final var replies = context.replies();
69          return replies.add( message )
70                      .then( replies.add().withPrivately( true ).withContent( "Test1" ) )
71                      .then( replies.add().withPrivately( false ).withContent( "Test2" ) )
72                      .then( replies.add().withPrivately( true ).withContent( "Test3" ) )
73                      .then( replies.add().withPrivately( false ).withContent( "Test4" ) )
74                      .thenReturn( Results.ok() );
75  
76      }

300     /**
301      * Creates a command for showing meter values.
302      *
303      * @param registry The meter registry to use.
304      * @return The command.
305      */
306     private static Command<?> metersCommand( final MeterRegistry registry ) {
307 
308         final var counter = registry.counter( "meter.shows", "test", "true", "prod", "false" );
309 
310         return Command.builder()
311                 .name( "meters" )
312                 .displayName( "Show meters" )
313                 .description( "Shows current meter values" )
314                 .handlers( Handlers.text( ctx -> {
315 
316                     counter.increment();
317 
318                     if ( registry.getMeters().isEmpty() ) {
319                         return Results.failureMono( "No meters!" );
320                     }
321 
322                     return Flux.fromIterable( registry.getMeters() )
323                             .map( meter -> {
324 
325                                 final var name = meter.getId().getName();
326                                 LOGGER.debug( "Found meter {}", name );
327 
328                                 final var tags = meter.getId().getTags().stream()
329                                         .map( tag -> "`%s`: `%s`".formatted( 
330                                                 tag.getKey(), tag.getValue() 
331                                         ) )
332                                         .toList();
333 
334                                 final var measurements = StreamSupport.stream( 
335                                         meter.measure().spliterator(), false )
336                                         .map( Measurement::toString )
337                                         .toList();
338 
339                                 return """
340                                         Meter: `%s`
341 
342                                         Tags:
343                                         %s
344 
345                                         Measurements:
346                                         %s
347                                         """.formatted(
348                                                 name,
349                                                 String.join( "\n", tags ),
350                                                 String.join( "\n", measurements )
351                                         );
352 
353                             } )
354                             .flatMap( ctx::reply )
355                             .then( Results.okMono() );
356 
357                 } ) )
358                 .id( "meters" )
359                 .build();
360         
361     }

363     /**
364      * Main runner.
365      *
366      * @param args Command line arguments.
367      */
368     public static void main( final String[] args ) {
369 
370         // Enable context propagation so tracing works
371         Hooks.enableAutomaticContextPropagation();
372 
373         final String token = args[0];
374 
375         final MeterRegistry meters = new SimpleMeterRegistry();
376         final ObservationRegistry observations = ObservationRegistry.create();
377 
378         final var observationHandler = new ObservationHandler<Observation.Context>() {
379 
380             @Override
381             public void onStart( final Observation.Context context ) {
382                 LOGGER.trace( "OBSERVATION START: {}", context.getName() );
383                 LOGGER.trace( "Tags: {}", context.getAllKeyValues() );
384             }
385 
386             @Override
387             public void onError( final Observation.Context context ) {
388                 LOGGER.error( "OBSERVATION ERROR: " + context.getName(), context.getError() );
389             }
390 
391             @Override
392             public void onEvent( final Observation.Event event, 
393                     final Observation.Context context ) {
394                 LOGGER.trace( "OBSERVATION EVENT: {} - {}", context.getName(), event );
395             }
396 
397             @Override
398             public void onStop( final Observation.Context context ) {
399                 LOGGER.trace( "OBSERVATION STOP: {}", context.getName() );
400                 LOGGER.trace( "Tags: {}", context.getAllKeyValues() );
401             }
402 
403             @Override
404             public boolean supportsContext( final Context context ) {
405 
406                 return true;
407 
408             }
409 
410         };
411         observations.observationConfig().observationHandler( observationHandler );
412 
413         final PrefixProvider prefix = new StaticPrefix( "t!" );
414         final Registry registry = Registries.simpleRegistry();
415 
416         registry.registerCommands( 
417                 pingCommand(),
418                 parrotCommand(),
419                 tweetCommand(),
420                 listCommand(),
421 
422                 adminCommand(),
423                 modCommand(),
424                 serverOwnerCommand(),
425                 botOwnerCommand(),
426 
427                 fileTextCommand(),
428 
429                 metersCommand( meters )
430         );
431 
432         final AliasProvider aliases = AliasProvider.of(
433                 Map.entry( Invocation.of( "pong" ), Invocation.of( "ping" ) )
434         );
435         
436         DiscordClient.create( token )
437             .withGateway( client -> {
438 
439                 final List<CommandExecutor> executors = List.of(
440                     new MessageCommandExecutor( 
441                             client, registry, AccessManager.basic(), 
442                             meters, observations, 
443                             prefix, aliases 
444                     )
445                 );
446 
447                 executors.forEach( CommandExecutor::start );
448                 
449                 return Mono.empty();
450             } ).block();
451 
452     }
